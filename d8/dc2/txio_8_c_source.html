149  if (32 <= c && c < 127)
150  fp << c;

152  uint z = c & 7;

154  uint x = (c >> 6) & 7;
155  fp << "\\" << uchar(x + '0') << uchar(y + '0') << uchar(z + '0');
156  }
157 }

241 // This reads the next byte.
242 // We assume buf[wptr]=0. We leave ptr unchanged in case it is >= wptr
243 // As a consequence, at(ptr) is valid after the call
244 auto Buffer::next_utf8_byte() -> uchar {
245  uchar x = at(ptr);
246  if ((x >> 6) == 2) {
247  ++ptr;
248  return x & 63;
249  }
250  utf8_error(false);
251  if (ptr < wptr) ++ptr;
252  return 0;
253 }

255 // This returns the number of bytes in a UTF8 character
256 // given the first byte. Returns 0 in case of error
257 // Note: max Unicode value is 10FFFF. this is represented by F48FBFBF
258 // if the first 3 bits are set, y is 0, 1, 2 3 or 4 plus 16.
259 auto io_ns::how_many_bytes(uchar C) -> int {
260  if (C < 128) return 1; // ascii
261  if ((C >> 5) == 6) return 2; // 2 bytes
262  if ((C >> 5) != 7) return 0; // cannot start with 10
263  uint y = C & 31;
264  if ((y & 16) == 0) return 3;
265  if (y == 16 || y == 17 || y == 18 || y == 19 || y == 20) return 4;
266  return 0; // overflow
267 }

269 // Creates a Unicode character from the bytes A, B, C and D.
270 // Return 0 if invalid. Return 0 if overflow
271 auto io_ns::make_utf8char(uchar A, uchar B, uchar C, uchar D) -> codepoint {
272  int n = io_ns::how_many_bytes(A);
273  if (n == 0) return codepoint(0);
274  if (n == 1) return codepoint(A);
275  if (n == 2) return codepoint(((A & 31) << 6) + (B & 63));
276  if (n == 3) return codepoint((C & 63) + ((B & 63) << 6) + ((A & 15) << 12));
277  return codepoint((D & 63) + ((C & 63) << 6) + ((B & 63) << 12) + ((A & 7) << 18));
278 }

280 // Returns 0 at end of line or error
281 // may set local_error
282 auto Buffer::next_utf8_char_aux() -> codepoint {
283  uchar c = next_char();
284  if (c == 0) return codepoint(0);
285  int n = io_ns::how_many_bytes(c);
286  if (n == 0) {
287  utf8_error(true);
288  the_converter.line_is_ascii = false;
289  return codepoint(0);
290  }
291  if (n == 1) return codepoint(c);
292  the_converter.line_is_ascii = false;
293  if (n == 2) {
294  uint x = next_utf8_byte();
295  return codepoint(((c & 31) << 6) + x);
296  }
297  if (n == 3) {
298  uint z = next_utf8_byte();
299  uint x = next_utf8_byte();
300  return codepoint(x + (z << 6) + ((c & 15) << 12));
301  }
302  uint z = next_utf8_byte();
303  uint y = next_utf8_byte();
304  uint x = next_utf8_byte();
305  return codepoint((x) + (y << 6) + (z << 12) + ((c & 7) << 18));
306 }

309 // This complains if the character is greater than 1FFFF
310 auto Buffer::next_utf8_char() -> codepoint {
311  the_converter.local_error = false;

313  if (the_converter.local_error) return codepoint(0);

321 // If the buffer contains a unique character, return it
322 // Otherwise return 0. No error signaled
323 auto Buffer::unique_character() const -> codepoint {
324  uchar c = at(0);
325  int n = io_ns::how_many_bytes(c);
326  if (n == 0) return codepoint(0);
327  if (n != size()) return codepoint(0);
328  if (n == 1) return codepoint(c);
329  if (n == 2) return io_ns::make_utf8char(at(0), at(1), 0, 0);
330  if (n == 3) return io_ns::make_utf8char(at(0), at(1), at(2), 0);
331  if (n == 4) return io_ns::make_utf8char(at(0), at(1), at(2), at(3));
332  return codepoint(0);
333 }

335 // This converts a line in UTF8 format. Returns true if no conversion needed
336 // Otherwise, the result is in utf8_out.
337 auto Buffer::convert_line0(int wc) -> bool {
338  Buffer &res = utf8_out;
339  res.reset();
340  reset_ptr();
341  codepoint c;
342  for (;;) {
343  if (wc == 0)
344  c = next_utf8_char();
345  else {
346  uchar C = next_char();
347  if (wc == 1)
348  c = codepoint(C);
349  else
350  c = custom_table[wc - 2][C];
351  if (!(c.is_ascii() && c == C)) the_converter.line_is_ascii = false;
352  }
353  if (c.non_null()) res.push_back(c);
354  if (at_eol()) break;
355  }
356  return the_converter.line_is_ascii;
357 }

416 auto io_ns::find_encoding(String cl) -> int {
417  if (strstr(cl, "-*-") != nullptr) {
418  if (strstr(cl, "coding: utf-8") != nullptr) return 0;
419  if (strstr(cl, "coding: utf8") != nullptr) return 0;
420  if (strstr(cl, "coding: latin1") != nullptr) return 1;
421  if (strstr(cl, "coding: iso-8859-1") != nullptr) return 1;
422  }
423  if (strstr(cl, "iso-8859-1") != nullptr) return 1;
424  if (strstr(cl, "utf8-encoded") != nullptr) return 0;
425  if (strstr(cl, "%&TEX encoding = UTF-8") != nullptr) return 1;
426  String s = strstr(cl, "tralics-encoding:");
427  if (s == nullptr) return -1;
428  if (!is_digit(s[17])) return -1;
429  int k = s[17] - '0';
430  if (is_digit(s[18])) { k = 10 * k + s[18] - '0'; }
431  if (k < max_encoding) return k;
432  return -1;
433 }

475 // This reads the file named x.
476 // If spec is 0, we are reading the config file.
477 // If 2 it's a tex file, and the file is converted later.
478 // If 3, no conversion done
479 // If 4, its is the main file, log not yet open.
480 void tralics_ns::read_a_file(LinePtr &L, const std::string &x, int spec) {
481  L.reset(x);
482  if (main_ns::use_pool(L)) return;
483  auto *fp = new std::fstream(x.c_str(), std::ios::in);
484  if (fp == nullptr) return;
485  std::string old_name = the_converter.cur_file_name;
486  the_converter.cur_file_name = x;
487  Buffer B;
488  int wc = the_main->get_input_encoding();
489  if (spec == 4) wc = -1;
490  bool converted = spec < 2;
491  L.set_encoding(the_main->get_input_encoding());
492  int co_try = spec == 3 ? 0 : 20;
493  for (;;) {
494  int c = fp->get();
495  bool emit = false;
496  if (c == '\r') { // pc or mac ?
497  emit = true;
498  c = fp->peek();
499  if (c == '\n') fp->ignore();
500  } else if (c == '\n')
501  emit = true;
502  else if (c == EOF) {
503  if (!B.empty()) emit = true;
504  fp->close();
505  delete fp;
506  the_converter.cur_file_name = old_name;
507  } else
508  B.push_back(c);
509  if (emit) {
510  if (spec == 0) // special case of config file
511  emit = B.push_back_newline_spec();
512  else
513  B.push_back_newline();
514  if (co_try != 0) {
515  co_try--;
516  int k = io_ns::find_encoding(B.c_str());
517  if (k >= 0) {
518  wc = k;
519  L.set_encoding(k);
520  co_try = 0;
521  the_log << lg_start_io << "Input encoding number " << k << " detected at line " << L.get_cur_line() + 1 << " of file "
522  << x << lg_end;
523  }
524  }
525  if (converted) B.convert_line(L.get_cur_line() + 1, wc);
526  if (emit)
527  L.insert(B.to_string(), converted);
528  else
529  L.incr_cur_line();
530  B.reset();
531  }
532  if (c == EOF) break;
533  }
534 }

536 // If a line ends with \, we take the next line, and append it to this one
537 void LinePtr::normalise_final_cr() {
538  auto C = value.begin();
539  auto E = value.end();
540  if (C == E) return;
541  Clines *prev = nullptr;
542  Clines *cur = nullptr;
543  for (;;) {
544  if (C == E) return;
545  cur = &*C;
546  ++C;
547  const std::string &s = cur->get_chars();
548  int n = s.size();
549  bool special = (n >= 2 && s[n - 2] == '\\' && s[n - 1] == '\n');
550  std::string normal = s;
551  if (special) normal = std::string(s, 0, n - 2);
552  if (prev != nullptr) {
553  prev->set_chars(prev->get_chars() + normal);
554  cur->set_chars("\n");
555  }
556  if (special) {
557  if (prev == nullptr) {
558  prev = cur;
559  prev->set_chars(normal);
560  }
561  } else
562  prev = nullptr;
563  }
564 }

568 // This puts x into the buffer in utf8 form
569 void Buffer::push_back(codepoint c) {
570  uint x = c.value;
571  if (x < 128) {
572  push_back(uchar(x));
573  return;
574  }
575  int x1 = x >> 18;
576  int x2 = (x >> 12) & 63;
577  int x3 = (x >> 6) & 63;
578  int x4 = x & 63;
579  if (x1 > 0 || x2 >= 16) {
580  push_back(uchar(x1 + 128 + 64 + 32 + 16));
581  push_back(uchar(x2 + 128));
582  push_back(uchar(x3 + 128));
583  push_back(uchar(x4 + 128));
584  } else if (x2 > 0 || x3 >= 32) {
585  push_back(uchar(x2 + 128 + 64 + 32));
586  push_back(uchar(x3 + 128));
587  push_back(uchar(x4 + 128));
588  } else {
589  push_back(uchar(x3 + 128 + 64));
590  push_back(uchar(x4 + 128));
591  }
592 }

608 // Converts in uppercase hex. If uni is ptrue, produces U+00AB
609 void Buffer::push_back16(uint n, bool uni) {
610  static uint dig[9];
611  int k = 0;
612  for (;;) { // construct the list of digits
613  dig[k] = n % 16;
614  n = n / 16;
615  k++;
616  if (n == 0) break;
617  }
618  if (uni) {
619  push_back("U+");
620  if (k < 4) push_back('0'); // print at least 4 digit
621  if (k < 3) push_back('0');
622  if (k < 2) push_back('0');
623  if (k < 1) push_back('0');
624  }
625  while (k > 0) { // print the list
626  k--;
627  push_back_Hex(dig[k]);
628  }
629 }

631 // Converts number in lower hex form (assumes n>=16, so k>=2)
632 // if hat is true, inserts hats before
633 void Buffer::push_back16l(bool hat, uint n) {
