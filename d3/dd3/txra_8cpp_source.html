31 // If c has no \footnote, copies it in aux;
32 // else copies only what is before, what is after is appended to d
33 void ra_ns::fnhack(TokenList &c, TokenList &d, TokenList &aux) {
34  Hashtab &H = the_parser.hash_table;
35  int bl = 0;
36  while (!c.empty()) {
37  Token x = c.front();
38  c.pop_front();
39  token_ns::check_brace(x, bl);
40  if (x == H.footnote_token && bl == 0) {
41  if (!d.empty()) {
42  d.push_front(H.space_token);
43  d.push_front(H.comma_token);
44  }
45  d.splice(d.begin(), c);
46  } else {
47  aux.push_back(x);
48  }
49  }
50 }
