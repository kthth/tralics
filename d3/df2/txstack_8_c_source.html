159 void Xml::add_last_string(const Buffer &B) {
160  if (B.empty()) return;

162  aux.reset();

243 // Returns true if frame is s, or argument, and next frame is s.
244 // ignores a font change
245 auto Stack::is_frame2(name_positions S) const -> bool {
246  Istring s = the_names[S];
247  int k = Table.size() - 1;
248  while (Table[k].frame.spec_empty()) k--;
249  if (Table[k].frame == s) return true;
250  if (Table[k].frame == the_names[cst_fonts]) {
251  if (k < 1) return false;
252  k--;
253  }
254  if (Table[k].frame == s) return true;
255  if (Table[k].frame != the_names[cst_argument]) return false;
256  if (k < 1) return false;
257  k--;
258  while (Table[k].frame.spec_empty()) k--;
259  if (Table[k].frame == s) return true;

375 // This pops an element, top-stack should be a.
376 void Stack::pop(Istring a) {
377  trace_pop(false);
378  while (Table.back().frame.spec_empty()) { // ignore dummy frames
379  the_parser.cur_font.not_on_stack();
380  Table.pop_back();
381  }
382  if (a != Table.back().frame) {
383  err_buf << bf_reset << "Error in pop; stack holds " << Table.back().frame << "; trying to pop " << a;
384  the_parser.signal_error();
385  return;
386  }
387  cur_mode = Table.back().md;
388  cur_lid = Table.back().uid;
389  Table.pop_back();
390  if (Table.empty()) {
391  err_buf << bf_reset << "Error in pop; stack empty; trying to pop " << a;
392  the_parser.signal_error();
393  Istring S = the_names[cst_document];
394  ipush(S, new Xml(S, nullptr)); // stack should never be empty
395  }
396 }

426 // Adds font info when required
427 void Stack::check_font() {
428  while (Table.back().frame.spec_empty()) Table.pop_back();
429  bool w = the_main->pack_font_elt();
430  name_positions s;
431  if (w) {
432  Buffer &aux = the_main->SH.shbuf(); // write directly in the buffer.
433  aux.reset();
434  bool nonempty = false;
435  s = the_parser.cur_font.size_change();
436  if (s != 0U) {
437  aux << the_names[s].c_str();
438  nonempty = true;
439  }
440  s = the_parser.cur_font.shape_change();
441  if (s != 0U) {
442  if (nonempty) aux.push_back(",");
443  aux << the_names[s].c_str();
444  nonempty = true;
445  }
446  s = the_parser.cur_font.family_change();
447  if (s != 0U) {
448  if (nonempty) aux.push_back(",");
449  aux << the_names[s].c_str();
450  nonempty = true;
451  }
452  s = the_parser.cur_font.series_change();
453  if (s != 0U) {
454  if (nonempty) aux.push_back(",");
455  aux << the_names[s].c_str();
456  nonempty = true;
457  }
458  if (nonempty) {
459  auto a = Istring(the_main->SH.hash_find());
460  Xml * res = new Xml(cst_hi, nullptr);
461  AttList &W = res->get_id().get_att();
462  W.push_back(the_names[np_rend], a);
463  W.push_back(the_names[cst_flaghi], Istring(1));
464  push(Istring(2), res);
465  }
466  } else {
467  s = the_parser.cur_font.size_change();
468  if (s != 0U) fonts0(s);
469  s = the_parser.cur_font.shape_change();
470  if (s != 0U) fonts0(s);
471  s = the_parser.cur_font.family_change();
472  if (s != 0U) fonts0(s);
473  s = the_parser.cur_font.series_change();
474  if (s != 0U) fonts0(s);
475  }
476  Istring c = the_parser.cur_font.get_color();
477  if (!(c.empty() || c.null())) {
478  Xml * res = new Xml(cst_hi, nullptr);
479  AttList &W = res->get_id().get_att();
480  W.push_back(np_color, c);
481  W.push_back(the_names[cst_flaghi], Istring(1));
482  push(Istring(2), res);
483  }
484  the_parser.cur_font.is_on_stack();
485 }

536 // This finds the cell row and table ID currently on the stack.
537 void Stack::find_cid_rid_tid(Xid &cid, Xid &rid, Xid &tid) {
538  int k = Table.size() - 1;
539  while (Table[k].frame.spec_empty()) k--;
540  if (Table[k].frame == the_names[np_cell]) {
541  cid = Table[k].obj->get_id();
542  k--;
543  }
544  while (Table[k].frame.spec_empty()) k--;
545  if (Table[k].frame == the_names[np_row]) {
546  rid = Table[k].obj->get_id();
547  k--;
548  }
549  while (Table[k].frame.spec_empty()) k--;
550  if (Table[k].frame == the_names[np_tabular] || Table[k].frame == the_names[np_tabular_star]) {
551  tid = Table[k].obj->get_id();
552  k--;
553  }
554 }

556 auto Stack::find_ctrid(subtypes m) -> int {
557  int k = Table.size() - 1;
558  while (k >= 0) {
559  Istring frame = Table[k].frame;
560  Xml * obj = Table[k].obj;
561  k--;
562  if (obj == nullptr) continue;
563  if (frame.spec_empty()) continue;
564  if (m == xmlcurrow_code && frame == the_names[np_row]) return obj->get_id().value;
565  if (m == xmlcurcell_code && frame == the_names[np_cell]) return obj->get_id().value;
566  if (m == xmlcurarray_code && (frame == the_names[np_tabular] || frame == the_names[np_tabular_star])) return obj->get_id().value;
567  }
568  return 0;
569 }
