21 struct Equivalent : public CmdChr {

23 
24  [[nodiscard]] auto must_push(int l) const -> bool { return level != l && l > 1; }
25  void reset() {

33  setnl(b);
34  if (!is_undef()) level = 1;

37  void set(CmdChr c, int lvl) {

40  }
