30 struct Equivalent : public CmdChr {
31  int level{};
32 
33  [[nodiscard]] auto must_push(int l) const -> bool { return level != l && l > 1; }
34  void reset() {
35  CmdChr::reset();
36  level = 0;
37  }
38 
39  void setnl(CmdChr c) { CmdChr::operator=(c); }
40 
41  void special_prim(CmdChr b) {
42  setnl(b);
43  if (!is_undef()) level = 1;
44  }
45 
46  void set(CmdChr c, int lvl) {
47  setnl(c);
48  level = lvl;
49  }
50 
51  void primitive(CmdChr b) { set(b, 1); }
52 };
