24 // Defined EQTB, the table of equivalents
25 // EQTB for commands. No Ctor(?) but reset() is called for every object.
26 // The values is a (symcodes, subtype) pair
27 class Equivalent {
28  int level; // level at which this is defined
29  CmdChr value; // the modifier
30 public:
31  [[nodiscard]] auto get_level() const -> int { return level; }
32  [[nodiscard]] auto get_cmdchr() const -> CmdChr { return value; }
33  [[nodiscard]] auto get_cmd() const -> symcodes { return value.get_cmd(); }
34  [[nodiscard]] auto get_chr() const -> subtypes { return value.get_chr(); }
35  [[nodiscard]] auto must_push(int l) const -> bool { return level != l && l > level_one; }
36  void reset() {
37  level = level_zero;
38  value.reset();
39  }
40  void special_prim(CmdChr b) {
41  value = b;
42  if (!b.is_undef()) level = level_one;
43  }
44  void primitive(CmdChr b) {
45  value = b;
46  level = level_one;
47  }
48  void set(CmdChr c, int lvl) {
49  value = c;
50  level = lvl;
51  }
52  void setnl(CmdChr c) { value = c; }
53  [[nodiscard]] auto is_undefined() const -> bool { return value.is_undef(); }
54  [[nodiscard]] auto is_user() const -> bool { return value.is_user(); }
55  [[nodiscard]] auto is_undef_or_relax() const -> bool { return value.is_undef_or_relax(); }
56 };
