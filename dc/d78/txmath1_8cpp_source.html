77 // Consider the expression \int_0^\infty f(x+y) dx = |z|b
78 // This will be translated to a list of 13 elements, the first one being
79 // \int_0^\infty. This element is marked big, and software like Amaya
80 // show the bars and parentheses with the same height as the integral.
81 // We want to add some <mrow>s in order to avoid this.
82 
83 // The next function will return a list of the form
84 // 0b 2l 4B 6r 9R 10m 12m 13b.
85 // Here b means big (the integral, and there is a special marker at the end),
86 // l r m stand for small_l, small_r and small_m (the opening parenthesis,
87 // the closing parenthesis, and the bar that can be open or close).
88 // the letters B and R stand for binary and relation (the plus and the equal)
89 
90 // Our problem is to add some <mrow> so that the small elements (l,r,m)
91 // will not change size in presence of a big element. The next function returns
92 // an empty list if there is no problem (no small or no big).
93 // Returns an empty list in the case $(\frac12)$.

95 void Math::find_paren0(MathP &aux) const {
96  int unbalance = 0;
97  bool seen_big = false;
98  bool inner_big = false;
99  bool seen_small = false;
100  int i = 0;
101  auto L = value.begin();
102  auto E = value.end();
103  int nb_pairs = 0;
104  while (L != E) { // create the list of indices
105  if (L->get_cmd() == math_xml_cmd) {
106  math_types t = L->get_xmltype();
107  bool keep = false;
108  switch (t) {
109  case mt_flag_small_l:
110  nb_pairs++;
111  unbalance++;
112  keep = true;
113  seen_small = true;
114  break;
115  case mt_flag_small_r:
116  unbalance--;
117  keep = true;
118  seen_small = true;
119  break;
120  case mt_flag_small_m:
121  keep = true;
122  seen_small = true;
123  break;
124  case mt_flag_bin: keep = true; break;
125  case mt_flag_rel: keep = true; break;
126  case mt_flag_big:
127  if (unbalance <= 0) inner_big = true;
128  seen_big = true;
129  keep = true;
130  break;
131  default: break;
132  }
133  if (keep) aux.push_back(MathPAux(i, t));
134  }
135  ++L;
136  i++;
137  }
138  if (seen_big && nb_pairs > 1) inner_big = true;
139  if (!inner_big || !seen_small) { // only small objects, or no parens
140  aux.clear();
141  } else
142  aux.push_back(MathPAux(i, mt_flag_big)); // add the end marker.
143 }

146 // Separators are big, rel or bin. We say that a formula is acceptable if
147 // it has either no small, or at least two small. For instance f(a+b)=c
148 // is acceptable if we split at =, not if we split at +. Something like

152 // The next function returns true if the formula is acceptable if we split
153 // at: rel and bin if w is true, rel if w is false.
154 auto MathP::analyse1(bool w) const -> bool {
155  int nb_small = 0;
156  auto C = value.begin();
157  auto E = value.end();
158  while (C != E) {
159  math_types t = C->get_type();
160  ++C;
161  if (t == mt_flag_small_m || t == mt_flag_small_l || t == mt_flag_small_r) nb_small++;
162  if (t == mt_flag_rel || (w && t == mt_flag_bin)) {
163  if (nb_small == 1) return false;
164  nb_small = 0;
165  }
166  }
167  return nb_small != 1;
168 }

292  auto B = value.begin();

294  int level = 0, p = -1;
295  bool allow_mid = false;

318 // The next function takes a math list, and returns a list of positions
319 // where to insert the <mrow> and </mrow>. In the case where find_paren0
320 // returns an empty list, there is nothing to do. Otherwise, the list is
321 // split into sublists, with the big as delimitors.
322 // For each sublist, if there is no small, nothing has to be done.
323 // After that, remove_binrel is called. In the example, we start with
324 // 0b 2l 4B 6r 9R 10m 12m 13b.
325 // The first sublist is empty, and the second will be
326 // start=0, 2l 6r 9R 10m 12m 13B
327 // (the start=0 means that the list starts after 0, the final 13B
328 // indicates that the list ends before the 13).
329 // Retval is false if rec, true otherwise
330 auto Math::find_parens(MathQList &res, bool verbose) const -> bool {
331  MathP aux;
332  find_paren0(aux);
333  if (aux.empty()) return true;
334  if (verbose) *(the_log.fp) << "MF: After find paren0 " << aux << "\n";
335  if (aux.find_paren_rec(res)) {
336  if (verbose) *(the_log.fp) << "MF: rec " << res << "\n";
337  return false;
338  }
339  res.clear();
340  int start = -1;
341  int end = -1;
342  while (!aux.empty()) {
343  if (aux.find_paren_matched1()) {
344  aux.find_paren_matched2(res);
345  if (verbose) the_log << "MF: matched " << aux_buffer << "\n";
346  return true;
347  }
348  MathP content = aux.find_big(end);
349  if (!content.has_small()) { // no small, look at next big
350  } else {
351  content.remove_binrel(); // remove useless bin/rel in content
352  content.push_back(MathPAux(end, mt_flag_rel)); // add end marker
353  if (verbose) *(the_log.fp) << "MF: sublist start=" << start << ' ' << content << "\n";
354  content.find_paren2(start, res, verbose);
355  }
356  start = end;
357  int seen_d1, seen_d2;
358  if (aux.is_lbr2(seen_d1, seen_d2)) {
359  if (verbose) *(the_log.fp) << "MF: LBR " << seen_d1 << ' ' << seen_d2 << "\n";
360  res.push_back(MathQ(seen_d1, seen_d2));
361  return true;
362  }
363  }
364  return true;
365 }

409 void MathP::find_paren1(int start1, int end1, MathQList &res, bool verbose) {
410  MathQList t;
411  bool state = false;
412  bool failed = false;
413  int start_pos = -1;
414  if (verbose) *(the_log.fp) << "MF: Find paren1 (" << start1 << ", " << end1 << ") " << *this << "\n";
415  while (!empty()) {
416  int i;
417  math_types k;
418  value.front().get_both(i, k);
419  bool is_small = value.front().is_small();
420  value.pop_front();
421  if (!is_small) continue;
422  if (!state) { // no opening seen
423  if (k != mt_flag_small_l) {
424  failed = true;
425  clear();
426  } else {
427  start_pos = i;
428  state = true;
429  }
430  } else { // we have seen an opening
431  if (k == mt_flag_small_l) {
432  failed = true;
433  clear();
434  }
435  if (k == mt_flag_small_r) {
436  state = false;
437  t.push_back(MathQ(start_pos, i));
438  if (verbose) *(the_log.fp) << "MF: OK " << start_pos << ' ' << i << ' ' << '\n';
439  }
440  }
441  }
442  if (t.empty()) failed = true;
443  if (failed || state) {
444  t.clear();
445  t.push_back(MathQ(start1, end1));
446  if (verbose) *(the_log.fp) << "MF: BB " << start1 << ' ' << end1 << '\n';
447  }
448  res.splice(res.end(), t);
449 }

468 // final is true for the final pass: calls remove_prefix when pushing in
469 auto Math::add_fence(bool final, MathF &M) -> bool {
470  int i = 0; // current index in the list
471  M.reset(); // state is now true
472  M.change_state();
473  bool ret_val = false;
474  bool after_dummy = false;
475  while (!empty()) {
476  math_types cur_type = front().get_xmltype();
477  if (cur_type == mt_flag_big) {
478  ret_val = true;
479  if (M.in_mrow()) M.make_t_big();
480  }
481  if (after_dummy) {
482  after_dummy = false;
483  Xml *xval = front().remove_prefix();
484  if ((xval != nullptr) && !xval->is_xmlc() &&
485  (xval->has_name(cst_msup) || xval->has_name(cst_msub) || xval->has_name(cst_msubsup))) {
486  M.pop_last(xval);
487  }
488  }
489  if (cur_type == mt_flag_dummy && (final || M.in_mrow())) after_dummy = true;
490  if (M.is_next_change(i)) {
491  Xml *xval = front().remove_prefix();
492  if (cur_type == mt_flag_dummy) {
493  if (M.in_mrow()) xval = nullptr;
494  // elsestd::cout<< "bad dummy\n"; no error ??
495  }
496  if (xval != nullptr) M.push_in_t(xval);
497  M.handle_t();
498  M.change_state();
499  if (M.is_next_change(i)) {
500  M.handle_t();
501  M.change_state();
502  }
503  } else if (cur_type == mt_flag_dummy) {
504  if (!(final || M.in_mrow())) M.push_in_res(front());
505  } else if (!M.in_mrow()) {
506  if (final)
507  M.push_in_res(front().remove_prefix());
508  else
509  M.push_in_res(front());
510  } else
511  M.push_in_t(front().remove_prefix());
512  pop_front();
513  ++i;
514  }
515  M.finish(value);
516  return ret_val;
517 }
