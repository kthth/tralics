112 // This computes the current date (first thing done after printing the banner)
113 // stores the date in the table of equivalents, computes the default year
114 // for the RA.
115 void Parser::boot_time() {
116  time_t xx;
117  time(&xx);
118  struct tm *x = localtime(&xx);
119  int sec = x->tm_sec;
120  int min = x->tm_min;
121  int hour = x->tm_hour;
122  int the_time = min + hour * 60;
123  int year = x->tm_year + 1900;
124  int month = x->tm_mon + 1;
125  int day = x->tm_mday;
126  eqtb_int_table[time_code].set_val(the_time);
127  eqtb_int_table[day_code].set_val(day);
128  eqtb_int_table[month_code].set_val(month);
129  eqtb_int_table[year_code].set_val(year);
130  std::srand(sec + 60 * (min + 60 * (hour + 24 * (day + 31 * month))));
131  Buffer b;
132  b << year << '/' << twodig(month) << month << '/' << twodig(day) << day;
133  std::string short_date = b.to_string();
134  b << bf_reset << short_date << ' ' << twodig(hour) << hour << ':' << twodig(min) << min << ':' << twodig(sec) << sec;
135  TokenList today_tokens = b.str_toks(nlt_space);
136  new_prim("today", today_tokens);
137  b << bf_reset << "Start compilation: " << short_date << ' ' << twodig(hour) << hour << ':' << twodig(min) << min << ':' << twodig(sec)
138  << sec << "\n";
139  the_main->set_start_date(b.to_string());
140  the_main->set_short_date(short_date);
141  // Default year for the raweb. Until April its last year
142  if (month <= 4) year--;
143  the_parser.set_ra_year(year);
144 }
