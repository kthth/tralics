61 // Main idea. The TeX file has commands like \cite , \nocite, which produce
62 // a CitationItem (new or old). There are stored in citation_table. They have
63 // a unique id, a Bid, and \cite creates a REF element with a reference to this
64 // the element having this bid as id.
65 // Such an element can be produced in the TeX source; we call it a solver
66 // of this bid.
67 // At the end we must construct a solver for each unsolved one. We use
68 // a function like dump_bibtex, this creates all_entries, a table of
69 // bibentry objects, one for each unsolved entry.
70 // we read some bibliography data bases, this fills the bibentry objects
71 // (maybe others are added). They are sorted, converted to TeX code
72 // and the result is translated.

157 // You can say \cite{foo,bar}. This is the same as
158 // \cite@one{}{foo}{}\cite@one{}{foo}{}, empty lists are replaced by the
159 // optional arguments. The `prenote' applies only to the first entry,
160 // and the `type' to all entries. If you do not like like, do not use commas.
161 // The natbib package proposes a postnote, that applies only to the last.
162 // We construct here a token list, and push it back.

164 // The command is fully handled in the \nocite case. The `prenote' is ignored.
165 // For the special entry `*', the type is ignored too.
166 void Parser::T_cite(subtypes sw) {
167  Token T = cur_tok;
168  bool is_natbib = sw == natcite_code;
169  if (sw == natcite_e_code) {
170  flush_buffer();
171  the_stack.pop(np_natcit);
172  return;
173  }
174  if (is_natbib) {
175  leave_v_mode();
176  the_stack.push1(np_natcit);
177  }
178  TokenList res;
179  TokenList prenote;
180  auto type = Istring("");
181  if (is_natbib) {
182  Istring x = nT_optarg_nopar();
183  if (!x.empty()) the_stack.add_att_to_last(the_names[np_cite_type], x);
184  read_optarg(res);
185  if (!res.empty()) res.push_back(hash_table.space_token);
186  res.push_front(hash_table.locate("NAT@open"));
187  }
188  cur_tok = T;
189  T_cite(sw, prenote, type); // reads optional arguments
190  type = normalise_for_bib(type);
191  if (sw == footcite_code) res.push_back(hash_table.footcite_pre_token);
192  Token sep = sw == footcite_code ? hash_table.footcite_sep_token
193  : sw == natcite_code ? hash_table.locate("NAT@sep") : hash_table.cite_punct_token;
194  cur_tok = T;
195  String List = fetch_name0_nopar();
196  Splitter S(List);
197  int n = 0;
198  Token my_cmd = is_natbib ? hash_table.citesimple_token : hash_table.citeone_token;
199  for (;;) {
200  if (S.at_end()) break;
201  String cur = S.get_next();
202  if (cur[0] == 0) continue;
203  if (sw == nocite_code) {
204  if (strcmp(cur, "*") == 0)
205  the_bibliography.set_nocite();
206  else
207  the_bibliography.find_citation_item(type, Istring(cur), true);
208  } else {
209  if (n > 0) res.push_back(sep);
210  TokenList tmp;
211  res.push_back(my_cmd);
212  if (!is_natbib) {
213  tmp = token_ns::string_to_list(type.c_str(), true);
214  res.splice(res.end(), tmp);
215  }
216  tmp = token_ns::string_to_list(cur, true);
217  res.splice(res.end(), tmp);
218  if (!is_natbib) {
219  res.push_back(hash_table.OB_token);
220  if (n == 0) res.splice(res.end(), prenote);
221  res.push_back(hash_table.CB_token);
222  }
223  n++;
224  }
225  }
226  if (is_natbib) {
227  if (!prenote.empty()) res.push_back(hash_table.locate("NAT@cmt"));
228  res.splice(res.end(), prenote);
229  res.push_back(hash_table.locate("NAT@close"));
230  res.push_back(hash_table.end_natcite_token);
231  }
232  if (tracing_commands()) the_log << lg_start << T << "->" << res << "." << lg_end;
233  back_input(res);
234 }

236 // \cite@one{foot}{Knuth}{p25}.
237 // Translates to <cit><ref target="xx">p25</ref></cit>
238 // and \cite@simple{Knuth} gives <ref target="xx"/>
239 void Parser::T_cite_one() {
240  bool is_simple = cur_cmd_chr.chr != 0;
241  Token T = cur_tok;
242  flush_buffer();
243  Istring type = is_simple ? Istring("") : Istring(fetch_name0_nopar());
244  cur_tok = T;
245  auto ref = Istring(fetch_name0_nopar());
246  Xml *arg = is_simple ? nullptr : xT_arg_nopar();
247  // signal error after argument parsing
248  if (bbl.is_too_late()) {
249  parse_error("Citation after loading biblio?");
250  return;
251  }
252  Xml *res = make_cit_ref(type, ref);
253  if (is_simple) {
254  flush_buffer();
255  the_stack.add_last(res);
256  return;
257  }
258  leave_v_mode();
259  TokenList L = get_mac_value(hash_table.cite_type_token);
260  auto xtype = Istring(fetch_name1(L));
261  L = get_mac_value(hash_table.cite_prenote_token);
262  auto prenote = Istring(fetch_name1(L));
263  res->add_tmp(arg);
264  the_stack.add_last(new Xml(np_cit, res));
265  if (!type.empty()) the_stack.add_att_to_last(the_names[np_rend], type);
266  if (!xtype.empty()) the_stack.add_att_to_last(the_names[np_cite_type], xtype);
267  if (!prenote.empty()) the_stack.add_att_to_last(the_names[np_prenote], prenote);
268 }
