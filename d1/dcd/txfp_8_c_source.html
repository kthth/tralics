318 // Computes the product XY. Both numbers are considered as a 12B number
319 // (where B=1000).. The product gives a 24B number.
320 void FpNum::mul(FpNum X, FpNum Y) {
321  bool xs = X.sign == Y.sign;
322  Digit x[12], y[12], z[24];
323  for (unsigned int &i : z) i = 0;
324  X.mul_split(x);
325  Y.mul_split(y);
326  int xmax, xmin, ymax, ymin;
327  set_xmax(x, xmin, xmax);
328  set_xmax(y, ymin, ymax);
329  for (int i = xmin; i < xmax; i++)
330  for (int j = ymin; j < ymax; j++) z[i + j + 1] += x[i] * y[j];
331  prop_carry(z);
332  finish_mul(xs, z);
333 }

410 // Assume that the buffer holds : sign digits dot digits
411 // Creates a number with it. Returns true if overflow
412 auto FpNum::create(Buffer &B) -> bool {
413  B.reset_ptr();
414  char c = B.next_char();
415  bool retval = false;
416  sign = c != '-';
417  for (;;) { // skip initial zeroes
418  if (B.head() == '0')
419  B.advance();
420  else
421  break;
422  }
423  auto k = B.ptr; // index of first non-zero digit.
424  for (;;) {
425  if (B.head() != '.')
426  B.advance();
427  else
428  break;
429  }
430  auto n = B.ptr - k; // number of chars before dot
431  B.ptr = k;
432  if (n > 18) retval = true;
433  data[0] = n >= 9 ? B.horner(n - 9) : 0;
434  data[1] = B.horner(n);
435  for (;;) {
436  if (B.next_char() == '.') break; // ignore possible unused chars
437  }
438  data[2] = B.reverse_horner();
439  data[3] = B.reverse_horner();
440  correct_sign();
441  return retval;
442 }
