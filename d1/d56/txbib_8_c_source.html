62 // Main idea. The TeX file has commands like \cite , \nocite, which produce
63 // a CitationItem (new or old). There are stored in citation_table. They have
64 // a unique id, a Bid, and \cite creates a REF element with a reference to this
65 // the element having this bid as id.
66 // Such an element can be produced in the TeX source; we call it a solver
67 // of this bid.
68 // At the end we must construct a solver for each unsolved one. We use
69 // a function like dump_bibtex, this creates all_entries, a table of
70 // bibentry objects, one for each unsolved entry.
71 // we read some bibliography data bases, this fills the bibentry objects
72 // (maybe others are added). They are sorted, converted to TeX code
73 // and the result is translated.

130 // In prenote we put `p. 25' or nothing, in type one of year, foot, refer, bar

132 void Parser::T_cite(subtypes sw, TokenList &prenote, Istring &type) {
133  if (sw == footcite_code) {
134  read_optarg_nopar(prenote);
135  type = the_names[cst_foot];
136  } else if (sw == yearcite_code) {
137  read_optarg_nopar(prenote);

139  } else if (sw == refercite_code) {
140  read_optarg_nopar(prenote);

142  } else if (sw == nocite_code) {
143  type = Istring(fetch_name_opt());
144  } else if (sw == natcite_code) {
145  read_optarg_nopar(prenote); // is really the post note

150  if (read_optarg_nopar(prenote))
151  type = Istring(fetch_name1(L1));
152  else
153  prenote = L1;
154  }

158 // You can say \cite{foo,bar}. This is the same as
159 // \cite@one{}{foo}{}\cite@one{}{foo}{}, empty lists are replaced by the
160 // optional arguments. The `prenote' applies only to the first entry,
161 // and the `type' to all entries. If you do not like like, do not use commas.
162 // The natbib package proposes a postnote, that applies only to the last.
163 // We construct here a token list, and push it back.

165 // The command is fully handled in the \nocite case. The `prenote' is ignored.
166 // For the special entry `*', the type is ignored too.
167 void Parser::T_cite(subtypes sw) {
168  Token T = cur_tok;
169  bool is_natbib = sw == natcite_code;
170  if (sw == natcite_e_code) {
171  flush_buffer();
172  the_stack.pop(np_natcit);
173  return;
174  }
175  if (is_natbib) {
176  leave_v_mode();
177  the_stack.push1(np_natcit);
178  }
179  TokenList res;
180  TokenList prenote;
181  auto type = Istring("");
182  if (is_natbib) {
183  Istring x = nT_optarg_nopar();
184  if (!x.empty()) the_stack.add_att_to_last(the_names[np_cite_type], x);
185  read_optarg(res);
186  if (!res.empty()) res.push_back(hash_table.space_token);
187  res.push_front(hash_table.locate("NAT@open"));
188  }
189  cur_tok = T;
190  T_cite(sw, prenote, type); // reads optional arguments
191  type = normalise_for_bib(type);
192  if (sw == footcite_code) res.push_back(hash_table.footcite_pre_token);
193  Token sep = sw == footcite_code ? hash_table.footcite_sep_token
194  : sw == natcite_code ? hash_table.locate("NAT@sep") : hash_table.cite_punct_token;
195  cur_tok = T;
196  String List = fetch_name0_nopar();
197  Splitter S(List);
198  int n = 0;
199  Token my_cmd = is_natbib ? hash_table.citesimple_token : hash_table.citeone_token;
200  for (;;) {
201  if (S.at_end()) break;
202  String cur = S.get_next();
203  if (cur[0] == 0) continue;
204  if (sw == nocite_code) {
205  if (strcmp(cur, "*") == 0)
206  the_bibliography.set_nocite();
207  else
208  the_bibliography.find_citation_item(type, Istring(cur), true);
209  } else {
210  if (n > 0) res.push_back(sep);
211  TokenList tmp;
212  res.push_back(my_cmd);
213  if (!is_natbib) {
214  tmp = token_ns::string_to_list(type.c_str(), true);
215  res.splice(res.end(), tmp);
216  }
217  tmp = token_ns::string_to_list(cur, true);
218  res.splice(res.end(), tmp);
219  if (!is_natbib) {
220  res.push_back(hash_table.OB_token);
221  if (n == 0) res.splice(res.end(), prenote);
222  res.push_back(hash_table.CB_token);
223  }
224  n++;
225  }
226  }
227  if (is_natbib) {
228  if (!prenote.empty()) res.push_back(hash_table.locate("NAT@cmt"));
229  res.splice(res.end(), prenote);
230  res.push_back(hash_table.locate("NAT@close"));
231  res.push_back(hash_table.end_natcite_token);
232  }
233  if (tracing_commands()) the_log << lg_start << T << "->" << res << "." << lg_end;
234  back_input(res);
235 }

237 // \cite@one{foot}{Knuth}{p25}.
238 // Translates to <cit><ref target="xx">p25</ref></cit>
239 // and \cite@simple{Knuth} gives <ref target="xx"/>
240 void Parser::T_cite_one() {
241  bool is_simple = cur_cmd_chr.get_chr() != 0;
242  Token T = cur_tok;
243  flush_buffer();
244  Istring type = is_simple ? Istring("") : Istring(fetch_name0_nopar());
245  cur_tok = T;
246  auto ref = Istring(fetch_name0_nopar());
247  Xml *arg = is_simple ? nullptr : xT_arg_nopar();
248  // signal error after argument parsing
249  if (bbl.is_too_late()) {
250  parse_error("Citation after loading biblio?");
251  return;
252  }
253  Xml *res = make_cit_ref(type, ref);
254  if (is_simple) {
255  flush_buffer();
256  the_stack.add_last(res);
257  return;
258  }
259  leave_v_mode();
260  TokenList L = get_mac_value(hash_table.cite_type_token);
261  auto xtype = Istring(fetch_name1(L));
262  L = get_mac_value(hash_table.cite_prenote_token);
263  auto prenote = Istring(fetch_name1(L));
264  res->add_tmp(arg);
265  the_stack.add_last(new Xml(np_cit, res));
266  if (!type.empty()) the_stack.add_att_to_last(the_names[np_rend], type);
267  if (!xtype.empty()) the_stack.add_att_to_last(the_names[np_cite_type], xtype);
268  if (!prenote.empty()) the_stack.add_att_to_last(the_names[np_prenote], prenote);
269 }
