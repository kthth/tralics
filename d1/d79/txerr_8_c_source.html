38 // The error mechanism is as follows: we put the error message in a buffer
39 // then use the following function to signal it.

41 void Parser::signal_error() {
42  int line = get_cur_line();
43  std::string file = get_cur_filename();
44  main_ns::nb_errs++;
45  flush_buffer();
46  the_log << lg_start;
47  main_ns::log_and_tty << "Error signaled at line " << line;
48  if (!file.empty()) main_ns::log_and_tty << " of file " << file;
49  main_ns::log_and_tty << ":\n";
50  main_ns::log_and_tty << local_buf;
51  main_ns::log_and_tty << ".\n";
52  the_log << lg_flush;
53  std::cout.flush();
54  if (main_ns::nb_errs >= 5000) {
55  main_ns::log_and_tty << "Translation aborted: Too many errors.\n";
56  main_ns::log_and_tty.finish(main_ns::nb_errs);
57  tralics_ns::close_file(main_ns::log_and_tty.L.fp);
58  exit(1);
59  }
60 }

62 // If T is non-null and the option -noxmlerror is not given,
63 // then an error element with value s name T is inserted in the XML result
64 
65 void Parser::signal_error(Token T, String s) {
66  signal_error();
67  if (main_ns::no_xml_error) return;
68  if (T.is_null()) return;
69  auto str = Istring(s);
70  the_stack.add_newid0(np_error);
71  Istring cmd = local_buf.convert_for_xml_err(T);
72  the_stack.add_att_to_last(np_letter_c, str);
73  the_stack.add_att_to_last(np_letter_l, cur_line_to_istring());
74  the_stack.add_att_to_last(np_letter_n, cmd);
75 }

77 // identical to ostream& operator<<(ostream&fp, const TokenList& L)
78 void err_ns::convert_to_string(const TokenList &L) {
79  Buffer & B = local_buf;
80  static Buffer buffer_for_log;
81  auto C = L.begin();
82  auto E = L.end();
83  while (C != E) {
84  buffer_for_log.reset();
85  if (buffer_for_log.push_back(*C)) buffer_for_log << ' ';
86  B << buffer_for_log.c_str();
87  ++C;
88  }
89 }
