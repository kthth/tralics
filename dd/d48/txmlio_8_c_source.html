84 // This reads the XML file, without conversion
85 auto XmlIO::init(const std::string &name) -> bool {
86  for (auto &i : Type) i = xt_invalid;
87  for (int i = '0'; i <= '9'; i++) Type[i] = xt_digit;
88  for (int i = 'a'; i <= 'z'; i++) Type[i] = xt_letter;
89  for (int i = 'A'; i <= 'Z'; i++) Type[i] = xt_letter;
90  Type[uchar(' ')] = xt_space;
91  Type[uchar('\r')] = xt_space;
92  Type[uchar('\n')] = xt_space;
93  Type[uchar('\t')] = xt_space;
94  Type[uchar('_')] = xt_ok_first;
95  Type[uchar(':')] = xt_ok_first;
96  Type[uchar('$')] = xt_other;
97  Type[uchar('-')] = xt_other;
98  tralics_ns::read_a_file(lines, name, 3);
99  enc = 0;
100  input_line_pos = cur_line_len = 0;
101  nb_cond = 0;
102  eof_ok = false;
103  if (lines.is_empty()) {
104  cur_line = 0;
105  error("Unable to read the XML input file");
106  return true;
107  }
108  return false;
109 }

111 // This parses the whole XML file
112 void XmlIO::run() {
113  cur_xml = new Xml(Istring("root"), nullptr);
114  cur_stack.push_back(cur_xml);
115  B.reset();
116  eof_ok = true;
117  for (;;) {
118  codepoint c = next_char();
119  // We should handle &foo; as well, but we cannot change it
120  if (c != '<')
121  B.push_back(c);
122  else {
123  eof_ok = false;
124  parse_lt();
125  eof_ok = true;
126  }
127  }
128 }

130 // This fetches a line from the file, true at eof
131 void XmlIO::next_line() {
132  line_buffer.reset();
133  int n = lines.get_next_cv(line_buffer, enc);
134  if (n == -1) {
135  if (eof_ok && cur_stack.size() == 1)
136  ;
137  else
138  error("Unexpected EOF");
139  throw EndOfData();
140  }
141  cur_line = n;
142  line_buffer.extract_chars(input_line);
143  input_line_pos = 0;
144  cur_line_len = input_line.size();
145 }

186 // Scans a Name with a simplified syntax
187 void XmlIO::scan_name() {
188  B.reset();
189  codepoint x = XmlIO::peek_char();
190  // x should be Letter underscore colon
191  B.push_back(x);
192  skip_char();
193  for (;;) {
194  codepoint xx = XmlIO::peek_char();
195  if (xx.is_ascii()) {
196  x_type w = Type[xx.value];
197  if (w == xt_space || w == xt_invalid) return;
198  }
199  skip_char();
200  B.push_back(xx);
201  }

227 // We may have <foo> </foo>, <?foo?> or <!foo>
228 // In any case, buffer B is flushed at the start, reset at the end
229 void XmlIO::parse_lt() {
230  flush_buffer();
231  cur_char = next_char();
232  if (cur_char == '/')
233  parse_end();
234  else if (cur_char == '!')
235  parse_dec();
236  else if (cur_char == '?')
237  parse_pi();
238  else
239  parse_tag();
240  B.reset();
241 }

294 void XmlIO::parse_quoted() {
295  skip_space();
296  codepoint delim{' '};
297  if (cur_char == '\'' || cur_char == '"')
298  delim = cur_char;
299  else {
300  error("Quote expected");
301  std::cout << " got " << cur_char << "\n";
302  }
303  skip_char();
304  B.reset();
305  for (;;) {
306  codepoint x = XmlIO::next_char();
307  if (x == delim) break;
308  B.push_back(x);
309  }

312 // Parses the attributes of the current element
313 void XmlIO::parse_attributes() {
314  for (;;) {
315  skip_space();
316  // handle the case <foo/>
317  if (cur_char == '/') {
318  skip_char();
319  cur_char = peek_char();
320  pop_this();
321  return;
322  }
323  // case of <foo>
324  if (cur_char == '>') {
325  skip_char(); // no need to pop
326  return;
327  }
328  // Now we have an attribute pair
329  scan_name();
330  auto att_name = Istring(B);
331  parse_att_val();
332  auto att_val = Istring(B);
333  cur_xml->get_id().add_attribute(att_name, att_val);
334  }
335 }

337 // Case of process instruction, say <?xml?>
338 // The PI is ignored
339 void XmlIO::parse_pi() {
340  scan_name('?');
341  bool is_tralics = B.is_equal("tralics");
342  bool is_xml = B.is_equal("xml");
343  aux.reset();
344  aux << B;
345  if (is_xml) {
346  for (;;) {
347  skip_space();
348  if (cur_char == '?') {
349  skip_char();
350  cur_char = next_char();
351  if (cur_char != '>') {
352  error("Greater than sign expected");
353  std::cout << " got " << cur_char << "\n";
354  }
355  break;
356  }
357  scan_name('=');
358  bool is_encoding = B.is_equal("encoding");
359  aux << " " << B << "='";
360  parse_att_val();
361  aux << B << "'";
362  if (is_encoding) {
363  if (B.is_equal("iso-8859-1"))
364  enc = 1;
365  else if (B.is_equal("UTF-8"))
366  enc = 0;
367  else
368  error("Unrecognised encoding");
369  }
370  }
371  } else
372  for (;;) {
373  codepoint c = next_char();
374  if (c == '?') {
375  cur_char = peek_char();
376  if (cur_char == '>') {
377  skip_char();
378  break;
379  }
380  }
381  B << c;
382  }
383  Xml *res = new Xml(aux);
384  res->change_id(-3); // mark this as a pi
385  cur_xml->push_back(res);
386  if (is_tralics) std::cout << "Unrecognised PI " << B << "\n";
387 }

389 // Scans a declaration. We test the first or second letter
390 void XmlIO::parse_dec() {
391  cur_char = next_char();
392  codepoint c = peek_char();
393  reread_list.push_back(cur_char);
394  if (c == '-')
395  parse_dec_comment();
396  else if (c == 'N')
397  parse_dec_entity();
398  else if (c == 'L')
399  parse_dec_element();
400  else if (cur_char == 'A')
401  parse_dec_attlist();
402  else if (cur_char == 'D')
403  parse_dec_doctype();
404  else if (c == 'C')
405  parse_dec_cdata();
406  else if (cur_char == '[')
407  parse_dec_conditional();
408  else
409  parse_dec_notation();
410 }

426 // Comments are of the form <!-- comment -->
427 // Three hyphens in a row are illegal
428 // Comments are ignored.
429 void XmlIO::parse_dec_comment() {
430  expect("--");
431  B.reset();
432  for (;;) {
433  codepoint c = next_char();
434  B.push_back(c);
435  if (c != '>') continue;
436  auto k = B.size(); // B[k-1] is >
437  if (k >= 3 && B[k - 2] == '-' && B[k - 3] == '-') break;
438  }
439  B.remove_last_n(3);
440  Xml *res = new Xml(B);
441  res->change_id(-1); // mark this as a comment
442  cur_xml->push_back(res);
443 }

446 void XmlIO::parse_dec_cdata() {
447  expect("[CDATA");

462 // to be completed
463 void XmlIO::parse_dec_conditional() {
464  B.reset();
465  skip_char(); // skip
466  B.push_back("<![");
467  bool keep = false;
468  skip_space();
469  codepoint c = peek_char();
470  if (c == '%') {
471  skip_char();
472  expand_PEReference();
473  }
474  c = next_char();
475  if (c != 'I') error("expected INCLUDE or IGNORE");
476  c = peek_char();
477  if (c == 'N') {
478  expect("NCLUDE");
479  keep = true;
480  } else if (c == 'G')
481  expect("GNORE");
482  else
483  error("expected INCLUDE or IGNORE");
484  skip_space();
485  if (cur_char != '[') error("expected [");
486  skip_char();
487  B.reset();
488  if (keep)
489  B << "[INCLUDE[";
490  else
491  B << "[IGNORE[";
492  Xml *res = new Xml(Istring(B));
493  res->change_id(-2); // mark this as a declaration
494  cur_xml->push_back(res);
495  cur_xml = res;
496  cur_stack.push_back(cur_xml);
497  ++nb_cond;
498  B.reset();
499 }

521 // We scan one of
522 // <!ENTITY foo PUBLIC 'pub-part' 'system-part'>
523 // <!ENTITY foo SYSTEM 'system-part'>
524 // We insert
525 void XmlIO::parse_dec_entity() {
526  aux.reset();
527  aux.push_back("ENTITY ");
528  expect("ENTITY");
529  skip_space();
530  bool parameter = false;
531  if (cur_char == '%') {
532  skip_char();
533  skip_space();
534  aux.push_back("% ");
535  parameter = true;
536  }
537  scan_name(0);
538  std::string name = B.to_string(); // This is the entity name
539  aux.push_back(name);
540  skip_space();
541  // entity value is defined by a name (system or public) or value
542  if (!parse_sys_pub()) {
543  parse_quoted();
544  aux << " '" << B << "'";
545  if (parameter) entities.emplace_back(name, B.to_string());
546  }
547  skip_space();
548  // We might have NDATA here, if parameter==false
549  if (cur_char != '>') {
550  B.reset();
551  for (;;) {
552  codepoint c = next_char();
553  if (c == '>') break;
554  B.push_back(c);
555  }
556  aux << " " << B;
557  } else
558  skip_char();
559  Xml *res = new Xml(aux);
560  res->change_id(-2); // mark this as a declaration
561  cur_xml->push_back(res);
562 }

564 // We might have <!ELEMENT %foo; %bar;>
565 void XmlIO::parse_dec_element() {
566  Buffer tmp;
567  aux.reset();
568  aux.reset();
569  tmp.push_back("ELEMENT");
570  expect("ELEMENT");
571  bool ok = true;
572  bool is_first = true;
573  bool prev_is_space = false;
574  bool first_space = true;
575  std::string elt_name;
576  for (;;) {
577  codepoint c = next_char();
578  if (c == '>') break;
579  if (c == '%') {
580  if (expand_PEReference()) continue;
581  ok = false;
582  }
583  if (is_spacer(c)) {
584  if (is_first || prev_is_space) continue;
585  if (first_space) {
586  elt_name = aux.to_string();
587  aux.reset();
588  first_space = false;
589  is_first = true;
590  continue;
591  }
592  aux.push_back(c);
593  prev_is_space = true;
594  continue;
595  }
596  aux.push_back(c);
597  is_first = false;
598  prev_is_space = false;
599  }
600  if (!ok) the_log << "Possible problem in XML scan on line " << the_parser.get_cur_line() << "\n";
601  if (prev_is_space) aux.rrl();
602  tmp << " " << elt_name;
603  if (aux.is_equal("EMPTY"))
604  ;
605  else if (aux.is_equal("ANY"))
606  ;
607  else /* check syntax */
608  tmp << " ";
609  tmp << " " << aux;
610  Xml *res = new Xml(tmp);
611  res->change_id(-2); // mark this as a declaration
612  cur_xml->push_back(res);
613 }

669 void XmlIO::parse_dec_doctype() {
670  expect("DOCTYPE");
671  Xml *res = new Xml(Istring("DOCTYPE"), nullptr);
672  res->change_id(-2); // mark this as a declaration
673  cur_xml->push_back(res); // Insert this in the tree
674  skip_space();
675  scan_name(0);
676  aux.reset();
677  aux << " " << B;
678  skip_space();
679  parse_sys_pub();
680  res->add_last_string(aux);
681  skip_space();
682  skip_char();
683  if (cur_char != '[') {
684  if (cur_char != '>') error("Expected [ or > at end of doctype");
685  return;
686  }
687  // push the element on the stack
688  cur_xml = res;
689  cur_stack.push_back(cur_xml);
690  // Read the local DTD
691  B.reset();
692  B << " [";
693  for (;;) {
694  codepoint c = next_char();
695  if (c == '<')
696  parse_lt();
697  else {
698  B.push_back(c);
699  if (c == ']') {
700  if (nb_cond != 0) {
701  c = next_char();
702  if (c != ']') error("Expected ]]>");
703  B.push_back("]");
704  flush_buffer();
705  cur_char = peek_char();
706  pop_this();
707  nb_cond--;
708  continue;
709  }
710  break;
711  }
712  }
