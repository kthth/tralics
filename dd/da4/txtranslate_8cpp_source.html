249 auto Parser::get_ctb_opt() -> name_positions {
250  TokenList L;
251  read_optarg_nopar(L);
252  if (L.empty()) return cst_invalid;
253  Token t = token_ns::get_unique(L);
254  if (t.is_null()) return cst_invalid;
255  if (t.cmd_val() != letter_catcode) return cst_invalid;
256  auto c = t.val_as_letter();
257  if (c == 'c') return np_letter_c;
258  if (c == 's') return np_letter_s;
259  if (c == 't') return np_letter_t;
260  if (c == 'b') return np_letter_b;
261  if (c == 'l') return np_letter_l;
262  if (c == 'r') return np_letter_r;
263  return cst_invalid;
264 }

266 // Nodes: tblr, or 2 letters
267 auto Parser::get_trees_opt() -> name_positions {
268  TokenList L;
269  read_optarg_nopar(L);
270  if (L.empty()) return cst_invalid;
271  Token t1, t2;
272  token_ns::get_unique(L, t1, t2);
273  if (t1.is_null()) return cst_invalid;
274  if (t1.cmd_val() != letter_catcode) return cst_invalid;
275  auto c = t1.val_as_letter();
276  if (t2.is_null()) {
277  if (c == 'l') return np_letter_l;
278  if (c == 'r') return np_letter_r;
279  if (c == 't') return np_letter_t;
280  if (c == 'b') return np_letter_b;
281  return cst_invalid;
282  }
283  if (t2.cmd_val() != letter_catcode) return cst_invalid;
284  auto c2 = t2.val_as_letter();
285  if (c != 't' && c != 'b') return cst_invalid;
286  if (c2 != 'l' && c2 != 'r') return cst_invalid;
287  if (c == 't') return c2 == 'l' ? np_letters_tl : np_letters_tr;
288  return c2 == 'l' ? np_letters_bl : np_letters_br;
289 }

554 // Translates \section, \chapter, etc.
555 void Parser::T_paras(subtypes x) {
556  if (x == toplevel_sec_code) {
557  TokenList L = read_arg();
558  if (!L.empty()) {
559  token_from_list(L.front());
560  sectionning_offset = section_code;
561  if (cur_cmd_chr.cmd == section_cmd) sectionning_offset = cur_cmd_chr.chr;
562  if (sectionning_offset == chapter_code) Xid(1).add_attribute(np_chapters, np_true);
563  if (sectionning_offset == part_code) Xid(1).add_attribute(np_part, np_true);
564  }
565  return;
566  }
567  int y = x - sectionning_offset;
568  if (x == endsec_code) {
569  y = 0;
570  TokenList L = read_arg();
571  if (!L.empty()) {
572  token_from_list(L.front());
573  if (cur_cmd_chr.cmd == section_cmd) y = cur_cmd_chr.chr - sectionning_offset;
574  }
575  }
576  if (y < 0) y = 0;
577  if (y > 6) y = 6;
578  Istring Y = the_names[cst_div0 + to_unsigned(y)];
579  leave_h_mode();
580  the_stack.para_aux(y); // this pops the stack...
581  the_stack.add_nl();
582  if (x == endsec_code) return;
583  the_stack.push1(Y, name_positions(np_div0 + y));
584  bool star = remove_initial_star();
585  if (chapter_has_star && x == chapter_code) star = true;
586  if (star)
587  last_att_list().push_back(np_rend, np_nonumber);
588  else {
589  if (x == part_code)
590  refstepcounter("part", false);
591  else if (x == chapter_code)
592  refstepcounter("chapter", false);
593  else if (x == section_code)
594  refstepcounter("section", false);
595  else if (x == subsection_code)
596  refstepcounter("subsection", false);
597  else if (x == subsubsection_code)
598  refstepcounter("subsubsection", false);
599  else if (x == paragraph_code)
600  refstepcounter("paragraph", false);
601  else if (x == subparagraph_code)
602  refstepcounter("subparagraph", false);
603  else
604  star = true;
605  }
606  start_paras(y, Y.c_str(), star);
607 }
