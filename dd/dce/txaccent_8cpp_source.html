93 auto accent_ns::combine_accents(int acc1, int acc2) -> int {
94  // start with acute and something
95  if (acc1 == '\'' && acc2 == '~') return 9;
96  if (acc2 == '\'' && acc1 == '~') return 9;
97  if (acc1 == '\'' && acc2 == '=') return 10;
98  if (acc2 == '\'' && acc1 == '=') return 10;
99  if (acc1 == '\'' && acc2 == '.') return 11;
100  if (acc2 == '\'' && acc1 == '.') return 11;
101  if (acc1 == '\'' && acc2 == '^') return 12;
102  if (acc2 == '\'' && acc1 == '^') return 12;
103  if (acc1 == '\'' && acc2 == 'H') return 16;
104  if (acc2 == '\'' && acc1 == 'H') return 16;
105  if (acc1 == '\'' && acc2 == 'c') return 17;
106  if (acc2 == '\'' && acc1 == 'c') return 17;
107  if (acc1 == '\'' && acc2 == '"') return 2;
108  if (acc2 == '\'' && acc1 == '"') return 2;
109  if (acc1 == '\'' && acc2 == '^') return 21;
110  if (acc2 == '\'' && acc1 == '^') return 21;
111  if (acc1 == '\'' && acc2 == 'u') return 34;
112  if (acc2 == '\'' && acc1 == 'u') return 34;
113  // now grave and something
114  if (acc1 == '`' && acc2 == '=') return 13;
115  if (acc2 == '`' && acc1 == '=') return 13;
116  if (acc1 == '`' && acc2 == '^') return 15;
117  if (acc2 == '`' && acc1 == '^') return 15;
118  if (acc1 == '`' && acc2 == 'H') return 18;
119  if (acc2 == '`' && acc1 == 'H') return 18;
120  if (acc1 == '`' && acc2 == '"') return 4;
121  if (acc2 == '`' && acc1 == '"') return 4;
122  if (acc1 == '`' && acc2 == 'u') return 22;
123  if (acc2 == '`' && acc1 == 'u') return 22;
124  // now dot and something
125  if (acc1 == '.' && acc2 == '=') return 5;
126  if (acc2 == '.' && acc1 == '=') return 5;
127  if (acc1 == '.' && acc2 == 'v') return 19;
128  if (acc2 == '.' && acc1 == 'v') return 19;
129  if (acc1 == '.' && acc2 == 'd') return 20;
130  if (acc2 == '.' && acc1 == 'd') return 20;
131  // now hat and something
132  if (acc1 == '^' && acc2 == 'h') return 23;
133  if (acc2 == '^' && acc1 == 'h') return 23;
134  if (acc1 == '^' && acc2 == '~') return 24;
135  if (acc2 == '^' && acc1 == '~') return 24;
136  if (acc1 == '^' && acc2 == 'd') return 25;
137  if (acc2 == '^' && acc1 == 'd') return 25;
138  // diaresis
139  if (acc1 == '"' && acc2 == '=') return 1;
140  if (acc2 == '"' && acc1 == '=') return 1;
141  if (acc1 == '"' && acc2 == 'v') return 3;
142  if (acc2 == '"' && acc1 == 'v') return 3;
143  if (acc1 == '"' && acc2 == '~') return 27;
144  if (acc2 == '"' && acc1 == '~') return 27;
145  // other
146  if (acc1 == '=' && acc2 == '~') return 6;
147  if (acc2 == '=' && acc1 == '~') return 6;
148  if (acc1 == 'k' && acc2 == '=') return 7;
149  if (acc2 == 'k' && acc1 == '=') return 7;
150  if (acc1 == 'd' && acc2 == '=') return 14;
151  if (acc2 == 'd' && acc1 == '=') return 14;
152  if (acc1 == 'u' && acc2 == 'c') return 26;
153  if (acc2 == 'u' && acc1 == 'c') return 26;
154  if (acc1 == 'H' && acc2 == 'h') return 28;
155  if (acc2 == 'H' && acc1 == 'h') return 28;
156  if (acc1 == 'H' && acc2 == '~') return 29;
157  if (acc2 == 'H' && acc1 == '~') return 29;
158  if (acc1 == 'H' && acc2 == 'd') return 30;
159  if (acc2 == 'H' && acc1 == 'd') return 30;
160  if (acc1 == 'u' && acc2 == 'h') return 31;
161  if (acc2 == 'u' && acc1 == 'h') return 31;
162  if (acc1 == 'u' && acc2 == '~') return 32;
163  if (acc2 == 'u' && acc1 == '~') return 32;
164  if (acc1 == 'u' && acc2 == 'd') return 33;
165  if (acc2 == 'u' && acc1 == 'd') return 33;
166  return 0;
167 }

347 // This implements \'e, or \'{a} or \a{'}{e}, \href{\~grimm}
348 // or \'{\= u}, a double accent
349 // cur_tok, cur_cmd_chr explain what to do.
350 void Parser::E_accent() {
351  if (tracing_macros()) the_log << lg_startbrace << "accent " << cur_tok << lg_endbrace;
352  int acc_code = cur_cmd_chr.chr;
353  if (InUrlHandler::global_in_url && acc_code == '~') {
354  if (tracing_macros()) { the_log << lg_startbrace << "\\~ gives ~ " << lg_endbrace; }
355  back_input(Token(other_t_offset, '~'));
356  return;
357  }
358  String msg1 = "Error in accent, command = ";
359  String msg2 = "bad accent";
360  Token tfe = cur_tok;
361  // Fetch the accent
362  TokenList y = read_arg();
363  if (y.empty()) {
364  err_buf << bf_reset << msg1 << tfe.tok_to_str();
365  err_buf << (acc_code == '~' ? "\n\\~{} is the wrong way to put a tilde in an URL" : "\nThings like {\\'{}} are a bit strange");
366  signal_error(err_tok, msg2);
367  return;
368  }
369  int acc_code2 = 0; // will be set in the double acc case.
370  Token tfe2; // Second token for error.
371  if (y.size() >= 2) {
372  token_from_list(y.front()); // Get the meaning of the start of y
373  // case \'{\a'e};
374  if (cur_cmd_chr.cmd == a_cmd) {
375  y.pop_front(); // discard the \a
376  accent_ns::special_acc_hack(y);
377  if (y.empty()) {
378  parse_error("Internal error, lost argument of \\a");
379  return;
380  }
381  Token t = y.front(); // get the '
382  if (!t.is_a_char()) {
383  parse_error(err_tok, "Bad syntax of \\a, argument not a character ", t, "", msg2);
384  return;
385  }
386  y.pop_front(); // discard the accent
387  tfe2 = Token(t.chr_val() + single_offset);
388  token_from_list(tfe2);
389  if (cur_cmd_chr.cmd != accent_cmd) {
390  parse_error(err_tok, "Bad syntax of \\a, argument is ", t, "", msg2);
391  return;
392  }
393  } else if (cur_cmd_chr.cmd == accent_cmd) {
394  tfe2 = cur_tok;
395  y.pop_front();
396  } else {
397  parse_error(err_tok, msg1, tfe, "\nWanted a single token", msg2);
398  return;
399  }
400  acc_code2 = cur_cmd_chr.chr;
401  accent_ns::special_acc_hack(y);
402  if (y.size() != 1) {
403  parse_error(err_tok, msg1, tfe, "\nWanted a single token", msg2);
404  return;
405  }
406  }
407  Token Y = token_ns::get_unique(y);
408  token_from_list(Y);
409  unsigned int achar = cur_cmd_chr.chr;
410  if (achar <= 8)
411  achar = 8; // make these invalid
412  else if (achar == 0xc6)
413  achar = 1;
414  else if (achar == 0xe6)
415  achar = 2;
416  else if (achar == 0xc5)
417  achar = 3;
418  else if (achar == 0xe5)
419  achar = 4;
420  else if (achar == 0xd8)
421  achar = 5;
422  else if (achar == 0xf8)
423  achar = 6;
424  if (cur_cmd_chr.cmd == cst1_cmd && achar == i_code)
425  achar = 'i';
426  else if ((cur_cmd_chr.cmd == specchar_cmd) || cur_cmd_chr.is_letter_other()) {
427  } else {
428  err_buf << bf_reset << msg1;
429  err_buf << tfe.tok_to_str();
430  if (acc_code2 != 0) err_buf << tfe2.tok_to_str();
431  if (Y.is_invalid())
432  err_buf << "\nEnd of data reached while scanning argument";
433  else if (!cur_tok.not_a_cmd())
434  err_buf << "\nLetter needed instead of " << cur_tok.tok_to_str();
435  else
436  err_buf << "\nLetter needed";
437  signal_error(err_tok, msg2);
438  return;
439  }
440  Token res;
441  bool spec = false;
442  if (achar < nb_accents) {
443  if (acc_code2 != 0) {
444  int acc3_code = accent_ns::combine_accents(acc_code, acc_code2);
445  res = accent_ns::fetch_double_accent(to_signed(achar), acc3_code);
446  } else {
447  res = accent_ns::fetch_accent(achar, acc_code);
448  if (res.is_null() && achar < 128) {
449  if (is_letter(static_cast<char>(achar))) {
450  spec = true; // T. Bouche veut une erreur
451  res = accent_ns::fetch_accent(0, acc_code);
452  } else
453  res = accent_ns::special_double_acc(to_signed(achar), acc_code);
454  }
455  }
456  }
457  if (res.is_null()) {
458  String s =
459  achar >= 128
460  ? "a non 7-bit character"
461  : is_letter(static_cast<char>(achar)) ? "letter" : is_digit(static_cast<char>(achar)) ? "digit" : "non-letter character";
462  err_buf << bf_reset << msg1;
463  err_buf << tfe.tok_to_str();
464  if (acc_code2 != 0) err_buf << tfe2.tok_to_str();
465  err_buf << "\nCannot put accent on " << s;
466  if (achar < 128) err_buf << " " << char(achar);
467  signal_error(err_tok, msg2);
468  return;
469  }
470  TokenList expansion;
471  expansion.push_back(res);
472  if (spec) {
473  expansion.push_front(Y);
474  brace_me(expansion);
475  expansion.push_front(hash_table.composite_token);
476  }
477 
478  if (tracing_macros()) { the_log << lg_startbrace << "accent on " << uchar(achar) << " -> " << expansion << lg_endbrace; }
479  back_input(expansion);
480 }
