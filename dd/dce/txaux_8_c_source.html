52 // If this is a TeX token representing a valid digit in base radix,
53 // returns the value,
54 // otherwise return -1.
55 auto Token::tex_is_digit(unsigned radix) -> int {
56  auto w = val_as_other();
57  if ('0' <= w && w <= radix + '0' && w <= '9') // do not use is_digit...
58  return static_cast<int>(w) - '0';
59  if (radix != 16) return -1;
60  if (w <= 'F' && w >= 'A') return static_cast<int>(w) - 'A' + 10;
61  w = val_as_letter(); // allow category code 11 digits
62  if (w <= 'F' && w >= 'A') return static_cast<int>(w) - 'A' + 10;
63  return -1;
64 }

98 // computes xn/d , returns quotient and remainder
99 // Assumes n>0 and d>0. Quotient is < 2^30
100 // This is used for instance when converting 38cm into 38*n/d pt
101 auto arith_ns::xn_over_d(int x, int n, int d, int &remainder) -> int {
102  const int two_fifteen = 1 << 15;
103  bool positive = true;
104  if (x < 0) {
105  positive = false;
106  x = -x;
107  }
108  int t = (x % two_fifteen) * n;
109  int u = (x / two_fifteen) * n + (t / two_fifteen);
110  int v = (u % d) * two_fifteen + (t % two_fifteen);
111  int w = u / d;
112  if (w >= two_fifteen) {
113  start_err("2^{30}");
114  err_buf << "\nin " << x << "*" << n << "/" << d;
115  end_err();
116  u = 1;
117  } else
118  u = two_fifteen * w + (v / d);
119  remainder = positive ? (v % d) : -(v % d);
120  return positive ? u : -u;
121 }

136 // Return true if overflow.
137 auto ScaledInt::scale(int x, int n, int d, int max_answer, bool &negative) -> bool {
138  value = 0;
139  if (x == 0) return false;
140  if (n == 0) return false;
141  if (d < 0) {
142  d = -d;
143  negative = true;
144  }
145  if (x < 0) {
146  x = -x;
147  negative = !negative;
148  }
149  if (n < 0) {
150  n = -n;
151  negative = !negative;
152  }
153  int t = n / d;
154  if (t > max_answer / x) return true;
155  int a = t * x;
156  n = n - t * d; // answer is a + xn/d, with n<d
157  if (n == 0) {
158  value = a;
159  return false;
160  }
161  t = x / d;
162  if (t > (max_answer - a) / n) return true;
163  a = a + t * n;
164  x = x - t * d; // answer is a + xn/d, with n<d and x<d
165  if (x == 0) {
166  value = a;
167  return false;
168  }
169  if (x < n) {
170  t = x;
171  x = n;
172  n = t;
173  }
174  // Invariants -d<=r<0<n<=x<d
175  // f+ floor[(xn+(r+d))/d] = floor(x0n0/d+1/2)
176  int f = 0;
177  int r = (d / 2) - d;
178  int h = -r;
179  for (;;) {
180  if ((n & 1) != 0) {
181  r = r + x;
182  if (r >= 0) {
183  r = r - d;
184  ++f;
185  }
186  }
187  n = n / 2;
188  if (n == 0) break;
189  if (x < h)
190  x = x + x;
191  else {
192  t = x - d;
193  x = t + x;
194  f = f + n;
195  if (x == 0) break;
196  if (x < n) {
197  t = x;
198  x = n;
199  n = t;
200  }
201  }
202  }
203  if (f > (max_answer - a)) return true;
204  value = a + f;
205  return false;
206 }

208 // Computes x/n, rounded
209 // No overflow possible. Division by zero must be checked by caller.
210 // This is used when you say \dimexpr 2pt/3
211 void ScaledInt::quotient(int d) {
212  bool negative = false;
213  int n = value;
214  if (n < 0) {
215  n = -n;
216  negative = !negative;
217  }
218  if (d < 0) {
219  d = -d;
220  negative = !negative;
221  }
222  int a = n / d;
223  n = n - a * d;
224  d = n - d;
225  if (d + n >= 0) ++a;
226  if (negative) a = -a;
227  value = a;
228 }
