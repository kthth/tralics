165 // Translate current character.
166 void Parser::translate_char(CmdChr X) {
167  if (!the_stack.in_h_mode()) LC();
168  codepoint c = X.char_val();
169  if (c.is_null()) return; // we do not want null chars in trace or Xml
170  if (tracing_commands()) the_log.out_single_char(c);
171  if (!c.is_small()) {
172  process_char(c);
173  return;
174  }
175  switch (c.value) {
176  case '-': minus_sign(X); return;
177  case '\'':
178  case '`':
179  case '<':
180  case '>': english_quotes(X); return;
181  case '"':
182  if (X.is_letter() || InUrlHandler::global_in_url || InLoadHandler::global_in_load)
183  process_char(c);
184  else
185  umlaut();
186  return;
187  case ';':
188  case ':':
189  case '!':
190  case '?':
191  case 0xBB: // <<
192  case 0xAB: // >>
193  french_punctuation(X);
194  return;
195  default: process_char(c);
196  }
197 }

199 // This translates `'<>
200 // In some case ``, '', << and >> are translated as 0xAB and 0xBB
201 void Parser::english_quotes(CmdChr X) {
202  auto c = X.char_val().value; // Should be a small int
203  if (InUrlHandler::global_in_url || InLoadHandler::global_in_load) {
204  if (c == '<')
205  process_string("&lt;");
206  else if (c == '>')
207  process_string("&gt;");
208  else
209  unprocessed_xml.push_back(static_cast<char>(c));
210  } else {
211  if (X.is_other() && (c == '\'' || c == '`' || cur_lang_fr())) {
212  get_x_token();
213  if (cur_cmd_chr.is_other() && cur_cmd_chr.char_val() == codepoint(c)) {
214  if (c == '\'')
215  process_char(0x201d);
216  else if (c == '`')
217  process_char(0x201c);
218  else if (c == '<')
219  french_punctuation(CmdChr(uchar(0xAB)));
220  else
221  french_punctuation(CmdChr(uchar(0xBB)));
222  return;
223  }
224  if (cur_tok.is_valid()) back_input();
225  }
226  if (c == '<')
227  process_string("&lt;");
228  else if (c == '>')
229  process_string("&gt;");
230  else if (c == '\'' && X.is_other())
231  process_char(rightquote_val);
232  else if (c == '`' && X.is_other())
233  process_char(leftquote_val);
234  else
235  unprocessed_xml.push_back(static_cast<char>(c));
236  }
237  if (X.is_letter() && !the_main->get_zws_mode()) process_string(the_main->get_zws_elt() ? "&#x200B;" : "<zws/>");
238 }

264 // This handles :;!? 0xAB 0xBB. Especially in French.
265 void Parser::french_punctuation(CmdChr X) {
266  auto c = X.char_val().value;
267  if (InUrlHandler::global_in_url || InLoadHandler::global_in_load || X.is_letter() || !cur_lang_fr()) {
268  extended_chars(c);
269  return;
270  }
271  if (c == 0xAB) {
272  for (;;) {
273  get_x_token();
274  if (cur_cmd_chr.is_space()) continue;
275  if (cur_cmd_chr.cmd == cst_cmd && cur_cmd_chr.chr == nobreakspace_code) break;
276  if (cur_cmd_chr.cmd == cst1_cmd && cur_cmd_chr.chr == comma_code) break;
277  if (cur_tok.is_valid()) back_input();
278  break;
279  }
280  extended_chars(c);
281  process_char(0xA0);
282  return;
283  }
284  unprocessed_xml.remove_last_space();
285  process_char(0xA0);
286  extended_chars(c);
287 }
