86 // Interprets Sections ="/foo/bar*/gee".
87 // In 2007, we have NewSections ="/foo/FOO/bar/BAR/gee/GEE"
88 // This is indicated by new_syntax
89 // Puts foo bar+ gee in the transcript file (we use + instead of *, meaning
90 // that the character was interpreted). Three RaSection objects are
91 // created, named foo, bar and gee. The second is marked: not for topic.
92 // In simplified mode, a section has a name and a number.
93 // Otherwise it has additional fields.
94 void config_ns::interpret_section_list(Buffer &B, bool new_syntax) {
95  ParamDataList *V = config_data.data[1];
96  if (start_interpret(B, "//")) {
97  V->reset();
98  sec_buffer.reset();
99  composition_section = -1;
100  }
101  std::string s, r;
102  for (;;) {
103  if (!B.slash_separated(s)) break;
104  if (new_syntax) {
105  if (!B.slash_separated(r)) break;
106  } else
107  r = "";
108  if (s.empty()) continue;
109  bool star = false;
110  auto ns = s.size();
111  if (ns > 1 && s[ns - 1] == '*') {
112  s = std::string(s, 0, ns - 1);
113  star = true;
114  }
115  if (r.empty()) r = s;
116  the_log << "Section: " << s << (star ? "+" : "") << " -> " << r << "\n";
117  if (s == "composition") composition_section = to_signed(V->size() + 1);
118  sec_buffer << " " << s;
119  V->push_back(ParamDataSlot(s, r, !star));
120  }
121 }

177 // return non-empty string only if section is new
178 auto config_ns::check_section(const std::string &s) -> std::string {
179  static long cur_section = -1;
180  long k = -1;
181  err_buf.reset();
182  std::vector<ParamDataSlot> &X = config_data.data[1]->data;
183  auto n = X.size(); // number of sections
184  if (s.empty())
185  k = cur_section;
186  else
187  for (size_t i = 0; i < n; i++)
188  if (X[i].key == s) {
189  k = to_signed(i + 1);
190  break;
191  }
192  if (k > 0 && k < cur_section) {
193  err_buf << "Bad section " << s << " after " << X[to_unsigned(cur_section - 1)].key << "\n"
194  << "Order of sections is" << sec_buffer;
195  the_parser.signal_error();
196  } else if (k == -1) {
197  if (n == 0) {
198  the_parser.parse_error("Illegal access to fullsection list.");
199  return "";
200  }
201  if (!s.empty()) {
202  err_buf << "Invalid section " << s << "\n"
203  << "Valid sections are" << sec_buffer;
204  the_parser.signal_error();
205  if (cur_section < 0) cur_section = 1;
206  }
207  } else
208  cur_section = k;
209  if (cur_section < 0) {
210  the_parser.parse_error("No default section");
211  cur_section = 1;
212  }
213  if (cur_section == composition_section) {
214  static bool first_module = true;
215  if (first_module)
216  first_module = false;
217  else {
218  the_parser.parse_error("Only one module accepted in composition");
219  cur_section++;
220  return "";
221  }
222  }
223  static long prev = -1;
224  if (prev == cur_section) return "";
225  prev = cur_section;
226  cur_sec_no_topic = X[to_unsigned(cur_section - 1)].no_topic();
227  X[to_unsigned(cur_section - 1)].mark_used(); // incompatible with topics
228  return X[to_unsigned(cur_section - 1)].value;
229 }

245  bool ret_val = false;
246  B.push_back(s);
247  B.reset_ptr();

275 // User says \UR{foo,bar}
276 // returns -1 if there no other RC in the buffer.
277 // returns -2 if the RC is invalid
278 // returns location in the table otherwise
279 auto config_ns::next_RC_in_buffer(Buffer &B, std::string &sname, std::string &lname) -> long {
280  std::vector<ParamDataSlot> &ur_list = config_data.data[0]->data;
281  B.skip_sp_tab_comma();
282  if (B.head() == 0) return -1;
283  if (strncmp(B.c_str(B.ptr), "\\UR", 3) == 0) {
284  static bool warned = false;
285  if (!warned && the_parser.get_ra_year() > 2006) {
286  log_and_tty << "You should use Lille instead of \\URLille,\n";
287  log_and_tty << "Nancy instead of \\URNancy, etc.\n";
288  warned = true;
289  }
290  B.advance(3);
291  }
292  B.set_ptr1_to_ptr();
293  B.skip_letter();
294  auto k = ur_list.size();
295  for (size_t j = 0; j < k; j++)
296  if (B.contains_here(ur_list[j].key.c_str())) {
297  sname = ur_list[j].key;
298  lname = ur_list[j].value;
299  ur_list[j].mark_used();
300  return to_signed(j);
301  }
302  return -2;
303 }

355 // Interprets the RC argument of a pers command
356 // This returns the short name, said otherwise, the argument.
357 // Notice the space case when argument is empty, or +foo or =foo.
358 auto config_ns::pers_rc(const std::string &rc) -> std::string {
359  if (rc.empty()) {
360  if (have_default_ur) return the_default_rc;
361  if (the_main->in_ra() && the_parser.get_ra_year() > 2006) {
362  // signal error, make a default
363  the_parser.parse_error("No default Research Centre defined");
364  the_default_rc = "unknown";
365  }
366  have_default_ur = true;
367  return the_default_rc;
368  }
369  if (rc[0] == '+') { return rc.c_str() + 1; }
370  bool spec = (rc.size() >= 2 && rc[0] == '=');
371  std::string RC = spec ? rc.c_str() + 1 : rc;
372  if (!is_good_ur(RC)) {
373  err_buf << bf_reset << "Invalid Unit Centre " << rc << "\n"
374  << "Use one of:";
375  std::vector<ParamDataSlot> &V = config_data.data[0]->data;
376  auto n = V.size();
377  for (size_t i = 0; i < n; i++)
378  if (V[i].is_used) V[i].key_to_buffer(err_buf);
379  the_parser.signal_error(the_parser.err_tok, "illegal data");
380  }
381  if (spec) {
382  the_default_rc = RC;
383  have_default_ur = true;
384  }
385  return RC;
386 }
