93 // and \iow_term:x \iow_log:x of latex3
94 // action is \immediate, since there is no shipout routine
95 void Parser::M_extension(int cc) {
96  static TexOutStream tex_out_stream; // the output streams
97  long chan = 0;
98  if (cc == openout_code)
99  chan = to_signed(scan_int(cur_tok, max_openout, "output channel number"));
100  else if (cc == write_log_code || cc == wlog_code)
101  chan = negative_out_slot;
102  else if (cc == write_term_code || cc == typeout_code)
103  chan = positive_out_slot;
104  else {
105  chan = scan_int(cur_tok);
106  if (chan < 0)
107  chan = negative_out_slot;
108  else if (chan > max_openout && chan != write18_slot)
109  chan = positive_out_slot;
110  if (chan == write18_slot && !the_main->is_shell_escape_allowed()) chan = positive_out_slot;
111  }
112  auto uchan = to_unsigned(chan);
113  if (cc == openout_code) {
114  scan_optional_equals();
115  std::string file_name = scan_file_name();
116  tex_out_stream.open(uchan, file_name);
117  } else if (cc == closeout_code)
118  tex_out_stream.close(uchan);
119  else { // write to uchan
120  String s = string_to_write(chan);
121  if (uchan == write18_slot)
122  system(s);
123  else if (tex_out_stream.is_open(uchan))
124  tex_out_stream.write(uchan, s);
125  else if (uchan == negative_out_slot)
126  the_log << s;
127  else
128  log_and_tty << s;
129  }

163 // This puts in cur_input_stack a new slot, containing the current state
164 // of affairs. It kills the input buffer and the current line.
165 void Parser::push_input_stack(const std::string &name, bool restore_at, bool re) {
166  auto *W = new InputStack(name, get_cur_line(), state, cur_file_pos, every_eof, require_eof);
167  cur_input_stack.push_back(W);
168  auto n = cur_input_stack.size();
169  if (tracing_io()) the_log << lg_start_io << "Input stack ++ " << n << " " << W->get_name() << lg_end;
170  W->set_line_ptr(lines);
171  W->set_line_pos(input_line_pos);
172  W->set_line_vector(input_line);
173  W->get_TL().swap(TL);
174  if (restore_at) {
175  W->set_at_val(eqtb_int_table[uchar('@')].val);
176  eqtb_int_table[uchar('@')].set_val(11);
177  the_log << lg_start_io << "Made @ a letter\n";
178  }
179  every_eof = false; // might be set to true
180  require_eof = re;
181  input_line.clear();
182  lines.set_interactive(false);
183  input_line_pos = 0;
184  cur_file_pos = 0;
185 }

187 // Undoes push_input_stack.
188 // If vb is false, we are reading from a token list or something like that.
189 // Otherwise, we read from a real file, and indicate that it is closed.
190 void Parser::pop_input_stack(bool vb) {
191  auto n = cur_input_stack.size();
192  if (n == 0) {
193  if (tracing_io()) the_log << lg_start_io << "Input stack empty\n";
194  return;
195  }
196  InputStack *W = cur_input_stack.back();
197  if (vb) lines.before_close(true);
198  set_cur_line(W->get_line_no());
199  state = W->get_state();
200  W->get_line_ptr(lines);
201  require_eof = W->get_eof_outer();
202  auto at = W->get_at_val();
203  if (at >= 0) {
204  eqtb_int_table[uchar('@')].set_val(at);
205  if (tracing_io()) the_log << lg_start_io << "Catcode of @ restored to " << at << lg_end;
206  }
207  input_line.clear();
208  input_line.insert(input_line.end(), W->get_B().begin(), W->get_B().end());
209  input_line_pos = W->get_line_pos();
210  back_input(W->get_TL());
211  if (cur_file_pos != 0) insert_hook(cur_file_pos);
212  cur_file_pos = W->get_file_pos();
213  every_eof = W->get_every_eof();
214  the_log << lg_start_io << "cur_file_pos restored to " << cur_file_pos << lg_end;
215  cur_input_stack.pop_back();
216  if (tracing_io()) the_log << lg_start_io << "Input stack -- " << n << " " << W->get_name() << lg_end;
217  delete W;
218 }

238 // A file name is a special thing in TeX.
239 // We read until we find a non-char, or a space.
240 auto Parser::scan_file_name() -> std::string {
241  static Buffer file_name;
242 
243  if (name_in_progress) return "sabotage!"; // recursion killer.
244  name_in_progress = true;
245  remove_initial_space_and_back_input();
246  file_name.reset();
247  for (;;) {
248  if (get_x_token()) break;
249  if (cur_cmd_chr.is_letter_other())
250  file_name.push_back(cur_cmd_chr.char_val());
251  else if (cur_cmd_chr.cmd == underscore_catcode) // allow foo_bar
252  file_name.push_back(cur_cmd_chr.char_val());
253  else if (cur_cmd_chr.is_space())
254  break;
255  else {
256  back_input();
257  break;
258  }
259  }
260  name_in_progress = false;
261  return file_name.to_string();
262 }

326 // This implements \IfFileExists, \InputIfFileExists,
327 // \input, \include, \openin, \closein
328 void Parser::T_input(int q) {
329  size_t stream = 0;
330  bool seen_star = false;
331  bool seen_plus = false;
332  Token T = cur_tok;
333  if (q == openin_code || q == closein_code) {
334  stream = scan_int(cur_tok, max_openin, "input channel number");
335  tex_input_files[stream].close();
336  }
337  if (q == closein_code) return;
338  std::string file;
339  if (q == Input_code || q == include_code || q == inputifexists_code) seen_star = remove_initial_star();
340  if (q == inputifexists_code || q == ifexists_code) seen_plus = remove_initial_plus(true);
341  if (q == Input_code) q = include_code;
342  if (q == openin_code) {
343  scan_optional_equals();
344  file = scan_file_name();
345  } else if (q == input_code || q == include_code)
346  file = latex_input(q);
347  else {
348  flush_buffer();
349  InLoadHandler somthing;
350  file = sT_arg_nopar();
351  }
352  if (q == readxml_code) {
353  the_xmlA = read_xml(file);
354  return;
355  }
356  bool res = false;
357  if (seen_plus)
358  res = tralics_ns::find_no_path(file);
359  else {
360  res = tralics_ns::find_in_path(file);
361  if (!res) {
362  Buffer &B = local_buf;
363  B << bf_reset << file;
364  if (!B.is_at_end(".tex")) {
365  B.push_back(".tex");
366  std::string F = B.to_string();
367  res = tralics_ns::find_in_path(F);
368  }
369  }
370  }
371  if (q == openin_code) {
372  tex_input_files[stream].open(file, res);
373  return;
374  }
375  if (q == ifexists_code) {
376  one_of_two(res);
377  return;
378  }
379  if (q == inputifexists_code) {
380  TokenList A = read_arg();
381  TokenList B = read_arg();
382  if (res) open_tex_file(seen_star);
383  if (A.empty() && B.empty()) return; // optimise
384  if (tracing_commands()) the_log << lg_start_io << (res ? "iftrue" : "iffalse") << "{" << A << "}{" << B << "}\n";
385  one_of_two(A, B, res);
386  return;
387  }
388  if (!res) {
389  parse_error(T, "Cannot open input file ", file, "cannot open input");
390  return;
391  }
392  open_tex_file(seen_star);
393 }

422 auto Parser::scan_double_hat(codepoint c) -> bool {
423  auto sz = input_line.size();
424  auto p = input_line_pos;
425  auto w = sz - p + 1; // number of unread chars aon the line
426  if (w < 3) return false;
427  if (input_line[p] != c) return false;
428  if (w >= 10 && input_line[p + 1] == c && input_line[p + 2] == c && input_line[p + 3] == c) {
429  auto hc0 = input_line[p + 4].hex_val();
430  auto hc1 = input_line[p + 5].hex_val();
431  auto hc2 = input_line[p + 6].hex_val();
432  auto hc3 = input_line[p + 7].hex_val();
433  auto hc4 = input_line[p + 8].hex_val();
434  if (hc0 && hc1 && hc2 && hc3 && hc4) {
435  input_line_pos = p + 8;
436  input_line[p + 8] = codepoint((*hc0 << 16) + (*hc1 << 12) + (*hc2 << 8) + (*hc3 << 4) + *hc4);
437  return true;
438  }
439  }
440  if (w >= 8 && input_line[p + 1] == c && input_line[p + 2] == c) {
441  auto hc1 = input_line[p + 3].hex_val();
442  auto hc2 = input_line[p + 4].hex_val();
443  auto hc3 = input_line[p + 5].hex_val();
444  auto hc4 = input_line[p + 6].hex_val();
445  if (hc1 && hc2 && hc3 && hc4) {
446  input_line_pos = p + 6;
447  input_line[p + 6] = codepoint((*hc1 << 12) + (*hc2 << 8) + (*hc3 << 4) + *hc4);
448  return true;
449  }
450  }
451  if (w >= 4) {
452  auto hc1 = input_line[p + 1].hex_val();
453  auto hc2 = input_line[p + 2].hex_val();
454  if (hc1 && hc2) {
455  input_line_pos = p + 2;
456  input_line[input_line_pos] = codepoint(*hc1 * 16 + *hc2);
457  return true;
458  }
459  }
460  codepoint C = input_line[p + 1];
461  if (!C.is_ascii()) return false;
462  auto c1 = C.value;
463  input_line_pos++;
464  input_line[p + 1] = codepoint(c1 < 64 ? c1 + 64 : c1 - 64);
465  return true;
466 }

468 // This constructs a command name from the current line
469 auto Parser::cs_from_input() -> Token {
470  if (at_eol()) return Token(null_tok_val);
471  codepoint c = get_next_char();
472  state = state_S;
473  if (c.is_big()) { // abort and return null_cs
474  --input_line_pos;
475  return Token(null_tok_val);
476  }
477  int C = get_catcode(c.value);
478  if (C == letter_catcode) {
479  mac_buffer.reset();
480  mac_buffer.push_back(c);
481  for (;;) {
482  if (at_eol()) break;
483  c = get_next_char();
484  if (c.is_big()) {
485  --input_line_pos;
486  break;
487  } // abort
488  C = get_catcode(c.value);
489  if (C == letter_catcode) {
490  mac_buffer.push_back(c);
491  continue;
492  }
493  if (C == hat_catcode) {
494  if (scan_double_hat(c)) continue;
495  }
496  --input_line_pos;
497  break;
498  }
499  return hash_table.locate(mac_buffer);
500  }
501  if (C == space_catcode) return Token(c);
502  if (C == hat_catcode) {
503  if (scan_double_hat(c)) return cs_from_input();
504  state = state_M;
505  return Token(c);
506  }
507  state = state_M;
508  return Token(c);
509 }

511 // This constructs a new token by reading characters from the buffer.
512 // Returns true in case of problem. Either because we are at EOL
513 // or because a line ended (% or \n), or because a character was ignored
514 // (space or invalid chars), or because of ^^
515 auto Parser::next_from_line0() -> bool {
516  if (at_eol()) return true;
517  codepoint c = get_next_char();
518  if (c.is_big()) { // convert to \char"ABCD
519  Buffer &B = local_buf;
520  B.reset();
521  B.push_back('"');
522  B.push_back16(c.value, false);
523  auto k = B.size();
524  back_input(hash_table.space_token);
525  while (k > 0) {
526  back_input(Token(other_t_offset, static_cast<uchar>(B[k - 1])));
527  k--;
528  }
529  state = state_M;
530  see_cs_token(hash_table.char_token);
531  return false;
532  }
533  cur_cmd_chr = CmdChr(get_catcode(c.value), subtypes(c.value));
534  switch (cur_cmd_chr.cmd) {
535  case escape_catcode:
536  cur_tok = cs_from_input();
537  see_cs_token();
538  return false;
539  case active_catcode:
540  cur_tok.active_char(c.value);
541  see_cs_token();
542  state = state_M;
543  return false;
544  case space_catcode:
545  if (state == state_M) {
546  state = state_S;
547  cur_cmd_chr = CmdChr(space_catcode, subtypes(' '));
548  cur_tok = hash_table.space_token;
549  return false;
550  }
551  return true;
552  case comment_catcode: kill_line(); return true;
553  case eol_catcode:
554  kill_line();
555  if (state == state_N) {
556  see_cs_token(hash_table.par_token);
557  return false;
558  }
559  if (state == state_M) {
560  // this is different from TeX
561  cur_cmd_chr = CmdChr(space_catcode, subtypes('\n'));
562  cur_tok = hash_table.newline_token;
563  return false;
564  }
565  return true; // stateS
566  case hat_catcode:
567  if (scan_double_hat(c)) return true; // cur char has changed
568  state = state_M;
569  cur_tok.from_cmd_chr(cur_cmd_chr);
570  return false;
571  case special_catcode: // case of ShortVerb.
572  verb_saved_char = c;
573  see_cs_token(hash_table.verb_token);
574  state = state_M;
575  return false;
576  case ignored_catcode: return true; // ignored
577  case invalid_catcode: return true; // invalid
578  default:
579  state = state_M;
580  cur_tok.from_cmd_chr(cur_cmd_chr);
581  return false;
582  }
583 }

718 // in_env is true, until \end is found a brace level zero (to be read again)

720 // Result is in the buffer.

725 auto Parser::scan_for_eval(Buffer &B, bool in_env) -> bool {
726  int b = 0; // brace level
727  int elevel = 0; // begin/end level at brace level zero when in_env true
728  for (;;) {

730  if (!TL.empty()) {
731  Token t = TL.front();
732  TL.pop_front();

739  elevel--;
740  } else if (t == hash_table.begin_token)
741  elevel++;
742  }
743  // Check brace level

766  if (c == '{') b++;
767  if (c == '}') {

770  B.remove_last();
