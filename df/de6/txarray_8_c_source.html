52 // as \newcommand\cmd[opt]{body}

54 void Parser::T_newcolumn_type() {

71  back_input();

87 // This piece of code expands all the \newcolumntype commands
88 // There is a special case: nct_size=0, case where nothing is
89 // defined, and nothing has to be done. Otherwise, we loop over all
90 // characters. If there is one substitution, we try again.
91 // If for some reason, the character is invalid, we remove it from the table.
92 void Parser::expand_nct(TokenList &L) {
93  bool action = true;
94  if (tracing_commands()) the_log << lg_start << "array preamble at start: " << L << "\n";
95  if (!new_array_object.has_a_nct()) return;
96  int max_iter = max_newcolumn_loops;
97  while (action) {
98  action = false;
99  max_iter--;
100  if (max_iter <= 0) {
101  parse_error("array preamble expansion: Infinite loop?");
102  return;
103  }
104  for (int i = 1; i < int(nb_newcolumn); i++) {
105  uchar c = i;
106  if (!new_array_object.nct_exists(c)) continue;
107  Token T = new_array_object.nct_token(c);
108  TokenList body;
109  int n = nct_aux(T, body);
110  if (n == -1) {
111  new_array_object.remove_a_type(c);
112  continue;
113  }
114  if (token_ns::expand_nct(L, n, c, max_iter, body)) {
115  action = true;
116  if (tracing_commands()) the_log << "array preamble after " << c << ": " << L << "\n";
117  }
118  }
119  }
120 }

122 // Here we have to find the character c.
123 // only top-level characters are considered. Active chars are allowed.
124 // MX is decreased. Job aborted if it becomes negative.
125 auto token_ns::expand_nct(TokenList &L, int n, uchar c, int &MX, TokenList &body) -> bool {
126  TokenList res;
127  bool result = false;
128  TokenList Table[10]; // arguments of the commands
129  while (!L.empty()) {
130  Token t = L.front();
131  L.pop_front();
132  if (t.is_a_left_brace()) { // get a whole block
133  L.push_front(t); // re-enter the brace in the list
134  token_ns::fast_get_block(L, res);
135  continue;
136  }
137  if (t.cmd_val() == 10) continue; // ignore spaces.
138  if (!(t.is_a_char() && t.char_val() == c)) { // just copy
139  res.push_back(t);
140  continue;
141  }
142  result = true; // We found something
143  MX--;
144  if (MX < 0) return true;
145  for (int k = 0; k < n; k++) Table[k + 1] = get_a_param(L, false);
146  TokenList W = the_parser.expand_mac_inner(body, Table);
147  L.splice(L.begin(), W);
148  }
149  L.splice(L.end(), res);
150  return result;
151 }

153 // Prints a (ch_class, ch_num) pair for debug.
154 auto NewArray::dump_slot() -> String {
155  switch (ch_class) {
156  case chc_cell:
157  switch (ch_num) {
158  case chn_c: return "c ";
159  case chn_l: return "l ";
160  case chn_r: return "r ";
161  case chn_m: return "m{} ";
162  case chn_p: return "p{} ";
163  case chn_b: return "b{} ";
164  default: return "impossible!";
165  }
166  case chc_bar: return ch_num == 0 ? "| " : "!{} ";
167  case chc_v: return "<{} ";
168  case chc_u: return ">{} ";
169  case chc_start: return "impossible!";
170  case chc_inter: return "@{} ";
171  case chc_pre_bar: return "!";
172  case chc_pre_inter: return "@";
173  case chc_pre_v: return "<";
174  case chc_pre_u: return ">";
175  case chc_pre_cell:
176  if (ch_num == chn_m) return "m";
177  if (ch_num == chn_p) return "p";
178  if (ch_num == chn_b) return "b";
179  return "impossible!";
180  default: return "impossible!";
181  }

184 // This sets (ch_class,ch_num) according to last_ch_class (case where
185 // a brace-enclosed expression is expected) or the next token
186 // (in current_token, or the first element of current_list)
187 void NewArray::test_pach() {
188  if (last_ch_class == chc_pre_bar) {
189  ch_class = chc_bar;
190  ch_num = chn_l;
191  return;
192  }
193  if (last_ch_class == chc_pre_inter) {
194  ch_class = chc_inter;
195  return;
196  }
197  if (last_ch_class == chc_pre_v) {
198  ch_class = chc_v;
199  return;
200  }
201  if (last_ch_class == chc_pre_u) {
202  ch_class = chc_u;
203  return;
204  }
205  ch_class = chc_cell; // we should have a cell here.
206  if (last_ch_class == chc_pre_cell) return; // was ok
207  if (have_token_list) {
208  if (current_list.empty())
209  current_token = Token();
210  else
211  current_token = current_list.front();
212  }
213  uint c = 0;
214  if (current_token.cmd_val() == 11 || current_token.cmd_val() == 12 || current_token.cmd_val() == 13) c = current_token.char_val().value;
215  if (c == 'c') {
216  cur_h_pos = np_c_center;
217  ch_num = chn_c;
218  return;
219  }
220  if (c == 'l') {
221  cur_h_pos = np_c_left;
222  ch_num = chn_l;
223  return;
224  }
225  if (c == 'r') {
226  cur_h_pos = np_c_right;
227  ch_num = chn_r;
228  return;
229  }
230  ch_num = chn_c; // in case of error
231  // These want an argument
232  char_for_error = c;
233  if (c == '|') {
234  ch_class = chc_bar;
235  return;
236  }
237  if (c == '!') {
238  ch_class = chc_pre_bar;
239  return;
240  }
241  if (c == '@') {
242  ch_class = chc_pre_inter;
243  return;
244  }
245  if (c == '<') {
246  ch_class = chc_pre_v;
247  return;
248  }
249  if (c == '>') {
250  ch_class = chc_pre_u;
251  return;
252  }
253  ch_class = chc_pre_cell;
254  if (c == 'm') {
255  ch_num = chn_m;
256  return;
257  }
258  if (c == 'p') {
259  ch_num = chn_p;
260  return;
261  }
262  if (c == 'b') {
263  ch_num = chn_b;
264  return;
265  }
266  ch_class = chc_cell;
267  the_parser.parse_error(the_parser.err_tok, "Wrong character in array preamble ", current_token, "", "Wrong char");
268 }

270 // Big function that handles the array preamble. We first read and
271 // expand the preamble, then loop over all characters, then check
272 // that it ends correctly. If main_fct is false, we are in a multicolumn.
273 // In this case, the array preamble should specify only material
274 // for one cell. We read the content C of the cell, and push back
275 // the {UCV} list (where U and V the optional <u> and <v> parts.
276 // braces are added because cells are assumed to be typeset in a group.
277 void NewArray::run(Xid ID, bool main_fct) {
278  id = ID;
279  if (!main_fct) { // read and set the column span
280  Istring x = P->nT_arg_nopar();
281  const std::string &s = the_main->SH[x.get_value()];
282  if (s != "1") id.add_attribute(the_names[np_cols], x);
283  }
284  preamble = P->read_arg(); // read the preamble
285  token_ns::expand_star(preamble);
286  P->expand_nct(preamble);
287  if (main_fct) AI = &P->the_stack.new_array_info(id);
288  u_list.clear();
289  v_list.clear();
290  ac_cell_no = 0;
291  last_ch_class = chc_start;
292  first_bar = true;
293  bool seen = false;
294  cur_h_pos = np_c_center;
295  if (P->tracing_commands()) the_log << "array preamble parse: ";
296  for (;;) {
297  if (preamble.empty()) break;
298  if (ac_next()) continue;
299  test_pach();
300  if (P->tracing_commands()) the_log << dump_slot();
301  switch (ch_class) {
302  case chc_cell:
303  if (main_fct)
304  ac_maybe_finish(); // finish previous
305  else
306  ac_maybe_finish_multi(seen);
307  attribs.push_back(np_halign, cur_h_pos, false); // set attributes
308  cur_h_pos = np_c_center; // add a default
309  current_list.clear();
310  break;
311  case chc_bar:
312  if (ch_num == chn_c) {
313  if (last_ch_class == chc_start)
314  attribs.push_back(np_leftborder, np_true);
315  else if (last_ch_class == chc_bar) {
316  } else
317  attribs.push_back(np_rightborder, np_true, false);
318  }
319  current_list.clear();
320  break;
321  case chc_v: // got a <v>list, add it
322  if (!have_token_list)
323  v_list.push_front(current_token);
324  else
325  v_list.splice(v_list.begin(), current_list);
326  break;
327  case chc_u: // got a <u>list, add it.
328  if (!have_token_list)
329  u_list.push_front(current_token);
330  else
331  u_list.splice(u_list.begin(), current_list);
332  break;
333  case chc_start: // preamble empty
334  break;
335  case chc_inter: current_list.clear(); break;
336  case chc_pre_bar:
337  if (last_ch_class == chc_cell)
338  break;
339  else if (last_ch_class == chc_bar)
340  break;
341  else if (last_ch_class == chc_v)
342  break;
343  else if (last_ch_class == chc_u)
344  P->parse_error("illegal ! or @");
345  break;
346  case chc_pre_inter:
347  if (last_ch_class == chc_u) P->parse_error("illegal ! or @");
348  break;
349  case chc_pre_v:
350  if (last_ch_class == chc_cell) break;
351  if (last_ch_class == chc_v) break;
352  P->parse_error("unexpected less-than sign");
353  ch_class = chc_pre_bar;
354  break;
355  case chc_pre_u:
356  case chc_pre_cell:
357  if (main_fct)
358  ac_maybe_finish();
359  else
360  ac_maybe_finish_multi(seen);
361  break;
362  }
363  last_ch_class = ch_class;
364  }
365  switch (last_ch_class) {
366  case chc_cell:
367  case chc_bar:
368  case chc_v:
369  case chc_u:
370  case chc_start:
371  case chc_inter: break;
372  default: {
373  errbuf.reset();
374  errbuf << "Array preamble: argument missing for " << char_for_error;
375  P->parse_error(errbuf.c_str());
376  }
377  }
378  if (P->tracing_commands()) the_log << "\n";
379  if (main_fct) {
380  ac_maybe_finish();
381  return;
382  }
383  ac_maybe_finish_multi(seen);
384  // Case of \multicolumn,
385  id.add_attribute(attribs, true);
386  attribs.reset();
387  TokenList cell = P->read_arg();
388  cell.splice(cell.begin(), u_list);
389  cell.splice(cell.end(), v_list);
390  u_list.clear();
391  v_list.clear();
392  P->brace_me(cell);
393  P->back_input(cell);
394 }

397 // case 1 we see a >
398 // case 2 we see m, p or b :
399 // case 3 we see m{} p{} b{} or clr : more will be done later

401 void NewArray::ac_maybe_finish() {

408  attribs.reset();
409  u_list.clear();
410  v_list.clear();
411  ac_cell_no++;
412  }

427 // Return true if this is a space.
428 // otherwise get the next token in current_token
429 // if it is a brace, get the list in current_list
430 auto NewArray::ac_next() -> bool {
431  current_token = preamble.front();
432  if (current_token.cmd_val() == 10) {
433  preamble.pop_front();
434  return true;
435  }
436  // strange
437  if (current_token == the_parser.hash_table.verb_token) {
438  preamble.pop_front();
439  return true;
440  }
441  if (current_token.is_a_left_brace()) {
442  current_list = token_ns::fast_get_block(preamble);
443  token_ns::remove_ext_braces(current_list);
444  have_token_list = true;
445  } else {
446  have_token_list = false;
447  preamble.pop_front();
448  }
449  return false;
450 }

478 // Start a row. If a=-1, it is the first row
479 // If a>0 it is a dimension to put between rows.
480 void Parser::start_a_row(int a) {
481  Xid prev_row(2); // dummy id as default value
482  {
483  Xml *V = the_stack.top_stack()->last_addr();
484  if (V != nullptr) prev_row = V->get_id();
485  }
486  bool initial_hline = false;
487  if (a > 0) prev_row.add_attribute(the_names[np_spaceafter], Istring(a));
488  for (;;) {
489  remove_initial_space_and_back_input(); // get first non-space xtoken
490  symcodes S = cur_cmd_chr.get_cmd();
491  if (S == ifnextchar_cmd) {
492  get_token();
493  T_ifnextchar(true);
494  continue;
495  }
496  if (S == hline_cmd) {
497  SaveErrTok sv(cur_tok);
498  subtypes c = cur_cmd_chr.get_chr();
499  get_token();
500  int w = T_hline_parse(c);
501  if (w == 2) T_cline();
502  if (w == 1) { // a hline
503  if (a < 0)
504  initial_hline = true;
505  else
506  prev_row.add_bottom_rule();
507  }
508  continue;
509  }
510  if (S == end_cmd) return;
511  the_stack.push1(np_row);
512  if (initial_hline) the_stack.cur_xid().add_top_rule();
513  start_a_cell(false);
514  return;
515  }
516 }

589 // implements \hline, \cline{p} or \hlinee[p]{w}{w}{w}
590 // p is a sequence of two integers, separated by a dash
591 // sets cline_first and cline_last;
592 // fills errbuf and returns a code in case of error
593 
594 auto Parser::scan_pair_ints(Token T, TokenList &L) -> bool {
595  back_input(hash_table.relax_token);
596  back_input(L);
597  cline_first = scan_int(T);
598  if (get_token()) {
599  errbuf.reset();
600  errbuf << "Error in " << T << " after first integer";
601  return true;
602  }
603  if (cur_tok == hash_table.relax_token) {
604  errbuf.reset();
605  errbuf << "Error in " << T << " after first integer";
606  return true;
607  }

669 // Case of cline, placed after a row. If action is false, we check whether the
670 // row, including the spans of the cells, is adapted to the pattern.
671 // If action is true, we do something
672 auto Xml::try_cline(bool action) -> bool {
673  int a = cline_first - 1;
674  bool a_ok = false; // true after skip
675  int len = size();
676  for (int k = 0; k < len; k++) {
677  if (a == 0) {
678  if (a_ok) return true;
679  a = (cline_last - cline_first) + 1;
680  a_ok = true;
681  }
682  if (tree[k] == nullptr) continue;
683  if (tree[k]->is_xmlc()) {
684  Istring N = tree[k]->get_name();
685  if (strcmp(N.c_str(), "\n") == 0) continue; // allow newline separator
686  return false;
687  }
688  int c = tree[k]->get_cell_span();
689  if (c == -1) return false;
690  if (c == 0) continue; // ignore null span cells
691  if (a_ok && action) tree[k]->get_id().add_bottom_rule();
692  a = a - c;
693  if (a < 0) return false;
694  }
695  return false; // list too small
696 }
