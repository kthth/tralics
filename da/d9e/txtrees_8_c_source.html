81 // This normalises spaces
82 void trees_ns::normalise_space(TokenList &L) {
83  bool prev_space = false;
84  TokenList res;
85  auto SP = Token(space_token_val);
86  auto C = L.begin();
87  auto E = L.end();
88  while (C != E) {
89  if (C->is_space_token()) {
90  if (!prev_space) res.push_back(SP);
91  prev_space = true;
92  } else {
93  res.push_back(*C);
94  prev_space = false;
95  }
96  ++C;
97  }
98  L.swap(res);
99 }

103 auto Parser::index_aux(TokenList &L, long father, size_t g) -> size_t {
104  static const Token escape_t(other_t_offset, '"');
105  static const Token actual_t(other_t_offset, '@');
106  static const Token actualb_t(letter_t_offset, '@');
107  static const Token encap_t(other_t_offset, '|');
108  std::string key, encap;
109  bool have_key = false;
110  TokenList z;
111  if (token_ns::split_at(escape_t, actual_t, actualb_t, L, z, false)) {
112  have_key = true;
113  token_ns::remove_first_last_space(z);
114  key = to_stringE(z);
115  }
116  if (token_ns::split_at(escape_t, encap_t, encap_t, L, z, false)) {
117  token_ns::remove_first_last_space(L);
118  encap = to_stringE(L);
119  swap(L, z);
120  }
121  if (!have_key) {
122  z = L;
123  token_ns::remove_first_last_space(z);
124  key = to_stringE(z);
125  }
126  z = L;
127  std::string aux = to_stringE(z);
128  // We have now: key@aux|encap
129  std::vector<Indexer *> &IR = the_index.get_data(g);
130  int level = 1;
131  if (father != -1) level = 1 + IR[to_unsigned(father)]->level;
132  auto n = IR.size();
133  for (size_t i = 0; i < n; i++)
134  if (IR[i]->is_same(level, aux)) return i;
135  Buffer &B = local_buf;
136  B.reset();
137  if (father != -1) B << IR[to_unsigned(father)]->key << "____";
138  B << key;
139  B.lowercase();
140  B.no_newline();
141  Xml *res = translate_list(L);
142  Xml *x = new Xml(np_index, res);
143  if (!encap.empty()) x->get_id().add_attribute(np_encap, Istring(encap));
144  x->get_id().add_attribute(np_level, name_positions(cst_dig0 + level));
145  auto iid = the_index.next_iid();
146  IR.push_back(new Indexer(B.to_string(), aux, x, level, iid));
147  return n;
148 }

150 // \index, \makeindex, \printindex, + glossary, and \newindex
151 void Parser::T_index(subtypes c) {
152  if (c == makeindex_code || c == makeglossary_code) return;
153  flush_buffer();
154  if (c == index_code) remove_initial_star();
155  if (c == newindex_code) {
156  std::string s = sT_arg_nopar();
157  std::string title = sT_arg_nopar();
158  the_index.new_index(s, title);
159  return;
160  }
161  size_t g = 0;
162  if (c == printindex_code || c == index_code) {
163  std::string s = sT_optarg_nopar();
164  g = the_index.find_index(s);
165  }
166  if (c == printindex_code || c == printglossary_code) {
167  the_index.mark_print(g);
168  return;
169  }
170  TokenList L = read_arg();
171  trees_ns::normalise_space(L);
172  TokenList z1, z2;
173  static const Token escape_t(other_t_offset, '"');
174  static const Token level_t(other_t_offset, '!');
175  flush_buffer();
176  int level = 1;
177  if (token_ns::split_at(escape_t, level_t, level_t, L, z1, true)) {
178  if (token_ns::split_at(escape_t, level_t, level_t, L, z2, true))
179  level = 3; // z1, z2, L
180  else
181  level = 2; // z1, L
182  }
183  // If level != 1, make sure parent exists
184  long position = -1;
185  if (level > 1) position = to_signed(index_aux(z1, position, g));
186  // If level == 3, make sure grand parent exists
187  if (level == 3) position = to_signed(index_aux(z2, position, g));
188  // make sure this exists
189  position = to_signed(index_aux(L, position, g));
190  // Now, add a label here
191  auto iid = the_index.get_data(g)[to_unsigned(position)]->iid;
192  int nid = the_index.next_index();
193  ;
194  local_buf << bf_reset << "lid" << nid;
195  std::string W = local_buf.to_string();
196  Istring id = the_stack.add_anchor(W, false);
197  create_label(W, id);
198  tralics_ns::add_ref(to_signed(iid), W, true);
199 }

201 void Parser::finish_index() {
202  auto labels = std::vector<std::string>(the_index.get_last_iid(), "");
203  tralics_ns::find_index_labels(labels);
204  int idx_size = 0;
205  int idx_nb = 0;
206  auto q = the_index.size();
207  for (size_t jj = 1; jj <= q; jj++) {
208  auto j = jj == q ? 0 : jj;
209  auto CI = the_index.get_index(j);
210  auto n = CI->size();
211  if (n == 0) continue;
212  idx_size += n;
213  idx_nb++;
214  Xml *res = new Xml(j == 0 ? np_theglossary : np_theindex, nullptr); // OK?
215  Xid id = res->get_id();
216  {
217  const std::string &t = CI->get_title();
218  if (!t.empty()) id.add_attribute(the_names[cstb_title], Istring(t));
219  }
220  {
221  AttList &L = the_stack.get_att_list(CI->get_AL());
222  id.add_attribute(L, true);
223  }
224  for (size_t i = 0; i < n; i++) {
225  Xml *A = CI->get_translation(i);
226  A->get_id().add_attribute(np_target, Istring(labels[CI->get_iid(i)]));
227  }
228  CI->do_sort();
229  for (size_t i = 0; i < n; i++) {
230  Xml *A = CI->get_translation(i);
231  res->push_back(A);
232  res->add_nl();
233  }
234  the_stack.document_element()->insert_bib(res, CI->get_position());
235  }
236  if (idx_size == 0) return;
237  log_and_tty << "Index has " << idx_size << " entries";
238  if (idx_nb > 1) log_and_tty << " in " << idx_nb << " clusters";
239  log_and_tty << "\n";
240 }

387 void Parser::T_gloss(bool c) {
388  Token A = hash_table.locate("eachwordone");
389  Token B = hash_table.locate("eachwordtwo");
390  Token C = hash_table.locate("eachwordthree");
391  Token amp = Token(ampersand_t_offset, '&');
392  ;
393  Token BS = hash_table.locate("\\");
394  TokenList first_line = read_until(BS);
395  TokenList second_line = read_until(BS);
396  TokenList third_line;
397  int n1 = 0, n2 = 0, n3 = 0;
398  if (!c) third_line = read_until(BS);
399  n1 = token_ns::replace_space(first_line, amp, A);
400  first_line.push_front(A);
401  first_line.push_back(BS);
402  n2 = token_ns::replace_space(second_line, amp, B);
403  second_line.push_front(B);
404  second_line.push_back(BS);
405  if (!c) {
406  n3 = token_ns::replace_space(third_line, amp, C);
407  third_line.push_front(C);
408  third_line.push_back(BS);
409  }
410 
411  if (n3 > n1) n1 = n3;
412  if (n2 > n1) n1 = n2;
413  n1++;
414  TokenList res;
415  res.push_back(hash_table.OB_token);
416  while (n1 > 0) {
417  --n1;
418  res.push_back(Token(other_t_offset, 'l'));
419  }
420  res.push_back(hash_table.CB_token);
421  res.splice(res.end(), first_line);
422  res.splice(res.end(), second_line);
423  res.splice(res.end(), third_line);
424  token_ns::add_env(res, "tabular");
425  if (tracing_commands()) the_log << lg_startbrace << "Gloss: " << res << lg_endbrace;
426  back_input(res);
427 }

552 // Return true if valid, signals error otherwise
553 auto date_ns::check_date(long y, long m, long d) -> bool {
554  String Bad = nullptr;
555  long ml = 0;
556  if (y <= 0)
557  Bad = "year<1";
558  else if (m <= 0)
559  Bad = "month<1";
560  else if (m > 12)
561  Bad = "month>12";
562  else if (d < 1)
563  Bad = "day<1";
564  else {
565  ml = date_ns::month_length(y, m);
566  if (d > ml)
567  Bad = ".";
568  else if (y == 1582 && m == 10 && d > 4 && d < 15)
569  Bad = "inexistant day";
570  }
571  if (Bad == nullptr) return true;
572  local_buf << bf_reset << "Date error: ";
573  if (Bad[0] == '.')
574  local_buf << "day>" << ml;
575  else
576  local_buf << Bad;
577  local_buf << " " << y << '/' << m << '/' << d;
578  the_parser.parse_error(the_parser.err_tok, local_buf.to_string());
579  return false;
580 }

582 // Returns the number of days between start/01/01 and cur/month/day
583 void Parser::count_days() {
