44 // return tl_end if seen End, tl_empty if empty (or comment),
45 // tl_normal otherwise
46 auto Buffer::tp_fetch_something() -> tpa_line {
47  ptr = 0;
48  if (strncmp(data(), "End", 3) == 0) return tl_end;
49  skip_sp_tab();
50  if (is_special_end()) return tl_empty;
51  if (ptr == 0) {
52  tpage_ns::init_error();
53  log_and_tty << data();
54  log_and_tty << "Wanted End, or line starting with space\n";
55  return tl_empty;
56  }
57  return tl_normal;
58 }

60 // This is the function that creates the title page data
61 // from a list of lines
62 void tralics_ns::Titlepage_create(LinePtr &lines) {
63  if (lines.is_empty()) return;
64  Titlepage.make_valid();
65  for (;;) {
66  tp_main_buf.reset();
67  int line = lines.get_next(tp_main_buf);
68  if (line < 0) return;
69  init_file_pos = line;
70  tpa_line k = tp_main_buf.tp_fetch_something();
71  if (k == tl_empty) continue;
72  if (k == tl_end) break;
73  int w0 = tpfl.read();
74  tpi_vals w = tpfl.classify(w0, Titlepage.state);
75  if (w == tpi_err) {
76  tpage_ns::init_error();
77  continue;
78  }
79  TitlePageAux tpa(tpfl);
80  bool res = tpa.classify(w, Titlepage.state);
81  if (!res) {
82  tpage_ns::init_error();
83  continue;
84  }
85  Titlepage.bigtable.push_back(tpa);
86  }
87 }

103 // Gets next non space. return false if EOL
104 auto Buffer::tp_next_char(char &res) -> bool {
105  skip_sp_tab();
106  if (is_special_end()) return false;
107  res = head();
108  advance();
109  return true;

112 // Returns an item that start with del
113 // result is in local_buf, return value is true if OK.
114 auto tpage_ns::scan_item(Buffer &in, Buffer &out, char del) -> bool {
115  out.reset();
116  if (del == '\\') { // scan a command
117  while (is_letter(in.head())) out.push_back(in.next_char());
118  return true;
119  }
120  if (del == '<') del = '>'; // else del is ""
121  while (in.head() != del) {
122  if (in.head() == 0) {
123  tpage_ns::init_error();
124  log_and_tty << "could not find end delimiter\n";
125  out << bf_reset << "notfound";
126  return false;
127  }
128  out.push_back(in.next_char());
129  }
130  in.advance();
131  return true;
132 }

134 // Scans an item: prefix plus value. Result in Toi
135 // returns the type of the thing to create.
136 auto tpage_ns::next_item(Buffer &in, Buffer &out) -> tpi_vals {
137  Toi.reset();
138  char c;
139  if (!in.tp_next_char(c)) return tpi_noval;
140  if (c == 'a' && in.is_here("lias")) return tpi_alias;
141  if (c == 'e' && in.is_here("xecute")) return tpi_execute;
142  if (c == 'a' && in.is_here("ction")) return tpi_execute;
143  if (!is_tp_delim(c)) {
144  Toi.set_p1(c);
145  if (!in.tp_next_char(c)) return tpi_noval;
146  if (!is_tp_delim(c)) {
147  Toi.set_p2(c);
148  if (!in.tp_next_char(c)) return tpi_noval;
149  if (!is_tp_delim(c)) return tpi_err;
150  }
151  }
152  if (!tpage_ns::scan_item(in, out, c)) return tpi_err;
153  Toi.set_value(out);
154  if (c == '<') return tpi_elt;
155  if (c == '\\') return tpi_cmd;
156  return tpi_str;
157 }

159 // This reads up to four items.
160 // returns a number that tells what types have been read.
161 auto TitlePageFullLine::read() -> int {
162  item1.reset();
163  item2.reset();
164  item3.reset();
165  item4.reset();
166  int res = 0;
167  tpi_vals w = tpage_ns::next_item(tp_main_buf, local_buf);
168  if (w == tpi_err) return -1;
169  item1 = Toi;
170  item1.set_v(w);
171  res += 1000 * (w - 1);
172  if (w == tpi_noval) return res;
173  w = tpage_ns::next_item(tp_main_buf, local_buf);
174  if (w == tpi_err) return -1;
175  item2 = Toi;
176  item2.set_v(w);
177  res += 100 * (w - 1);
178  if (w == tpi_noval) return res;
179  w = tpage_ns::next_item(tp_main_buf, local_buf);
180  if (w == tpi_err) return -1;
181  item3 = Toi;
182  item3.set_v(w);
183  res += 10 * (w - 1);
184  if (w == tpi_noval) return res;
185  w = tpage_ns::next_item(tp_main_buf, local_buf);
186  if (w == tpi_err) return -1;
187  item4 = Toi;
188  item4.set_v(w);
189  res += (w - 1);
190  if (w == tpi_noval) return res;
191  char c;
192  if (tp_main_buf.tp_next_char(c)) return -1;
193  return res;
194 }

196 // First attempt at classification
197 // E=1 C=2 S=3 A=4 e=5
198 auto TitlePageFullLine::classify(int w, int state) -> tpi_vals {
199  if (w == -1) return tpi_err; // forget about this
200  flags = 0;
201  if (item1.has_a_char() || item4.has_a_char()) return tpi_err;
202  if (w == 0) return tpi_zero; // ignore this item
203  if (w == 2133) {
204  if (state != 0) return tpi_err;
205  } else {
206  if (state == 0) return tpi_err;
207  }
208  if (w == 1000) return item2.only_dash() ? tpi_E : tpi_err;
209  if (w == 3000) return tpi_S;
210  if (w == 4200 || w == 4233) {
211  if (item2.has_a_char() || item3.has_a_char()) return tpi_err;
212  if (state == 1 && w == 4233) return tpi_ACSS;
213  if (state == 2 && w == 4200) return tpi_AC;
214  } else if (w == 2230 || w == 2133) {
215  if (item2.has_a_char() || item3.has_a_char()) return tpi_err;
216  return w == 2230 ? tpi_CCS : tpi_CESS;
217  } else if (w == 2100)
218  return item2.quest_plus() && !item3.has_a_char() ? tpi_CE : tpi_err;
219  else if (w == 5200)
220  return !item2.has_a_char() ? tpi_EC : tpi_err;
221  else if (w == 2110)
222  return item2.question() && !item3.has_a_char() ? tpi_CEE : tpi_err;
223  else if (w == 2113)
224  return item2.plus() && !item3.has_a_char() ? tpi_CEES : tpi_err;
225  if (w != 2130) return tpi_err;
226  if (!encode_flags(item2.get_p1(), item3.get_p1())) return tpi_err;
227  return tpi_CES;
228 }

230 // More classification. This allocates memory, and modifies state
231 auto TitlePageAux::classify(tpi_vals w, int &state) -> bool {
232  switch (w) {
233  case tpi_CESS: // \makeRR <rrstart> "ok" "ok"
234  state = 1;
235  idx = 0;
236  type = tpi_rt_tp;
237  return true;
238  case tpi_ACSS: // alias \makeRT "ok" "ok"
239  T1 = T2;
240  T2 = Titlepage.bigtable[0].get_T2();
241  type = tpi_rt_tp;
242  return true; // state is still 1
243  case tpi_CES: // \title <title> "notitle"
244  state = 2;
245  type = tpi_rt_normal;
246  T4 = T3;
247  T3 = T2;
248  idx = Titlepage.increase_data();
249  if (get_flags2() == tp_A_flag) {
250  Buffer &B = local_buf;
251  B << bf_reset << "\\" << T1;
252  B.push_back_braced(T4);
253  B.push_back("%\n");
254  the_main->add_to_from_config(1, B);
255  }
256  return true;
257  case tpi_CEES: // \author +<aulist> <au> "default"
258  state = 2;
259  type = tpi_rt_list;
260  idx = Titlepage.increase_data();
261  return true;
262  case tpi_E: // <UR> -
263  state = 3; // next line cannot be alias
264  type = tpi_rt_urlist;
265  idx = Titlepage.increase_data();
266  T2 = T1;
267  return true;
268  case tpi_CE: // case \URsop ?+ <UR>
269  {
270  int k = Titlepage.find_UR(T2, T1);
271  if (k == 0) return false;
272  state = 2;
273  type = tpi_rt_ur;
274  idx = k;
275  T2 = local_buf.to_string();
276  return true;
277  }
278  case tpi_S: // case "<foo/>"
279  type = tpi_rt_constant;
280  idx = Titlepage.increase_data();
281  return true;
282  case tpi_EC: // case execute \foo
283  type = tpi_rt_exec;
284  idx = Titlepage.increase_data();
285  return true;
286  case tpi_CEE: // \Paris ?<UR> <Rocqu>
287  {
288  int k = Titlepage.find_UR(T2, "");
289  if (k == 0) return false;
290  state = 2;
291  type = tpi_rt_ur;
292  idx = k;
293  T2 = T3;
294  T3 = "";
295  return true;
296  }
297  case tpi_CCS: // case \myself \author "JG"
298  {
299  int k = Titlepage.find_cmd(T2);
300  if (k == 0) return false;
301  TitlePageAux &R = Titlepage.bigtable[k];
302  state = 2;
303  idx = R.idx;
304  xflags = k; // hack...
305  T2 = T3;
306  T3 = R.T3;
307  type = R.type == tpi_rt_normal ? tpi_rt_normal_def : tpi_rt_list_def;
308  }
309  return true;
310  case tpi_AC:
311  type = tpi_rt_alias;
312  T1 = T2;
313  T2 = Titlepage.bigtable.back().get_T1();
314  return true;
315  default: return false;
316  }
317 }

319 // Evaluates at start of translation
320 void TitlePageAux::exec_start(int k) {
321  if (type == tpi_rt_urlist) { // no command here
322  Titlepage[idx] = convert(2, new Xml(the_names[cst_nl]));
323  return;
324  }
325  if (type == tpi_rt_alias) { // \let\this\that
326  the_parser.hash_table.eval_let(T1.c_str(), T2.c_str());
327  return;
328  }
329  // in all other cases define this as (titlepage,k)
330  the_parser.hash_table.primitive(T1.c_str(), titlepage_cmd, subtypes(k));
331  if (type == tpi_rt_list) { // initialise the list
332  Titlepage[idx] = convert(2, convert(3, Istring(T4)));
333  return;
334  }
335  // remaining types are: tp, ur, normal_def, list_def, and normal
336  if (type != tpi_rt_normal) return;
337  int fl = get_flags2(); // this is 0,1, 2 or 3
338  if (fl == 0) {
339  Titlepage[idx] = convert(2, Istring(T4));
340  return;
341  }
342  Buffer &B = local_buf;
343  B << bf_reset << "\\" << T1;
344  B.push_back_braced(T4);
345  Titlepage[idx] = new Xml(Istring("empty"));
346  B.push_back("%\n");
347  if (fl == tp_A_flag) {
348  } // already done
349  // the_main->add_to_from_config(1,B);
350  else if (fl == tp_B_flag)
351  the_parser.add_buffer_to_document_hook(B, "(tpa init)");
352  else
353  T4 = B.to_string();
354 }

356 // This is executed when we see the \Titlepage cmd
357 void TitlePageAux::exec_post() {
358  if (type == tpi_rt_constant) Titlepage[idx] = new Xml(Istring(T1));
359  if (type == tpi_rt_exec) Titlepage[idx] = the_parser.tpa_exec(T2.c_str());
360  if (type != tpi_rt_normal) return;
361  if (get_flags2() == tp_C_flag) the_parser.titlepage_evaluate(T4.c_str(), T1);
362  if (!has_plus_flags()) return;
363  if (has_u_flags()) return;
364  the_parser.parse_error(the_parser.err_tok, "No value given for command \\", T1, "");
365 }

367 // This is executed when the user asks for a titlepage command.
368 void TitlePageAux::exec(int v, bool vb) {
369  if (vb) the_log << lg_startbrace << "\\titlepage " << v << "=\\" << T1 << lg_endbrace;
370  if (type == tpi_rt_tp) {
371  the_parser.T_titlepage_finish(v);
372  return;
373  }
374  if (type == tpi_rt_ur) { // easy case
375  Titlepage[idx]->add_last_nl(convert(2));
376  return;
377  }
378  Xml *R = nullptr;
379  if (type == tpi_rt_normal_def || type == tpi_rt_list_def)
380  R = convert(3, Istring(T2));
381  else { // we have to read the argument.
382  R = the_parser.special_tpa_arg(T1.c_str(), T3.c_str(), has_p_flags(), has_e_flags(), has_q_flags());
383  }
384  bool replace = (type == tpi_rt_normal || type == tpi_rt_normal_def);
385  if (increment_flag()) {
386  if (!replace) Titlepage[idx]->remove_last();
387  }
388  if (replace)
389  Titlepage[idx] = R;
390  else
391  Titlepage[idx]->add_last_nl(R);
392 }

416 // This is executed when we see the \titlepage command,
417 // After that, no more titlepage ...
418 void Parser::T_titlepage_finish(int v) {
419  int kmax = Titlepage.bigtable.size();
420  for (int k = 0; k < kmax; k++) Titlepage.bigtable[k].exec_post();
421  add_language_att();
422  TitlePageAux &tpa = Titlepage.bigtable[v];
423  std::string tmp = tpa.get_T4();
424  bool finished = false;
425  bool also_bib = false;
426  if (strstr(tmp.c_str(), "'only title page'") != nullptr) finished = true;
427  if (strstr(tmp.c_str(), "'translate also bibliography'") != nullptr) also_bib = true;
428  Xid(1).add_special_att(tmp);
429  Xml *res = tpa.convert(2);
430  res->get_id().add_special_att(tpa.get_T3());
431  kmax = Titlepage.get_len2();
432  for (int k = 1; k < kmax; k++) res->add_last_nl(Titlepage[k]);
433  the_stack.pop_if_frame(the_names[cst_p]);
434  the_stack.add_nl();
435  the_stack.add_last(res);
436  the_stack.add_nl();
437  ileave_v_mode();
438  Titlepage.make_invalid();
439  if (the_main->get_tpa_mode() == 1) return;
440  if (the_main->get_tpa_mode() == 2)
441  finished = true;
442  else if (the_main->get_tpa_mode() == 0) {
443  if (main_ns::nb_errs > 0) finished = true;
444  }
445  if (finished && also_bib) {
446  tralics_ns::bibtex_set_nocite();
447  tralics_ns::bibtex_insert_jobname();
448  }
449  if (finished) {
450  log_and_tty << "Translation terminated after title page\n";
451  E_input(end_all_input_code);
452  }
453 }

495 // For the case CE, \URsop ?+ <UR myflags>
496 // We put `URsop myflags' in the buffer local_buf in case of success.
497 auto TitlePage::find_UR(const std::string &s, const std::string &name) const -> int {
498  Buffer &B = local_buf;
499  B << bf_reset << s;
500  int j = 0;
501  while ((B[j] != 0) && !is_space(B[j])) j++;
502  bool have_space = B[j] != 0;
503  B.kill_at(j);
504  String match = B.c_str();
505  int res = 0;
506  for (const auto &k : bigtable) {
507  res = k.find_UR(match, j);
508  if (res != 0) break;
509  }
510  if (res == 0) return 0;
511  if (!name.empty()) {
512  std::string w = B.to_string(j + 1);
513  B << bf_reset << name;
514  if (have_space) B << ' ' << w;
515  }
516  return res;
517 }

553 // This compares a Begin line with the string s.
554 // Returns : 0 not a begin; 1 not this type; 2 not this object
555 // 3 this type; 4 this object; 5 this is a type
556 auto Buffer::is_begin_something(String s) -> int {
557  if (strncmp("Begin", data(), 5) != 0) return 0;
558  if (strncmp("Type", data() + 5, 4) == 0) {
559  ptr = 9;
560  skip_sp_tab();
561  if (ptr == 9) return 2; // bad
562  set_ptr1();
563  skip_letter();
564  if (ptr == ptr1) return 2; // bad
565  kill_at(ptr); // what follows the type is a comment
566  if (s == nullptr) return 5; // s=0 for type lookup
567  if (strcmp(data() + ptr1, s) == 0) return 3;
568  return 1;
569  }
570  if (s == nullptr) return 0;
571  ptr = 5;
572  set_ptr1();
573  skip_letter();
