50 // Splits key=val in two parts
51 auto classes_ns::make_keyval(TokenList &key_val) -> KeyAndVal {
52  TokenList key;
53  Token equals = Token(other_t_offset, '=');
54  bool have_equals = split_at(equals, key_val, key);
55  remove_first_last_space(key_val);
56  remove_first_last_space(key);
57  if (have_equals) key_val.push_front(equals);
58  std::string key_name = the_parser.list_to_string_c(key, "Invalid option");
59  std::string key_full = key_name;
60  if (have_equals) {
61  Buffer &B = local_buf;
62  B << bf_reset << key_name << key_val;
63  key_full = B.to_string();
64  }
65  return KeyAndVal(key_name, key_val, key_full);
66 }

210 // Date is something like 2004/12/03 converted to 20041203
211 // Hack: we read at most 8 digits, ignore everything else
212 auto classes_ns::parse_version(const std::string &s) -> int {
213  auto n = s.size();
214  int r = 0;
215  int k = 0;
216  for (size_t i = 0; i < n; i++) {
217  char c = s[i];
218  if (is_digit(c)) {
219  r = 10 * r + (c - '0');
220  k++;
221  if (k == 8) break;
222  }
223  }
224  return r;
225 }

434 void classes_ns::unknown_option(KeyAndVal &cur, TokenList &res, TokenList &spec, int X) {
435  LatexPackage * C = the_class_data.cur_pack();
436  const std::string &name = cur.get_full_name();
437  if (!C->has_a_default) {
438  if (C->is_class()) {
439  } else
440  log_and_tty << "Unknown option `" << name << "' for package `" << C->real_name() << "'\n";
441  } else {
442  TokenList u = cur.to_list();
443  if (X == 1) {
444  TokenList w;
445  w = string_to_list(cur.get_name(), true);
446  spec.push_back(the_parser.hash_table.def_token);
447  spec.push_back(the_parser.hash_table.CurrentOptionKey_token);
448  spec.splice(spec.end(), w);
449 
450  w = cur.get_val();
451  if (w.empty())
452  w.push_back(the_parser.hash_table.relax_token);
453  else
454  w.pop_front();
455  the_parser.brace_me(w);
456  spec.push_back(the_parser.hash_table.def_token);
457  spec.push_back(the_parser.hash_table.CurrentOptionValue_token);
458  spec.splice(spec.end(), w);
459  the_parser.brace_me(u);
460  spec.push_back(the_parser.hash_table.def_token);
461  spec.push_back(the_parser.hash_table.CurrentOption_token);
462  spec.splice(spec.end(), u);
463  u = C->default_handler;
464  spec.splice(spec.end(), u);
465  } else if (X == 2) {
466  if (!res.empty()) res.push_back(Token(other_t_offset, ','));
467  res.splice(res.end(), u);
468  } else {
469  the_parser.brace_me(u);
470  res.push_back(the_parser.hash_table.def_token);
471  res.push_back(the_parser.hash_table.CurrentOption_token);
472  res.splice(res.end(), u);
473  u = C->default_handler;
474  res.splice(res.end(), u);
475  }
476  }
