172 // Defines something at position A, with type and subtype B and C.
173 // Definition is global if gbl is true. In the case of a user defined
174 // command, we must increase the reference count, and if we override
175 // a user defined command we must decrease its reference count (maybe kill).
176 // If the definition is local, the old definition is saved on the save stack.
177 void Parser::eq_define(size_t a, CmdChr bc, bool gbl) {
178  if (bc.is_user()) mac_table.incr_macro_ref(bc.chr);
179  if (!gbl && hash_table.eqtb[a].must_push(cur_level))
180  push_save_stack(new SaveAuxCmd(a, hash_table.eqtb[a]));
181  else if (hash_table.eqtb[a].is_user())
182  mac_table.delete_macro_ref(hash_table.eqtb[a].chr);
183  if (gbl)
184  hash_table.eqtb[a].primitive(bc);
185  else
186  hash_table.eqtb[a].set(bc, cur_level);
187 }

189 // This defines a user command, associated to the token A.
190 // Depending of redef, this may be valid or not
191 // (implements \provideXX, \newXX and \renewXX)
192 // We either kill the macro, or increment its reference count.
193 void Parser::mac_define(Token a, Macro *b, bool gbl, rd_flag redef, symcodes what) {
194  if (ok_to_define(a, redef)) {
195  CmdChr nv = CmdChr(what, mac_table.new_macro(b));
196  if (tracing_assigns()) {
197  the_log << lg_startbrace << gbl_or_assign(gbl, false) << a << "=";
198  token_for_show(hash_table.eqtb[a.eqtb_loc()]);
199  the_log << "}\n{into " << a << "=";
200  token_for_show(nv);
201  the_log << lg_endbrace;
202  }
203  eq_define(a.eqtb_loc(), nv, gbl);
204  } else
205  delete b;
206 }

208 // For \newcommand, \renewcommand: we signal an error if we cannot define
209 // the thing. For \providecommand: the function returns true if we have
210 // to define, false otherwise.
211 auto Parser::ok_to_define(Token a, rd_flag redef) -> bool {
212  if (redef == rd_never) return false;
213  if (a.not_a_cmd()) return false; // Should not happen
214  if (a == hash_table.frozen_protection) return false;
215  if (redef == rd_always) return true;
216  auto A = a.eqtb_loc();
217  bool undef_or_relax = hash_table.eqtb[A].is_undef_or_relax();
218  if (redef == rd_if_defined && undef_or_relax) {
219  bad_redefinition(1, a);
220  return false;
221  }
222  if (redef == rd_if_undef && !undef_or_relax) {
223  bad_redefinition(0, a);
224  return false;
225  }
226  return !(redef == rd_skip && !undef_or_relax);
227 }

230 void Parser::word_define(size_t a, long c, bool gbl) {
231  EqtbInt &W = eqtb_int_table[a];

233  if (tracing_assigns()) {

237  the_log << lg_endbrace;

241  else if (reassign)

245  W = {c, cur_level};

423 // Restore command. We have to take care to free memory for user commands.
424 void SaveAuxCmd::unsave(bool trace, Parser &P) {
425  int lvl = P.hash_table.eqtb[cs].level;
426  if (trace) {
427  String S = lvl == 1 ? "retaining " : (val.is_undef() ? "killing " : "restoring ");
428  the_log << lg_startstack << S << Token(cs + eqtb_offset);
429  if (lvl > 1 && !val.is_undef()) {
430  the_log << "=";
431  if (val.is_user())
432  P.token_for_show(val);
433  else
434  Parser::print_cmd_chr(val);

439  if (val.is_user()) P.mac_table.delete_macro_ref(val.chr);

441  if (P.hash_table.eqtb[cs].is_user()) // kill cur and change

443  P.hash_table.eqtb[cs].set(val, level);

517 // case where a table preamble says >{}c<{xx$yy} and we see &
518 // here xy can be } or \endgroup
519 auto Parser::stack_math_in_cell() -> bool {
520  auto n = the_save_stack.size();
521  bool first = true;
522  for (size_t i = n; i > 0; i--) {
523  SaveAux *p = the_save_stack[i - 1];
524  if (p->type != st_boundary) continue;
525  boundary_type cur = dynamic_cast<SaveAuxBoundary *>(p)->val;
526  if (cur == bt_brace || cur == bt_semisimple) continue;
527  if (first) {
528  if (cur != bt_math) return false;
529  first = false;
530  continue;
531  }
532  return cur == bt_cell;
533  }
534  return false;
535 }

553 // The function called when we see a closing brace or \endgroup.
554 // \end{foo} expands to \endfoo\endenv\endfoo, the last \endfoo is in cur_tok
555 // there should be an \endfoo token in cur_tok
556 void Parser::pop_level(boundary_type v) {
557  bool trace = tracing_stack();
558  bool must_throw = false;
559  boundary_type w = first_boundary();
560  if (w == bt_impossible) {
561  if (v == bt_brace)
562  parse_error(err_tok, "Extra }");
563  else if (v == bt_semisimple)
564  parse_error(err_tok, "Extra \\endgroup");
565  else
566  parse_error(err_tok, "Empty save stack");
567  return;
568  }
569  if (v != w) {
570  if (v == bt_brace && w == bt_tpa)
571  must_throw = true;
572  else if (w == bt_env) {
573  if (v == bt_semisimple) v = bt_esemisimple;
574  err_buf << bf_reset << "Extra " << parser_ns::to_string(v) << " found in unclosed environment " << cur_env_name;
575  signal_error(err_tok, "extra brace");
576  return;
577  } else {
578  if (w == bt_semisimple) w = bt_esemisimple;
579  if (v == bt_semisimple) v = bt_esemisimple;
580  wrong_pop(err_tok, parser_ns::to_string(v), parser_ns::to_string(w));
581  }
582  }
583  if (v == bt_env && cur_tok.is_valid()) {
584  std::string foo = cur_tok.tok_to_str();
585  if (foo != "\\end" + cur_env_name) {
586  err_buf << bf_reset << "Environment '" << cur_env_name << "' started at line " << first_boundary_loc << " ended by " << cur_tok;
587  signal_error(err_tok, "bad end env");
588  }
589  // this is the wrong env, we nevertheless pop
590  }
591  for (;;) {
592  if (the_save_stack.empty()) {
593  parse_error(err_tok, "Internal error: empty save stack");
594  return;
595  }
596  SaveAux *tmp = the_save_stack.back();
597  bool ok = tmp->type == st_boundary;
598  the_save_stack.pop_back();
599  tmp->unsave(trace, *this);
600  my_stats.one_more_down();
601  delete tmp;
602  if (ok) {
603  if (must_throw) {
604  cur_level++;
605  throw EndOfData();
606  }
607  return;
608  }
609  }
610 }
