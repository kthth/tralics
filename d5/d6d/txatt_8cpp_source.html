135 void Buffer::push_back(const AttPair &X) {
136  const Istring &b = X.name;
137  const Istring &a = X.value;
138  if (a.null()) return;
139  if (b.null()) return;
140  String B = b.p_str();
141  String A = a.p_str();
142  if (A == nullptr) A = "_bug_";
143  if (B == nullptr) B = "_bug_";
144  if (B[0] == '\'') return;
145  push_back(' ');
146  push_back(B);
147  push_back('=');
148  push_back('\'');
149  while (*A != 0) {

184 // Returns true if there is a space. Kills at the space. Advance
185 auto Buffer::look_at_space(const std::string &s) -> bool {
186  reset();
187  push_back(s);
188  bool has_space = false;
189  reset_ptr();
190  for (int i = 0;; i++) {
191  if (head() == 0) break;
192  if (is_space(head())) {
193  has_space = true;

195  advance();
196  break;
197  }
198  advance();
199  }
200  return has_space;
