241 auto Parser::get_ctb_opt() -> name_positions {
242  TokenList L;
243  read_optarg_nopar(L);
244  if (L.empty()) return cst_invalid;
245  Token t = token_ns::get_unique(L);
246  if (t.is_null()) return cst_invalid;
247  if (t.cmd_val() != letter_catcode) return cst_invalid;
248  char c = t.val_as_letter();
249  if (c == 'c') return np_letter_c;
250  if (c == 's') return np_letter_s;
251  if (c == 't') return np_letter_t;
252  if (c == 'b') return np_letter_b;
253  if (c == 'l') return np_letter_l;
254  if (c == 'r') return np_letter_r;
255  return cst_invalid;
256 }

258 // Nodes: tblr, or 2 letters
259 auto Parser::get_trees_opt() -> name_positions {
260  TokenList L;
261  read_optarg_nopar(L);
262  if (L.empty()) return cst_invalid;
263  Token t1, t2;
264  token_ns::get_unique(L, t1, t2);
265  if (t1.is_null()) return cst_invalid;
266  if (t1.cmd_val() != letter_catcode) return cst_invalid;
267  uint c = t1.val_as_letter();
268  if (t2.is_null()) {
269  if (c == 'l') return np_letter_l;
270  if (c == 'r') return np_letter_r;
271  if (c == 't') return np_letter_t;
272  if (c == 'b') return np_letter_b;
273  return cst_invalid;
274  }
275  if (t2.cmd_val() != letter_catcode) return cst_invalid;
276  uint c2 = t2.val_as_letter();
277  if (c != 't' && c != 'b') return cst_invalid;
278  if (c2 != 'l' && c2 != 'r') return cst_invalid;
279  if (c == 't') return c2 == 'l' ? np_letters_tl : np_letters_tr;
280  return c2 == 'l' ? np_letters_bl : np_letters_br;
281 }

546 // Translates \section, \chapter, etc.
547 void Parser::T_paras(subtypes x) {
548  if (x == toplevel_sec_code) {
549  TokenList L = read_arg();
550  if (!L.empty()) {
551  token_from_list(L.front());
552  sectionning_offset = section_code;
553  if (cur_cmd_chr.get_cmd() == section_cmd) sectionning_offset = cur_cmd_chr.get_chr();
554  if (sectionning_offset == chapter_code) Xid(1).add_attribute(np_chapters, np_true);
555  if (sectionning_offset == part_code) Xid(1).add_attribute(np_part, np_true);
556  }
557  return;
558  }
559  int y = x - sectionning_offset;
560  if (x == endsec_code) {
561  y = 0;
562  TokenList L = read_arg();
563  if (!L.empty()) {
564  token_from_list(L.front());
565  if (cur_cmd_chr.get_cmd() == section_cmd) y = cur_cmd_chr.get_chr() - sectionning_offset;
566  }
567  }
568  if (y < 0) y = 0;
569  if (y > 6) y = 6;
570  Istring Y = the_names[cst_div0 + y];
571  leave_h_mode();
572  the_stack.para_aux(y); // this pops the stack...
573  the_stack.add_nl();
574  if (x == endsec_code) return;
575  the_stack.push1(Y, name_positions(np_div0 + y));
576  bool star = remove_initial_star();
577  if (chapter_has_star && x == chapter_code) star = true;
578  if (star)
579  last_att_list().push_back(np_rend, np_nonumber);
580  else {
581  if (x == part_code)
582  refstepcounter("part", false);
583  else if (x == chapter_code)
584  refstepcounter("chapter", false);
585  else if (x == section_code)
586  refstepcounter("section", false);
587  else if (x == subsection_code)
588  refstepcounter("subsection", false);
589  else if (x == subsubsection_code)
590  refstepcounter("subsubsection", false);
591  else if (x == paragraph_code)
592  refstepcounter("paragraph", false);
593  else if (x == subparagraph_code)
594  refstepcounter("subparagraph", false);
595  else
596  star = true;
597  }
598  start_paras(y, Y.c_str(), star);
599 }
