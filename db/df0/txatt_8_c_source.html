136 void Buffer::push_back(const AttPair &X) {
137  const Istring &b = X.name;
138  const Istring &a = X.value;
139  if (a.null()) return;
140  if (b.null()) return;
141  String B = b.p_str();
142  String A = a.p_str();
143  if (A == nullptr) A = "_bug_";
144  if (B == nullptr) B = "_bug_";
145  if (B[0] == '\'') return;
146  push_back(' ');
147  push_back(B);
148  push_back('=');
149  push_back('\'');
150  while (*A != 0) {
151  if (*A == '\'')
152  push_back("&apos;");
153  else
154  push_back(*A);
155  A++;
156  }
157  push_back('\'');
158 }

185 // Returns true if there is a space. Kills at the space. Advance
186 auto Buffer::look_at_space(const std::string &s) -> bool {
187  reset();
188  push_back(s);
189  bool has_space = false;
190  reset_ptr();
191  for (int i = 0;; i++) {
192  if (head() == 0) break;
193  if (is_space(head())) {
194  has_space = true;
195  kill_at(ptr);
196  advance();
197  break;
198  }
199  advance();
200  }
201  return has_space;
202 }
