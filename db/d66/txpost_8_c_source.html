327 // This finds a sub element named match, and does some action X
328 // Recursion stops when found.
329 void Xml::recurse0(XmlAction &X) {
330  auto tl = size();
331  for (size_t k = 0; k < tl; k++) {
332  Xml *y = tree[k];
333  if ((y == nullptr) || y->is_xmlc()) continue;
334  if (y->has_name(X.get_match())) switch (X.get_what()) {
335  case rc_contains: X.mark_found(); return;
336  case rc_delete_first:
337  X.set_int_val(y->get_id().value);
338  tree[k] = nullptr;
339  return;
340  case rc_return_first:
341  X.set_xml_val(y);
342  X.mark_found();
343  return;
344  case rc_return_first_and_kill:
345  X.set_xml_val(y);
346  tree[k] = nullptr;
347  X.mark_found();
348  return;
349  default: log_and_tty << "illegal value in recurse0\n" << lg_fatal; abort();
350  }
351  y->recurse0(X);
352  if (X.is_ok()) return;
353  }
354 }

356 // This does some action for every element named X.
357 void Xml::recurse(XmlAction &X) {
358  auto len = size();
359  for (size_t k = 0; k < len; k++) {
360  Xml *T = tree[k];
361  if ((T == nullptr) || T->is_xmlc()) continue;
362  if (T->has_name(X.get_match())) {
363  switch (X.get_what()) {
364  case rc_delete: tree[k] = nullptr; continue;
365  case rc_delete_empty:
366  if (T->is_empty()) {
367  tree[k] = nullptr;
368  continue;
369  }
370  break;
371  case rc_empty_par:
372  if (T->is_empty_spec()) {
373  tree[k] = nullptr;
374  continue;
375  }
376  break;
377  case rc_how_many: X.incr_int_val(); break;
378  case rc_subst: tree[k] = X.get_xml_val(); continue;
379  case rc_move:
380  X.get_xml_val()->push_back(T);
381  tree[k] = nullptr;
382  continue;
383  case rc_composition: // a module in the comp section
384  {
385  Istring an = T->get_id().has_attribute(the_names[np_id]);
386  if (!an.null()) post_ns::remove_label("module in composition", an);
387  tree[k] = nullptr;
388  k--;
389  auto Len = T->size();
390  for (size_t j = 0; j < Len; j++) {
391  Xml *W = T->tree[j];
392  if (W == nullptr) continue;
393  if (!W->is_xmlc() && W->has_name(np_head)) {
394  post_ns::remove_me_from_heads(W);
395  continue;
396  }
397  insert_at(k + 1, W);
398  k++;
399  }
400  recurse(X); // this->tree has changed
401  return; // nothing more to do.
402  }
403  case rc_rename: T->change_name(X.get_string_val()); break;
404  default: log_and_tty << "illegal value in recurse\n" << lg_fatal; abort();
405  }
406  }
407  T->recurse(X);
408  }
409 }
