89 void Parser::T_fancy() {
90  subtypes c = cur_cmd_chr.chr;
91  int hf = 0;
92  if (c == fancy_head_code)
93  hf = 1;
94  else if (c == fancy_foot_code)
95  hf = 2;
96  else if (c == fancy_hf_code)

98  if (hf == 0) {
99  TokenList L1;

101  TokenList L2 = read_arg();
102  if (!o) L1 = L2;
103  if (c == fancy_lhead_code) {
104  T_fancy("elh", L1);
105  T_fancy("olh", L2);
106  return;
107  }
108  if (c == fancy_chead_code) {
109  T_fancy("ech", L1);
110  T_fancy("och", L2);
111  return;
112  }
113  if (c == fancy_rhead_code) {
114  T_fancy("erh", L1);
115  T_fancy("orh", L2);
116  return;
117  }
118  if (c == fancy_lfoot_code) {
119  T_fancy("elf", L1);
120  T_fancy("olf", L2);
121  return;
122  }
123  if (c == fancy_cfoot_code) {
124  T_fancy("ecf", L1);
125  T_fancy("ocf", L2);
126  return;
127  }
128  if (c == fancy_rfoot_code) {
129  T_fancy("erf", L1);
130  T_fancy("orf", L2);
131  return;
132  }
133  }
134  TokenList L1;
135  read_optarg_nopar(L1);
136  TokenList L = read_arg();
137  for (;;) {
138  bool use_e = false, use_o = false;
139  bool use_h = false, use_f = false;
140  bool use_l = false, use_r = false, use_c = false;
141  for (;;) {
142  if (L1.empty()) break;
143  Token t = L1.front();
144  L1.pop_front();
145  if (t.is_comma_token()) break;
146  if (t.no_case_letter('e')) {
147  use_e = true;
148  } else if (t.no_case_letter('o')) {
149  use_o = true;
150  } else if (t.no_case_letter('l')) {
151  use_l = true;
152  } else if (t.no_case_letter('c')) {
153  use_c = true;
154  } else if (t.no_case_letter('r')) {
155  use_r = true;
156  } else if (t.no_case_letter('h')) {
157  use_h = true;
158  } else if (t.no_case_letter('f')) {
159  use_f = true;
160  } else
161  continue;
162  }
163  if (!use_e && !use_o) {
164  use_e = true;
165  use_o = true;
166  }
167  if (hf == 1) use_h = true;
168  if (hf == 2) use_f = true;
169  if (!use_h && !use_f) {
170  use_h = true;
171  use_f = true;
172  }
173  if (!use_l && !use_c && !use_r) {
174  use_l = true;
175  use_c = true;
176  use_r = true;
177  }
178  if (use_h) {
179  if (use_e) {
180  if (use_l) T_fancy("elh", L);
181  if (use_c) T_fancy("ech", L);
182  if (use_r) T_fancy("erh", L);
183  }
184  if (use_o) {
185  if (use_l) T_fancy("olh", L);
186  if (use_c) T_fancy("och", L);
187  if (use_r) T_fancy("orh", L);
188  }
189  }
190  if (use_f) {
191  if (use_e) {
192  if (use_l) T_fancy("elf", L);
193  if (use_c) T_fancy("ecf", L);
194  if (use_r) T_fancy("erf", L);

197  if (use_l) T_fancy("olf", L);
198  if (use_c) T_fancy("ocf", L);
199  if (use_r) T_fancy("orf", L);
200  }
201  }
202  if (L1.empty()) break;
203  }

448 // Implementation of some loops
449 // should be expand rather than translate
450 void Parser::T_xkv_for(subtypes c) {
451  Token comma = hash_table.comma_token;
452  Token nil = hash_table.nil_token;
453  Token doubleat = hash_table.doubleat_token;
454  Token T = cur_tok;
455  TokenList res;
456  switch (c) {
457  case for_code: { // case \@for \A:= list \do code
458  Token cmd = read_for_variable();
459  TokenList L = read_until(hash_table.do_token);
460  TokenList function = read_arg();
461  expand_first(L);
462  if (L.empty()) break;
463  brace_me(function);
464  res.push_back(hash_table.forloop_token);
465  res.splice(res.end(), L);
466  res.push_back(comma);
467  res.push_back(nil);
468  res.push_back(comma);
469  res.push_back(nil);
470  res.push_back(doubleat);
471  res.push_back(cmd);
472  res.splice(res.end(), function);
473  break;
474  }
475  case forloop_code: { // \@forloop
476  TokenList entry = read_until(comma);
477  TokenList next_entry = read_until(comma);
478  TokenList remainder = read_until(doubleat);
479  Token cmd = get_r_token();
480  TokenList function = read_arg();
481  if (token_ns::has_a_single_token(entry, nil)) break;
482  new_macro(entry, cmd);
483  TokenList aux = function;
484  res.splice(res.end(), aux);
485  if (token_ns::has_a_single_token(next_entry, nil)) break;
486  brace_me(next_entry);
487  TokenList aux2 = function;
488  brace_me(function);
489  res.push_back(hash_table.def_token);
490  res.push_back(cmd);
491  res.splice(res.end(), next_entry);
492  res.splice(res.end(), aux2);
493  res.push_back(hash_table.iforloop_token);
494  res.splice(res.end(), remainder);
495  res.push_back(doubleat);
496  res.push_back(cmd);
497  res.splice(res.end(), function);
498  break;
499  }
500  case tfor_code: { // \@tfor
501  Token cmd = read_for_variable();
502  back_input(hash_table.space_token);
503  TokenList L = read_until(hash_table.do_token);
504  TokenList function = read_arg();
505  if (token_ns::has_a_single_token(L, hash_table.space_token)) break;
506  brace_me(function);
507  res.push_back(hash_table.tforloop_token);
508  res.splice(res.end(), L);
509  res.push_back(nil);
510  res.push_back(nil);
511  res.push_back(doubleat);
512  res.push_back(cmd);
513  res.splice(res.end(), function);
514  break;
515  }
516  case xkv_breaktfor_code: { // \tralics@for@break
517  // should we insert \fi here
518  TokenList W;
519  W.push_back(nil);
520  W.push_back(comma);
521  read_delimited(W);
522  res.push_back(hash_table.fi_token);
523  break;
524  }
525  case breaktfor_code: // \@break@tfor
526  read_until(hash_table.doubleat_token);
527  ignore_arg(); // should this be long ?
528  ignore_arg();
529  res.push_back(hash_table.fi_token);
530  break;
531  case iforloop_code: // \@iforloop \@tforloop
532  case tforloop_code: {
533  TokenList entry = c == iforloop_code ? read_until(comma) : read_arg();
534  TokenList remainder = read_until(doubleat);
535  Token cmd = get_r_token();
536  TokenList function = read_arg();
537  if (token_ns::has_a_single_token(entry, nil)) break;
538  new_macro(entry, cmd);
539  TokenList aux = function;
540  res.splice(res.end(), aux);
541  brace_me(function);
542  res.push_back(hash_table.relax_token);
543  res.push_back(c == iforloop_code ? hash_table.iforloop_token : hash_table.tforloop_token);
544  res.splice(res.end(), remainder);
545  res.push_back(doubleat);
546  res.push_back(cmd);
547  res.splice(res.end(), function);
548  break;
549  }
550  case xkv_for_n_code: // \tralics@for@n and variants
551  case xkv_for_o_code:
552  case xkv_for_en_code:
553  case xkv_for_eo_code: {
554  TokenList L = read_arg(); // the list to operate on
555  if (c == xkv_for_o_code || c == xkv_for_eo_code) expand_first(L);
556  TokenList cmd = read_arg();
557  TokenList function = read_arg();
558  if (L.empty() && (c == xkv_for_n_code || c == xkv_for_o_code)) break;
559  brace_me(function);
560  res.push_front(hash_table.xkv_for_i_token);
561  res.splice(res.end(), cmd);
562  res.splice(res.end(), function);
563  res.splice(res.end(), L);
564  res.push_back(comma);
565  res.push_back(nil);
566  res.push_back(comma);
567  break;
568  }
569  case xkv_for_i_code: { // \tralics@for@i
570  Token cmd = get_r_token(); // the command
571  TokenList function = read_arg(); // the code
572  TokenList entry = read_until(comma);
573  if (token_ns::has_a_single_token(entry, nil)) break;
574  new_macro(entry, cmd);
575  TokenList aux = function;
576  brace_me(function);
577  res.splice(res.end(), aux);
578  res.push_back(hash_table.xkv_for_i_token);
579  res.push_back(cmd);
580  res.splice(res.end(), function);
581  break;
582  }
583  // case xkv_whilist_code:
584  //
585  default:;
586  }
587  if (tracing_commands()) the_log << lg_start << T << "<- " << res << lg_end;
588  back_input(res);
589 }
