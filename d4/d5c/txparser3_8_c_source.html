172 // Defines something at position A, with type and subtype B and C.
173 // Definition is global if gbl is true. In the case of a user defined
174 // command, we must increase the reference count, and if we override
175 // a user defined command we must decrease its reference count (maybe kill).
176 // If the definition is local, the old definition is saved on the save stack.
177 void Parser::eq_define(int a, CmdChr bc, bool gbl) {
178  if (bc.is_user()) mac_table.incr_macro_ref(bc.get_chr());
179  if (!gbl && hash_table.eqtb[a].must_push(cur_level))
180  push_save_stack(new SaveAuxCmd(a, hash_table.eqtb[a]));
181  else if (hash_table.eqtb[a].is_user())
182  mac_table.delete_macro_ref(hash_table.eqtb[a].get_chr());
183  if (gbl)
184  hash_table.eqtb[a].primitive(bc);
185  else
186  hash_table.eqtb[a].set(bc, cur_level);
187 }

189 // This defines a user command, associated to the token A.
190 // Depending of redef, this may be valid or not
191 // (implements \provideXX, \newXX and \renewXX)
192 // We either kill the macro, or increment its reference count.
193 void Parser::mac_define(Token a, Macro *b, bool gbl, rd_flag redef, symcodes what) {
194  if (ok_to_define(a, redef)) {
195  CmdChr nv = CmdChr(what, mac_table.new_macro(b));
196  if (tracing_assigns()) {
197  the_log << lg_startbrace << gbl_or_assign(gbl, false) << a << "=";
198  token_for_show(hash_table.eqtb[a.eqtb_loc()].get_cmdchr());
199  the_log << "}\n{into " << a << "=";
200  token_for_show(nv);
201  the_log << lg_endbrace;
202  }
203  eq_define(a.eqtb_loc(), nv, gbl);
204  } else
205  delete b;
206 }

208 // For \newcommand, \renewcommand: we signal an error if we cannot define
209 // the thing. For \providecommand: the function returns true if we have
210 // to define, false otherwise.
211 auto Parser::ok_to_define(Token a, rd_flag redef) -> bool {
212  if (redef == rd_never) return false;
213  if (a.not_a_cmd()) return false; // Should not happen
214  if (a == hash_table.frozen_protection) return false;
215  if (redef == rd_always) return true;
216  int A = a.eqtb_loc();
217  bool undef_or_relax = hash_table.eqtb[A].is_undef_or_relax();
218  if (redef == rd_if_defined && undef_or_relax) {
219  bad_redefinition(1, a);
220  return false;
221  }
222  if (redef == rd_if_undef && !undef_or_relax) {
223  bad_redefinition(0, a);
224  return false;
225  }
226  if (redef == rd_skip && !undef_or_relax) return false;
227  return true;
228 }

231 void Parser::word_define(int a, int c, bool gbl) {
232  EqtbInt &W = eqtb_int_table[a];
233  bool reassign = !gbl && W.get_val() == c;
234  if (tracing_assigns()) {
235  CmdChr tmp(assign_int_cmd, subtypes(a));
236  the_log << lg_startbrace << gbl_or_assign(gbl, reassign) << "\\" << tmp.name() << "=" << W.get_val();
237  if (!reassign) the_log << " into \\" << tmp.name() << "=" << c;
238  the_log << lg_endbrace;
239  }
240  if (gbl)
241  W.val_and_level(c, level_one);
242  else if (reassign)
243  return;
244  else {
245  if (W.must_push(cur_level)) push_save_stack(new SaveAuxInt(W.get_level(), a, W.get_val()));
246  W.val_and_level(c, cur_level);
247  }

424 // Restore command. We have to take care to free memory for user commands.
425 void SaveAuxCmd::unsave(bool trace, Parser &P) {
426  int lvl = P.hash_table.eqtb[cs].get_level();
427  if (trace) {
428  String S = lvl == level_one ? "retaining " : (val.is_undef() ? "killing " : "restoring ");
429  the_log << lg_startstack << S << Token(cs + eqtb_offset);
430  if (lvl > level_one && !val.is_undef()) {
431  the_log << "=";
432  if (val.is_user())
433  P.token_for_show(val);
434  else
435  P.print_cmd_chr(val);
436  }
437  the_log << lg_endsentence;
438  }
439  if (lvl == level_one) { // retain old value, so kill val
440  if (val.is_user()) P.mac_table.delete_macro_ref(val.get_chr());
441  } else {
442  if (P.hash_table.eqtb[cs].is_user()) // kill cur and change
443  P.mac_table.delete_macro_ref(P.hash_table.eqtb[cs].get_chr());
444  P.hash_table.eqtb[cs].set(val, level);
445  }
446 }

508  int n = the_save_stack.size();

518 // case where a table preamble says >{}c<{xx$yy} and we see &
519 // here xy can be } or \endgroup
520 auto Parser::stack_math_in_cell() -> bool {
521  int n = the_save_stack.size();
522  bool first = true;
523  for (int i = n - 1; i >= 0; i--) {
524  SaveAux *p = the_save_stack[i];
525  if (p->type != st_boundary) continue;
526  boundary_type cur = static_cast<SaveAuxBoundary *>(p)->get_val();
527  if (cur == bt_brace || cur == bt_semisimple) continue;
528  if (first) {
529  if (cur != bt_math) return false;
530  first = false;
531  continue;
532  }

555 // The function called when we see a closing brace or \endgroup.
556 // \end{foo} expands to \endfoo\endenv\endfoo, the last \endfoo is in cur_tok
557 // there should be an \endfoo token in cur_tok
558 void Parser::pop_level(boundary_type v) {
559  bool trace = tracing_stack();
560  bool must_throw = false;
561  boundary_type w = first_boundary();
562  if (w == bt_impossible) {
563  if (v == bt_brace)
564  parse_error(err_tok, "Extra }");
565  else if (v == bt_semisimple)
566  parse_error(err_tok, "Extra \\endgroup");
567  else
568  parse_error(err_tok, "Empty save stack");
569  return;
570  }
571  if (v != w) {
572  if (v == bt_brace && w == bt_tpa)
573  must_throw = true;
574  else if (w == bt_env) {
575  if (v == bt_semisimple) v = bt_esemisimple;
576  err_ns::local_buf << bf_reset << "Extra " << parser_ns::to_string(v) << " found in unclosed environment " << cur_env_name;
577  signal_error(err_tok, "extra brace");
578  return;
579  } else {
580  if (w == bt_semisimple) w = bt_esemisimple;
581  if (v == bt_semisimple) v = bt_esemisimple;
582  wrong_pop(err_tok, parser_ns::to_string(v), parser_ns::to_string(w));
583  }
584  }
585  if (v == bt_env && cur_tok.is_valid()) {
586  std::string foo = cur_tok.tok_to_str();
587  if (foo != "\\end" + cur_env_name) {
588  err_ns::local_buf << bf_reset << "Environment '" << cur_env_name << "' started at line " << first_boundary_loc << " ended by "
589  << cur_tok;
590  signal_error(err_tok, "bad end env");
591  }
592  // this is the wrong env, we nevertheless pop
593  }
594  for (;;) {
595  if (the_save_stack.empty()) {
596  parse_error(err_tok, "Internal error: empty save stack");
597  return;
598  }
599  SaveAux *tmp = the_save_stack.back();
600  bool ok = tmp->type == st_boundary;
601  the_save_stack.pop_back();
602  tmp->unsave(trace, *this);
603  my_stats.one_more_down();
604  delete tmp;
605  if (ok) {
606  if (must_throw) {
607  cur_level++;
608  throw EndOfData();
609  }
610  return;
611  }
612  }
613 }
