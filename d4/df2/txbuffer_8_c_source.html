154 // Inserts an integer (this is the print_int procedure from TeX)
155 // at least one digit is printed.
156 void Buffer::push_back_int(long n) {
157  static std::array<long, 30> dig;
158  size_t k = 0;
159  if (n < 0) {
160  push_back('-');
161  if (n > -100000000)
162  n = -n;
163  else {
164  auto m = -1 - n;
165  n = m / 10;
166  m = (m % 10) + 1;
167  k = 1;
168  if (m < 10)
169  dig[0] = m;
170  else {
171  dig[0] = 0;
172  n++;
173  }
174  }
175  }
176  for (;;) { // construct the list of digits
177  dig[k] = n % 10;
178  n = n / 10;
179  k++;
180  if (n == 0) break;
181  }
182  while (k > 0) { // print the list
183  k--;
184  push_back('0' + static_cast<char>(dig[k]));
185  }
186 }
