108  case 'O': translate_char(CmdChr(uchar('\326'))); break;
109  case 'U': translate_char(CmdChr(uchar('\334'))); break;
110  case 'I': translate_char(CmdChr(uchar('\317'))); break;
111  case 'E': translate_char(CmdChr(uchar('\313'))); break;

168 // Translate current character.
169 void Parser::translate_char(CmdChr X) {
170  if (!the_stack.in_h_mode()) LC();
171  codepoint c = X.char_val();
172  if (c.is_null()) return; // we do not want null chars in trace or Xml
173  if (tracing_commands()) the_log.out_single_char(c);
174  if (!c.is_small()) {
175  process_char(c);
176  return;
177  }
178  switch (c.value) {
179  case '-': minus_sign(X); return;
180  case '\'':
181  case '`':
182  case '<':
183  case '>': english_quotes(X); return;
184  case '"':
185  if (X.is_letter() || InUrlHandler::global_in_url || InLoadHandler::global_in_load)
186  process_char(c);
187  else
188  umlaut();
189  return;
190  case ';':
191  case ':':
192  case '!':
193  case '?':
194  case 0xBB: // <<
195  case 0xAB: // >>
196  french_punctuation(X);
197  return;
198  default: process_char(c);
199  }
200 }

202 // This translates `'<>
203 // In some case ``, '', << and >> are translated as 0xAB and 0xBB
204 void Parser::english_quotes(CmdChr X) {
205  uchar c = X.char_val().value; // Should be a small int
206  if (InUrlHandler::global_in_url || InLoadHandler::global_in_load) {
207  if (c == '<')
208  process_string("&lt;");
209  else if (c == '>')
210  process_string("&gt;");
211  else
212  unprocessed_xml.push_back(c);
213  } else {
214  if (X.is_other() && (c == '\'' || c == '`' || cur_lang_fr())) {
215  get_x_token();
216  if (cur_cmd_chr.is_other() && cur_cmd_chr.char_val() == uint(c)) {
217  if (c == '\'')
218  process_char(0x201d);
219  else if (c == '`')
220  process_char(0x201c);
221  else if (c == '<')
222  french_punctuation(CmdChr(uchar(0xAB)));
223  else
224  french_punctuation(CmdChr(uchar(0xBB)));
225  return;
226  }
227  if (cur_tok.is_valid()) back_input();
228  }
229  if (c == '<')
230  process_string("&lt;");
231  else if (c == '>')
232  process_string("&gt;");
233  else if (c == '\'' && X.is_other())
234  process_char(rightquote_val);
235  else if (c == '`' && X.is_other())
236  process_char(leftquote_val);
237  else
238  unprocessed_xml.push_back(c);
239  }
240  if (X.is_letter() && !the_main->get_zws_mode()) process_string(the_main->get_zws_elt() ? "&#x200B;" : "<zws/>");
241 }

262  process_char('-');

267 // This handles :;!? 0xAB 0xBB. Especially in French.
268 void Parser::french_punctuation(CmdChr X) {
269  uchar c = X.char_val().value;
270  if (InUrlHandler::global_in_url || InLoadHandler::global_in_load || X.is_letter() || !cur_lang_fr()) {
271  extended_chars(c);
272  return;
273  }
274  if (c == 0xAB) {
275  for (;;) {
276  get_x_token();
277  if (cur_cmd_chr.is_space()) continue;
278  if (cur_cmd_chr.get_cmd() == cst_cmd && cur_cmd_chr.get_chr() == nobreakspace_code) break;
279  if (cur_cmd_chr.get_cmd() == cst1_cmd && cur_cmd_chr.get_chr() == comma_code) break;
280  if (cur_tok.is_valid()) back_input();
281  break;
282  }
283  extended_chars(c);
284  process_char(0xA0);
285  return;
286  }
287  unprocessed_xml.remove_last_space();
288  process_char(0xA0);
289  extended_chars(c);
290 }
