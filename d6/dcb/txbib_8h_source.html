18 struct CitationKey {
19  std::string cite_key; // cite key, say Foo
20  std::string lower_cite_key; // cite key, lower case, say foo
21  bib_from cite_prefix; // symbolic prefix, say from_foot

23  void make_key(String s); // creates the key.
24 
25  CitationKey() = default;
26  CitationKey(String a, String b);
27  CitationKey(bib_from a, String b);
28 
29  [[nodiscard]] auto is_similar(const CitationKey &w) const -> bool { return cite_key == w.cite_key; }
30  [[nodiscard]] auto is_similar_lower(const CitationKey &w) const -> bool { return lower_cite_key == w.lower_cite_key; }
31  [[nodiscard]] auto is_same(const CitationKey &w) const -> bool { return is_similar(w) && cite_prefix == w.cite_prefix; }
32  [[nodiscard]] auto is_same_old(const CitationKey &w) const -> bool { return full_key == w.full_key; }
33  [[nodiscard]] auto is_same_lower(const CitationKey &w) const -> bool { return is_similar_lower(w) && cite_prefix == w.cite_prefix; }

35  [[nodiscard]] auto from_to_string() const -> String;

38  void move_to_refer() { cite_prefix = from_refer; }
