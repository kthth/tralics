17 class CitationKey {
18  std::string cite_key; // cite key, say Foo
19  std::string lower_cite_key; // cite key, lower case, say foo
20  bib_from cite_prefix; // symbolic prefix, say from_foot
21  std::string full_key; // something like footcite:Foo
22  void make_key(String); // creates the key.
23 public:
24  [[nodiscard]] auto is_similar(const CitationKey &w) const -> bool { return cite_key == w.cite_key; }
25  [[nodiscard]] auto is_similar_lower(const CitationKey &w) const -> bool { return lower_cite_key == w.lower_cite_key; }
26  [[nodiscard]] auto is_same(const CitationKey &w) const -> bool { return is_similar(w) && cite_prefix == w.cite_prefix; }
27  [[nodiscard]] auto is_same_old(const CitationKey &w) const -> bool { return full_key == w.full_key; }
28  [[nodiscard]] auto is_same_lower(const CitationKey &w) const -> bool { return is_similar_lower(w) && cite_prefix == w.cite_prefix; }
29  [[nodiscard]] auto is_same_lower_old(const CitationKey &w) const -> bool;
30  CitationKey() = default;
31  CitationKey(String, String);
32  CitationKey(bib_from, String);
33  [[nodiscard]] auto get_name() const -> std::string { return full_key; }
34  [[nodiscard]] auto get_from() const -> bib_from { return cite_prefix; }
35  [[nodiscard]] auto from_to_string() const -> String;
36  void move_to_year() { cite_prefix = from_year; } // leaves full_key unchanged
