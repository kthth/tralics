138 // This implements a command like \small.
139 void FontInfo::change_size(long c) {
140  if (c < 0) c = 0;
141  if (c > 12) c = 12;
142  size = c << 11;
143  if (!the_main->use_all_sizes()) {
144  if (c < 6)
145  tsize = fi_small_size;
146  else if (c > 6)
147  tsize = fi_big_size;
148  else
149  tsize = 0;
150  return;
151  }
152  switch (c) {
153  case 0: tsize = fi_small_size6; return;
154  case 1: tsize = fi_small_size5; return;
155  case 2: tsize = fi_small_size4; return;
156  case 3: tsize = fi_small_size3; return;
157  case 4: tsize = fi_small_size2; return;
158  case 5: tsize = fi_small_size1; return;
159  case 7: tsize = fi_big_size1; return;
160  case 8: tsize = fi_big_size2; return;
161  case 9: tsize = fi_big_size3; return;
162  case 10: tsize = fi_big_size4; return;
163  case 11: tsize = fi_big_size5; return;
164  case 12: tsize = fi_big_size6; return;
165  default: tsize = 0;
166  }
167 }

191 // This implements \fontfamily etc
192 void FontInfo::ltfont(const std::string &s, subtypes c) {
193  switch (c) {
194  case fontencoding_code: return; // Output encoding is always Unicode
195  case fontfamily_code: // rm, sf, or tt
196  if (s == "cmr" || s == "ptm")
197  family = 0;
198  else if (s == "cmss" || s == "phv")
199  family = fi_sf_family;
200  else if (s == "cmtt" || s == "pcr")
201  family = fi_tt_family;
202  else {
203  family = 0;
204  log_and_tty << lg_start << "Unknown font family " << s << lg_end;
205  }
206  return;
207  case fontseries_code: // md bf
208  if (s == "m")
209  series = 0;
210  else if (s == "b")
211  series = fi_bf_series;
212  else if (s == "bx")
213  series = fi_bx_series;
214  else if (s == "sb")
215  series = fi_sb_series;
216  else if (s == "c")
217  series = fi_c_series;
218  else {
219  series = 0;
220  log_and_tty << lg_start << "Unknown font series " << s << lg_end;
221  }
222  return;
223  case fontshape_code: // it sl sc
224  if (s == "n")
225  shape = 0;
226  else if (s == "it")
227  shape = fi_it_shape;
228  else if (s == "sl")
229  shape = fi_sl_shape;
230  else if (s == "sc")
231  shape = fi_sc_shape;
232  else {
233  shape = 0;
234  log_and_tty << lg_start << "Unknown font shape " << s << lg_end;
235  }
236  return;
237  default: return; // impossible case
238  }
239 }

246  for (unsigned i = 0; i < size(); i++)

264  if (size() >= 256) {
