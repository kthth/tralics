52 auto check_for_control(unsigned char c) -> unsigned char {
53  if (c == orig_termio.c_cc[VERASE])
54  c = 8; //^H
55  else if (c == orig_termio.c_cc[VEOF])
56  c = 4; // ^D
57  else if (c == orig_termio.c_cc[VKILL])
58  c = 11; // ^K
59  else if (c == orig_termio.c_cc[VWERASE])
60  c = 23; //^W
61  else if (c == orig_termio.c_cc[VSUSP])
62  c = 26; // ^Z
63  else if (c == orig_termio.c_cc[VINTR])
64  c = 3; // ^C
65  else if (c == orig_termio.c_cc[VQUIT])
66  c = 28;
67  return c;
68 }
69 inline auto is_interrupt(char cc) -> bool {
70  return cc == orig_termio.c_cc[VINTR] || cc == 3 || cc == orig_termio.c_cc[VQUIT] || cc == 'y' || cc == 'Y';
71 }

190 // Copy len characters from source to target. Strings may overlap.
191 void readline_ns::shift_string(char *S, int len, int source, int target) {
192  if (len == 1)
193  S[target] = S[source];
194  else if (len == 2) {
195  char x = S[source + 1];
196  S[target] = S[source];
197  S[target + 1] = x;
198  } else if (target <= source) {
199  for (int i = 0; i < len; i++, S++) S[target] = S[source];
200  } else {
201  S += len - 1;
202  for (int i = 0; i < len; i++, S--) S[target] = S[source];
203  }
204 }

209 // Returns the cursor position y of position x. Returns -1-y if the cursor
210 // is invisible. It is assumed that position n corresponds to m_pos.
211 auto Slined::newpos(int x, int n) -> int {
212  int i = 0, y = 0;
213  if (x >= n) { // We can hack a bit.
214  i = n;
215  y = m_pos;
216  }
217  while (i < x) {
218  if (not_printable(m_inbuf[i]))
219  y += 2;
220  else
221  y++;
222  i++;
223  }
224  if (y >= m_left && y < m_right) return y;
225  return -1 - y;
226 }

231 // Copy a string of size s into m_buffer or g_buffer, depending on sw.
232 // Control-A is copied as a two character sequence ^A. Returns
233 // the number of chars put in the buffer. If character at position
234 // inpos is copied, its location will be in m_pos.
235 
236 auto Slined::copystring(String string, int s, int inpos, bool sw) -> int {
237  int j = 0;
238  char *buf = sw ? m_buffer.data() : g_buffer.data();
239  for (int i = 0; i < s; i++) {
240  if (i == inpos) m_pos = j;
241  auto cn = string[i];
242  if (not_printable(cn)) {
243  buf[j] = '^';
244  j++;
245  cn = cn | 64;
246  }
247  buf[j] = cn;
248  j++;
249  }
250  return j;
251 }

253 // Full redisplay.
254 void Slined::redisplay() {
255  int j = copystring(m_inbuf, m_inmax, m_inpos, true);
256  if (m_inpos >= m_inmax) m_pos = j; // set cursor to EOL
257  m_max = j;

265 // Redisplay after prompt.
266 void Slined::redisplay0() {
267  int left, right; // position of left and right chars
268  int s = m_size, p = m_pos, m = m_max;
269  if (m <= s) { // we have a short line, ok
270  left = 0;
271  right = m;
272  } else if (p < s - 2) { // cursor visible if we show first char
273  left = 0;
274  right = s - 2;
275  } else if (m - p < s - 2) { // cursor visible if we show last char
276  right = m;
277  left = m - (s - 2);
278  } else { // put the cursor in the middle
279  right = p + (s - 4) / 2;
280  left = right - (s - 4);
281  }
282  m_left = left;
283  if (left != 0) std::cerr << "> "; // show that there are unseen characters there
284  finish_redisplay(left, right);
285 }

334 // insert char c1 (or c1+c2) n times at current position
335 void Slined::insert_in_image(int n, char c1, char c2) {
336  int m = m_max, p = m_pos;
337  int size = m - p; // number of chars after cur pos
338  int k = n;
339  if (c1 != 0) k += n; // k is the number of chars to insert
340  char *buf = m_buffer.data();
341  buf = buf + p;
342  if (size != 0) shift_string(buf, size, 0, k); // make room
343  for (int i = 0; i < n; i++) { // copy
344  if (c1 != 0) *(buf++) = c1;
345  *(buf++) = c2;
346  }
347  m += k;
348  m_max = m; // new size
349  p += k;
350  m_pos = p; // new position, cursor is after inserted chars
351  int d = m_right + k;
352  m_right = d; // new position of last char
353  int s = m_size;
354  if (m_left != 0) s -= 2;
355  if (m != d) s -= 2; // s is number of free positions on line
356  if (d - m_left > s) // cursor invisible, full redisplay needed
357  redisplay();
358  else {
359  for (int i = 0; i < n; i++) {
360  if (c1 != 0) std::cerr.put(c1);
361  std::cerr.put(c2);
362  }
363  if (p == m) {
364  } // we are at the right end
365  else if (p == d) // we are at the boundary
366  right_fill();
367  else
368  redisplay_eol(); // insert, and redisplay after position
369  }
370 }

400 // The real function that inserts a character c n times
401 void Slined::insert1(int n, unsigned char c) {
402  int pos = m_inpos;
403  int size = m_inmax;
404  int aux = size - pos;
405  char *buffer = m_inbuf;
406  buffer = buffer + pos;
407  if (aux != 0) // make room
408  shift_string(buffer, size, 0, n);
409  for (int i = 0; i < n; i++) buffer[i] = c; // insert
410  size += n;
411  pos += n;
412  m_inpos = pos;
413  m_inmax = size;
414  unsigned char c1 = 0, c2 = c;
415  if (not_printable(c)) {
416  c1 = '^';
417  c2 |= 64;
418  }
419  insert_in_image(n, c1, c2);
420  cur_line_modified = true;
421 }

432 // Move the cursor from Y to X
433 void Slined::rl_move(int y, int x) {
434  if (x == y) return;
435  m_inpos = x;
436  if (x == 0) { // move to left
437  if (m_left == 0) { // first char is visible
438  I_move(m_pos, 0);
439  m_pos = 0;
440  } else
441  redisplay1(0); // else general case
442  } else if (x == m_inmax) { // move to last char
443  if (m_right == m_max) { // last char is visible
444  I_move(m_pos, m_right);
445  m_pos = m_right;
446  } else
447  redisplay1(m_max); // else general case
448  } else {
449  y = newpos(x, y);
450  if (y >= 0) { // target is visible
451  I_move(m_pos, y);
452  m_pos = y;
453  } else
454  redisplay1(-1 - y); // target not visible
455  }
456 }

473 // Toggle two chars: the character on the left and the character under
474 // the cursor and advance by one char.
475 void Slined::toggle_char() {
476  int max = m_inmax;
477  int pos = m_inpos;
478  if (max < 2 || pos < 1) return;
479  if (pos == max) { // go back by 1 character
480  rl_move(pos, pos - 1);
481  pos--;
482  }
483  unsigned char c1 = m_inbuf[pos - 1], c2 = m_inbuf[pos];
484  if (c1 == c2) // we can hack
485  rl_move(pos, pos + 1);
486  else {
487  rl_delete(true); // must delete 2 chars and insert 2 chars
488  rl_move(pos, pos - 1);
489  rl_delete(true);
490  insert(1, c2);
491  insert(1, c1);
492  }
493 }

495 // this deletes a string between deb and fn.
496 void Slined::delete_string(int sw, int deb, int fn) {
497  int debut, fin;
498  if (deb < fn) {
499  debut = deb;
500  fin = fn;
501  } else {
502  debut = fn;
503  fin = deb;
504  }
505  if (debut < 0) debut = 0;
506  if (fin >= m_inmax) fin = m_inmax - 1;
507 
508  int taille = fin - debut + 1; // Size of what must be killed.
509  m_killbuf = std::string(m_inbuf + debut, taille);
510  if (sw != 0) {
511  rl_move(m_inpos, debut);
512  for (int i = 0; i < taille; i++)
513  if (m_inpos < m_inmax) rl_delete(false);
514  redisplay();
515  }

530 // insert in the buffer then redisplay all
531 void Slined::fast_ins(int n, String s, int l) {
532  int pos = m_inpos;
533  int size = m_inmax;
534  char *buffer = m_inbuf;
535  int aux = size - pos;
536  if (aux != 0) shift_string(buffer, aux, pos, pos + n * l); // make room
537  for (int i = 0; i < n; i++) {
538  strncpy(buffer + pos, s, l);
539  pos += l;
540  size += l;
541  }
542  m_inpos = pos;
543  m_inmax = size;
544  int j = copystring(s, l, -1, false);
545  pos = m_pos;
546  buffer = m_buffer.data();
547  size = m_max;
548  aux = size - pos;
549  if (aux != 0) shift_string(buffer, aux, pos, pos + j * n); // make room here also
550  for (int i = 0; i < n; i++) {
551  strncpy(buffer + pos, g_buffer.data(), j);
552  pos += j;
553  size += j;
554  }
555  m_max = size;
556  redisplay_eol();
557  rl_move(m_pos, pos); // put cursor where it should be
558 }

560 // show history
561 void Slined::Hshow() {
562  int n = 0;
563  int sz = m_history.size();
564  Slined * new_ed = the_editor_c;
565  std::array<char, 4> p{};
566  p[2] = ' ';
567  p[3] = 0;
568  new_ed->m_size = m_size + m_prompt.size() - 3;
569  for (int i = 0; i < sz; i++) {
570  n++;
571  if (n >= 100) n = n - 100;
572  if (n < 10)
573  p[0] = ' ';
574  else
575  p[0] = '0' + (n / 10);
576  p[1] = '0' + (n % 10);
577  new_ed->m_prompt = p.data();
578  new_ed->m_inpos = 0;
579  String S = m_history[i].c_str();
580  int k = strlen(S);
581  if (k > buf_size - 2) k = buf_size - 2;
582  new_ed->m_inmax = k;
583  strncpy(new_ed->m_inbuf, S, k);
584  new_ed->m_inbuf[k] = 0;
585  new_ed->redisplay();
586  std::cerr.put('\n');
587  }
588 }

600 // Find a string in the history
601 auto Slined::Hfind(int n) -> bool {
602  String s = m_search.c_str();
603  bool seen_once = false; // if seen at least once, do not kill
604  int pos = m_hpos;
605  while (pos > 0) {
606  pos--;
607  if (strstr(m_history[pos].c_str(), s) != nullptr) {
608  seen_once = true;
609  n--;
610  if (n == 0) {
611  m_hpos = pos;
612  return true;
613  }
614  }
615  }
616  tybeep();
617  if (!seen_once) m_search = "";
618  return false;
619 }

683 void Slined::forword(bool sw, int n) {
684  int i = find_word_beg(m_inbuf, m_inmax);
685  int y = m_inpos;

688  if (k >= i)
689  y = m_inmax;
690  else
691  y = word_beg[k]; // y is here boundary
692  k = k + n - 1;

699  else if (k > y) // do nothing if k=0
