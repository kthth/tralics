71 // Finds the first M, not escaped by E in the list L.
72 // If so, puts the prefix in z, removes it from L
73 // A pair EX is replaced by X in Z if S is false
74 auto token_ns::split_at(Token e, Token m, Token m1, TokenList &L, TokenList &z, bool s) -> bool {
75  z.clear();
76  while (!L.empty()) {
77  Token x = L.front();
78  L.pop_front();
79  if (x == e) {
80  if (s) z.push_back(x);
81  if (L.empty()) break;
82  x = L.front();
83  L.pop_front();
84  z.push_back(x);
85  } else if (x == m)
86  return true;
87  else if (x == m1)
88  return true;
89  else
90  z.push_back(x);
91  }
92  swap(L, z);
93  return false;
94 }

96 // Checks that x is a brace, increases or decreases brace level
97 // returns true if this is the toplevel closing brace
98 auto token_ns::check_brace(Token x, int &bl) -> bool {
99  if (x.is_a_brace()) {
100  if (x.is_a_left_brace()) {
101  bl++;
102  } else {
103  bl--;
104  if (bl == 0) return true;
105  }
106  }
107  return false;
108 }

110 // Replace x1 by x2 at toplevel in the list a
111 void token_ns::replace(TokenList &A, Token x1, Token x2) {
112  auto C = A.begin();
113  auto E = A.end();
114  int bl = 0;
115  while (C != E) {
116  Token x = *C;
117  check_brace(x, bl);
118  if (bl == 0 && x == x1) *C = x2;
119  ++C;
120  }

123 // Replace space by x2,x3 at brace-level zero.
124 // Two consecutive space tokens are replaced by
125 // a single occurence.
126 auto token_ns::replace_space(TokenList &A, Token x2, Token x3) -> int {
127  remove_first_last_space(A);
128  auto C = A.begin();
129  auto E = A.end();
130  int bl = 0;
131  int n = 0;
132  bool prev_is_space = false;
133  while (C != E) {
134  Token x = *C;
135  check_brace(x, bl);
136  if (bl == 0 && x.is_space_token()) {
137  if (!prev_is_space) {
138  A.insert(C, x2);
139  *C = x3;
140  ++n;
141  }
142  prev_is_space = true;
143  } else
144  prev_is_space = false;
145  ++C;
146  }
147  return n;
148 }

161 // Given a string s, with hash code p, we have p<hash_prime. In the good case,
162 // thhe string is at location p, otherwise at next(p), next(next(p)), etc.
163 // The idea is to use locations after hash prime. This means that all strings in
164 // the list has hash code p. If there is not enough room, we may use a location q
165 // less then hash prime. This means that, when looking for s string s' with
166 // hash code q we may encourer strings with hash code p.
167 // Note: initially Next[p] is zero; if non-zero the Text[Next[p]] is non-empty.

169 // Finds an empty slot in the hash table; fills it with name s.
170 // Return value is hash_used
171 // The string s must be a permanent string
172 auto Hashtab::find_empty(String s) -> int {
173  for (;;) {
174  hash_used--;
175  if (hash_used <= 0) {
176  log_and_tty << "Size of hash table is " << hash_size << "\n";
177  log_and_tty << "Value of hash_prime is " << hash_prime << "\n";
178  log_and_tty << "hash table full\n" << lg_fatal;
179  abort();
180  }
181  if (Text[hash_used] == nullptr) break;
182  }
183  Text[hash_used] = s;
184  hash_bad++;
185  return hash_used;
186 }

196 // Returns the hash location of the name in the buffer.
197 // If a new slot has to be created, uses the string name, if not empty.
198 // The string name must be a permanent string
199 auto Hashtab::hash_find(const Buffer &b, String name) -> int {
200  int p = b.hashcode(hash_prime);
201  for (;;) {
202  if ((Text[p] != nullptr) && b == Text[p]) return p;
203  if (Next[p] != 0)
204  p = Next[p];
205  else
206  break;
207  }
208  if (name == nullptr) name = b.convert_to_str();
209  return find_aux(p, name);
210 }

212 // Finds the object in the buffer B.
213 auto Hashtab::hash_find() -> int {
214  int p = B.hashcode(hash_prime);
215  for (;;) {
216  if ((Text[p] != nullptr) && B == Text[p]) return p;
217  if (Next[p] != 0)
218  p = Next[p];
219  else
220  break;
221  }
222  return find_aux(p, B.convert_to_str());
223 }

225 // This inserts name in the hash table.
226 // If Text[p] is empty, then p is not Next[q] so p is the hash code of s
227 // non empty, use this position. Otherwise find an empty position,
228 // and set Next[p] to this position.
229 auto Hashtab::find_aux(int p, String name) -> int {
230  if (Text[p] != nullptr) {
231  int q = find_empty(name);
232  Next[p] = q;
233  return q;
234  }
235  hash_usage++;
236  Text[p] = name;
237  return p;
238 }

240 // Defines the command named a, but hash_find will not find it.
241 // The string a must be a permanent string
242 // This must be used at bootstrap code.
243 auto Hashtab::nohash_primitive(String a, CmdChr b) -> Token {
244  hash_used--;
245  int p = hash_used;
246  if ((Text[p] != nullptr) || p < hash_prime) {
247  log_and_tty << "Size of hash table is " << hash_size << "\n";
248  log_and_tty << "Value of hash_prime is " << hash_prime << "\n";
249  log_and_tty << "Current position is " << p << "\n";
250  log_and_tty << "Bug in nohash_primitive\n" << lg_fatal;
251  abort();
252  }
253  hash_bad++;
254  Text[p] = a;
255  int t = p + hash_offset;
256  eqtb[t - eqtb_offset].special_prim(b); // allows to define an undefined command
257  return Token(t);
258 }

260 // Returns the hashcode of the string in the buffer (assumed zero-terminated).
261 auto Buffer::hashcode(int prime) const -> int {
262  int j = 1;
263  unsigned int h = (unsigned char)(at(0));
264  if (h == 0) return 0;
265  for (;;) {
266  unsigned char c = at(j);
267  if (c == 0) return h;
268  h = (h + h + c) % prime;
269  j++;
270  }
271 }

295 // This returns true if the token associated to the string in the buffer
296 // exists in the hash table and is not undefined.
297 // Sets last_tok to the result
298 auto Hashtab::is_defined(const Buffer &b) -> bool {
299  int T = 0;
300  if (b.size() == 0)
301  T = null_tok_val;
302  else {
303  codepoint c = b.unique_character();
304  if (c.non_null())
305  T = c.value + single_offset;
306  else {
307  int p = b.hashcode(hash_prime);
308  for (;;) {
309  if ((Text[p] != nullptr) && b == Text[p]) break;
310  if (Next[p] != 0)
311  p = Next[p];
312  else
313  return false;
314  }
315  T = p + hash_offset;
316  }
317  }
318  last_tok = Token(T);
319  return !eqtb[T - eqtb_offset].is_undefined();
320 }

355 auto token_ns::compare(const TokenList &A, const TokenList &B) -> bool {
356  auto C1 = A.begin();
357  auto E1 = A.end();
358  auto C2 = B.begin();
359  auto E2 = B.end();
360  for (;;) {
361  if (C1 == E1 || C2 == E2) return C1 == E1 && C2 == E2;
362  if (!(*C1).is_same_token(*C2)) return false;
363  ++C1;
364  ++C2;
365  }

378 // Removes the external braces in {foo}, but not in {foo}{bar}.
379 void token_ns::remove_ext_braces(TokenList &L) {
380  if (L.empty()) return;
381  if (!L.front().is_OB_token()) return;
382  if (!L.back().is_CB_token()) return;
383  auto C = L.begin();
384  auto E = L.end();
385  ++C;
386  --E;
387  int b = 0;
388  while (C != E) {
389  if (C->is_OB_token())
390  b++;
391  else if (C->is_CB_token()) {
392  b--;
393  if (b < 0) return;
394  }
395  ++C;
396  }
397  if (b != 0) return; // should not happen.
398  L.pop_front();
399  L.pop_back();
400 }

409 // If the token list is `{foo}bar{gee}'
410 // returns foo, leaves `bar{gee}' in the list
411 auto token_ns::get_block(TokenList &L) -> TokenList {
412  int bl = 0;
413  TokenList res;
414  remove_initial_spaces(L);
415  while (!L.empty()) {
416  Token t = L.front();
417  L.pop_front();
418  res.push_back(t);
419  if (check_brace(t, bl)) {
420  // Here we have {foo} in res
421  remove_ext_braces(res);
422  return res;
423  }
424  }
425  return TokenList();
426 }

428 // Assumes that the list starts with a brace.
429 // Returns the number of tokens in sublist with its braces.
430 // If the sublist is everything, returns -1.
431 // in case of problem, returns -2.
432 auto token_ns::block_size(const TokenList &L) -> int {
433  int res = 0;
434  int bl = 0;
435  auto C = L.begin();
436  auto E = L.end();
437  while (C != E) {
438  Token t = *C;
439  ++C;
440  ++res;
441  if (check_brace(t, bl)) {
442  if (C == E) return -1;
443  return res;
444  }
445  }
446  return -2;
447 }

449 // Assumes that the list starts with a brace.
450 // Returns the sublist with its braces.
451 auto token_ns::fast_get_block(TokenList &L) -> TokenList {
452  int len = block_size(L);
453  TokenList res;
454  if (len == -2) {
455  L.clear();
456  return res;
457  }
458  if (len == -1) {
459  L.swap(res);
460  return res;
461  }
462  auto C = L.begin();
463  while (len > 0) {
464  len--;
465  ++C;
466  }
467  res.splice(res.begin(), L, L.begin(), C);
468  return res;
469 }

522 // Interprets *{3}{abc} as 3 copies of abc
523 // This is used for expanding the header of a table
524 void token_ns::expand_star(TokenList &L) {
525  TokenList res;
526  while (!L.empty()) {
527  Token t = L.front();
528  if (t.is_a_left_brace()) {
529  fast_get_block(L, res);
530  } else if (!t.is_star_token()) {
531  L.pop_front();
532  res.push_back(t);
533  } else {
534  L.pop_front();
535  TokenList u = fast_get_block(L);
536  TokenList v = fast_get_block(L);
537  remove_ext_braces(u);
538  remove_ext_braces(v);
539  int n = 0;
540  while (!u.empty()) {
541  Token q = u.front();
542  u.pop_front();
543  if (!q.is_digit_token()) break;
544  n = 10 * n + q.val_as_digit();
545  if (n > 1000000) break; // bug?
546  }
547  while (n > 0) {
548  TokenList w = v;
549  L.splice(L.begin(), w);
550  n--;
551  }
552  }
553  }
554  L.swap(res);
555 }

557 // Converts the string in the buffer into a token list.
558 // Everything is of \catcode 12, except space.
559 // If the switch is true, \n is converted to space, otherwise newline
560 
561 auto Buffer::str_toks(nl_to_tok nl) -> TokenList {
562  TokenList L;
563  auto SP = Token(space_token_val);
564  auto CR = Token(space_t_offset + '\n'); // behaves as space
565  auto NL = Token(other_t_offset + '\n'); // is ^^J
566  reset_ptr();
567  for (;;) {
568  if (at_eol()) return L;
569  codepoint c = next_utf8_char();
570  if (c == 0) {
571  } // ignore bad chars
572  else if (c == ' ')
573  L.push_back(SP);
574  else if (c == '\n')
575  L.push_back(nl == nlt_space ? SP : (nl == nlt_cr ? CR : NL));
576  else
577  L.push_back(Token(other_t_offset, c));
578  }
579 }
