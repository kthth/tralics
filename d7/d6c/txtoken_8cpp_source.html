72 // Finds the first M, not escaped by E in the list L.
73 // If so, puts the prefix in z, removes it from L
74 // A pair EX is replaced by X in Z if S is false
75 auto token_ns::split_at(Token e, Token m, Token m1, TokenList &L, TokenList &z, bool s) -> bool {
76  z.clear();
77  while (!L.empty()) {
78  Token x = L.front();
79  L.pop_front();
80  if (x == e) {
81  if (s) z.push_back(x);
82  if (L.empty()) break;
83  x = L.front();
84  L.pop_front();
85  z.push_back(x);
86  } else if (x == m)
87  return true;
88  else if (x == m1)
89  return true;
90  else
91  z.push_back(x);
92  }
93  swap(L, z);
94  return false;
95 }

98 // returns true if this is the toplevel closing brace
99 auto token_ns::check_brace(Token x, int &bl) -> bool {
100  if (x.is_a_brace()) {
101  if (x.is_a_left_brace()) {
102  bl++;
103  } else {
104  bl--;
105  if (bl == 0) return true;
106  }
107  }
108  return false;

112 void token_ns::replace(TokenList &A, Token x1, Token x2) {

124 // Replace space by x2,x3 at brace-level zero.
125 // Two consecutive space tokens are replaced by
126 // a single occurence.
127 auto token_ns::replace_space(TokenList &A, Token x2, Token x3) -> int {
128  remove_first_last_space(A);
129  auto C = A.begin();
130  auto E = A.end();
131  int bl = 0;
132  int n = 0;
133  bool prev_is_space = false;
134  while (C != E) {
135  Token x = *C;
136  check_brace(x, bl);
137  if (bl == 0 && x.is_space_token()) {
138  if (!prev_is_space) {
139  A.insert(C, x2);
140  *C = x3;
141  ++n;
142  }
143  prev_is_space = true;
144  } else
145  prev_is_space = false;
146  ++C;
147  }
148  return n;
149 }

162 // Given a string s, with hash code p, we have p<hash_prime. In the good case,
163 // thhe string is at location p, otherwise at next(p), next(next(p)), etc.
164 // The idea is to use locations after hash prime. This means that all strings in
165 // the list has hash code p. If there is not enough room, we may use a location q
166 // less then hash prime. This means that, when looking for s string s' with
167 // hash code q we may encourer strings with hash code p.
168 // Note: initially Next[p] is zero; if non-zero the Text[Next[p]] is non-empty.

170 // Finds an empty slot in the hash table; fills it with name s.
171 // Return value is hash_used
172 // The string s must be a permanent string
173 auto Hashtab::find_empty(String s) -> size_t {
174  for (;;) {
175  hash_used--;
176  if (hash_used <= 0) {
177  log_and_tty << "Size of hash table is " << hash_size << "\n";
178  log_and_tty << "Value of hash_prime is " << hash_prime << "\n";
179  log_and_tty << "hash table full\n" << lg_fatal;
180  abort();
181  }
182  if (Text[hash_used] == nullptr) break;
183  }
184  Text[hash_used] = s;
185  hash_bad++;
186  return hash_used;
187 }

197 // Returns the hash location of the name in the buffer.
198 // If a new slot has to be created, uses the string name, if not empty.
199 // The string name must be a permanent string
200 auto Hashtab::hash_find(const Buffer &b, String name) -> size_t {
201  auto p = b.hashcode(hash_prime);
202  for (;;) {
203  if ((Text[p] != nullptr) && (b == Text[p])) return p;
204  if (Next[p] != 0)
205  p = Next[p];
206  else
207  break;
208  }
209  if (name == nullptr) name = b.convert_to_str();
210  return find_aux(p, name);
211 }

213 // Finds the object in the buffer B.
214 auto Hashtab::hash_find() -> size_t {
215  auto p = B.hashcode(hash_prime);
216  for (;;) {
217  if ((Text[p] != nullptr) && (B == Text[p])) return p;
218  if (Next[p] != 0)
219  p = Next[p];
220  else
221  break;
222  }
223  return find_aux(p, B.convert_to_str());
224 }

226 // This inserts name in the hash table.
227 // If Text[p] is empty, then p is not Next[q] so p is the hash code of s
228 // non empty, use this position. Otherwise find an empty position,
229 // and set Next[p] to this position.
230 auto Hashtab::find_aux(size_t p, String name) -> size_t {
231  if (Text[p] != nullptr) {
232  auto q = find_empty(name);
233  Next[p] = q;
234  return q;
235  }
236  hash_usage++;
237  Text[p] = name;
238  return p;
239 }

241 // Defines the command named a, but hash_find will not find it.
242 // The string a must be a permanent string
243 // This must be used at bootstrap code.
244 auto Hashtab::nohash_primitive(String a, CmdChr b) -> Token {
245  hash_used--;
246  auto p = hash_used;
247  if ((Text[p] != nullptr) || p < hash_prime) {
248  log_and_tty << "Size of hash table is " << hash_size << "\n";
249  log_and_tty << "Value of hash_prime is " << hash_prime << "\n";
250  log_and_tty << "Current position is " << to_signed(p) << "\n";
251  log_and_tty << "Bug in nohash_primitive\n" << lg_fatal;
252  abort();
253  }
254  hash_bad++;
255  Text[p] = a;
256  auto t = p + hash_offset;
257  eqtb[t - eqtb_offset].special_prim(b); // allows to define an undefined command
258  return Token(t);

261 // Returns the hashcode of the string in the buffer (assumed zero-terminated).
262 auto Buffer::hashcode(size_t prime) const -> size_t {
263  size_t j = 1;
264  size_t h = static_cast<uchar>(at(0));
265  if (h == 0) return 0;
266  for (;;) {
267  size_t c = static_cast<uchar>(at(j));
268  if (c == 0) return h;
269  h = (h + h + c) % prime;
270  j++;
271  }
272 }

296 // This returns true if the token associated to the string in the buffer
297 // exists in the hash table and is not undefined.
298 // Sets last_tok to the result
299 auto Hashtab::is_defined(const Buffer &b) -> bool {
300  size_t T = 0;
301  if (b.size() == 0)
302  T = null_tok_val;
303  else {
304  codepoint c = b.unique_character();
305  if (c.non_null())
306  T = c.value + single_offset;
307  else {
308  auto p = b.hashcode(hash_prime);
309  for (;;) {
310  if ((Text[p] != nullptr) && b == Text[p]) break;
311  if (Next[p] != 0)
312  p = Next[p];
313  else
314  return false;
315  }
316  T = p + hash_offset;
317  }
318  }
319  last_tok = Token(T);
320  return !eqtb[T - eqtb_offset].is_undef();
321 }

357  auto C1 = A.begin();
358  auto E1 = A.end();

360  auto E2 = B.end();
361  for (;;) {

379 // Removes the external braces in {foo}, but not in {foo}{bar}.
380 void token_ns::remove_ext_braces(TokenList &L) {
381  if (L.empty()) return;
382  if (!L.front().is_OB_token()) return;
383  if (!L.back().is_CB_token()) return;
384  auto C = L.begin();
385  auto E = L.end();
386  ++C;
387  --E;
388  int b = 0;
389  while (C != E) {
390  if (C->is_OB_token())
391  b++;
392  else if (C->is_CB_token()) {
393  b--;
394  if (b < 0) return;
395  }
396  ++C;
397  }
398  if (b != 0) return; // should not happen.
399  L.pop_front();
400  L.pop_back();
401 }

411 // returns foo, leaves `bar{gee}' in the list
412 auto token_ns::get_block(TokenList &L) -> TokenList {
413  int bl = 0;
414  TokenList res;
415  remove_initial_spaces(L);
416  while (!L.empty()) {
417  Token t = L.front();
418  L.pop_front();

423  return res;

430 // Returns the number of tokens in sublist with its braces.
431 // If the sublist is everything, returns -1.
432 // in case of problem, returns -2.
433 auto token_ns::block_size(const TokenList &L) -> int {

450 // Assumes that the list starts with a brace.
451 // Returns the sublist with its braces.
452 auto token_ns::fast_get_block(TokenList &L) -> TokenList {
453  int len = block_size(L);
454  TokenList res;
455  if (len == -2) {
456  L.clear();
457  return res;
458  }
459  if (len == -1) {
460  L.swap(res);
461  return res;
462  }
463  auto C = L.begin();
464  while (len > 0) {
465  len--;
466  ++C;
467  }
468  res.splice(res.begin(), L, L.begin(), C);
469  return res;

523 // Interprets *{3}{abc} as 3 copies of abc
524 // This is used for expanding the header of a table
525 void token_ns::expand_star(TokenList &L) {
526  TokenList res;
527  while (!L.empty()) {
528  Token t = L.front();
529  if (t.is_a_left_brace()) {
530  fast_get_block(L, res);
531  } else if (!t.is_star_token()) {
532  L.pop_front();
533  res.push_back(t);
534  } else {
535  L.pop_front();
536  TokenList u = fast_get_block(L);
537  TokenList v = fast_get_block(L);
538  remove_ext_braces(u);
539  remove_ext_braces(v);
540  size_t n = 0;
541  while (!u.empty()) {
542  Token q = u.front();
543  u.pop_front();
544  if (!q.is_digit_token()) break;
545  n = 10 * n + q.val_as_digit();
546  if (n > 1000000) break; // bug?
547  }
548  while (n > 0) {
549  TokenList w = v;
550  L.splice(L.begin(), w);
551  n--;
552  }
553  }
554  }
555  L.swap(res);
556 }

576  L.push_back(nl == nlt_space ? SP : (nl == nlt_cr ? CR : NL));
577  else
578  L.push_back(Token(other_t_offset, c));

643 auto operator<<(std::ostream &fp, const TokenList &L) -> std::ostream & {
